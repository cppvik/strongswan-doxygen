150 static int iterenv_cb(void *null, int num, char **key, char **value)
151 {
152  private_fast_request_t *this;
153 
154  *key = NULL;
155  *value = NULL;
156  this = (private_fast_request_t*)thread_this->get(thread_this);
157 
158  if (num < this->req_env_len)
159  {
160  char *eq;
161 
162  eq = strchr(this->req.envp[num], '=');
163  if (eq)
164  {
165  *key = strndup(this->req.envp[num], eq - this->req.envp[num]);
166  *value = strdup(eq + 1);
167  }
168  if (*key == NULL || *value == NULL)
169  {
170  free(*key);
171  free(*value);
172  return 1;
173  }
174  }
175  return 0;
176 }

293 METHOD(fast_request_t, sendfile, bool,
294  private_fast_request_t *this, char *path, char *mime)
295 {
296  chunk_t *data;
297  int written;
298  char buf[24];
299 
300  data = chunk_map(path, FALSE);
301  if (!data)
302  {
303  return FALSE;
304  }
305  /* FCGX does not like large integers, print to a buffer using libc */
306  snprintf(buf, sizeof(buf), "%lld", (int64_t)data->len);
307  FCGX_FPrintF(this->req.out, "Content-Length: %s\n", buf);
308  if (mime)
309  {
310  FCGX_FPrintF(this->req.out, "Content-Type: %s\n", mime);
311  }
312  FCGX_FPrintF(this->req.out, "\n");
313 
314  while (data->len)
315  {
316  written = FCGX_PutStr(data->ptr, data->len, this->req.out);
317  if (written == -1)
318  {
319  chunk_unmap(data);
320  return FALSE;
321  }
322  *data = chunk_skip(*data, written);
323  }
324 
325  chunk_unmap(data);
326  return TRUE;
327 }

444  if (FCGX_InitRequest(&this->req, fd, 0) != 0 ||

461  }

467  hdf_set_value(this->hdf, "Config.NoCache", "true");
468  if (!debug)

472  hdf_set_value(this->hdf, "Config.WhiteSpaceStrip", "2");
