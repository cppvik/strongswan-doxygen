45 static bool p_hash(prf_t *prf, char *label, chunk_t seed, size_t block_size,
46  size_t bytes, char *out)
47 {
48  char buf[block_size], abuf[block_size];
49  chunk_t a;
50 
51  /* seed = label + seed */
52  seed = chunk_cata("cc", chunk_create(label, strlen(label)), seed);
53  /* A(0) = seed */
54  a = seed;
55 
56  while (TRUE)
57  {
58  /* A(i) = HMAC_hash(secret, A(i-1)) */
59  if (!prf->get_bytes(prf, a, abuf))
60  {
61  return FALSE;
62  }
63  a = chunk_from_thing(abuf);
64  /* HMAC_hash(secret, A(i) + seed) */
65  if (!prf->get_bytes(prf, a, NULL) ||
66  !prf->get_bytes(prf, seed, buf))
67  {
68  return FALSE;
69  }
70 
71  if (bytes <= block_size)
72  {
73  memcpy(out, buf, bytes);
74  break;
75  }
76  memcpy(out, buf, block_size);
77  out += block_size;
78  bytes -= block_size;
79  }
80  return TRUE;
81 }
