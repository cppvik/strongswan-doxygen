102 METHOD(ntru_public_key_t, encrypt, bool,
103  private_ntru_public_key_t *this, chunk_t plaintext, chunk_t *ciphertext)
104 {
105  hash_algorithm_t hash_algid;
106  size_t t_len, seed1_len, seed2_len;
107  uint16_t *t1, *t = NULL;
108  uint8_t b[MAX_SEC_STRENGTH_LEN];
109  uint8_t *t2, *Mtrin, *M, *mask_trits, *ptr;
110  uint16_t mod_q_mask, mprime_len = 0;
111  int16_t m1 = 0;
112  chunk_t seed = chunk_empty;
113  ntru_trits_t *mask;
114  ntru_poly_t *r_poly;
115  bool msg_rep_good, success = FALSE;
116  int i;
117 
118  *ciphertext = chunk_empty;
119 
120  if (plaintext.len > this->params->m_len_max)
121  {
122  DBG1(DBG_LIB, "plaintext exceeds maximum size");
123  return FALSE;
124  }
125 
126  if (this->params->sec_strength_len > MAX_SEC_STRENGTH_LEN)
127  {
128  DBG1(DBG_LIB, "required security strength exceeds %d bits",
129  MAX_SEC_STRENGTH_LEN * BITS_PER_BYTE);
130  return FALSE;
131  }
132 
133  /* allocate temporary array t */
134  t_len = (sizeof(uint16_t) + 3*sizeof(uint8_t)) * this->params->N;
135  t = malloc(t_len);
136  t1 = t;
137  t2 = (uint8_t *)(t1 + this->params->N);
138  Mtrin = t2 + this->params->N;
139  M = Mtrin + this->params->N;
140 
141  /* set hash algorithm based on security strength */
142  hash_algid = (this->params->sec_strength_len <= 20) ? HASH_SHA1 :
143  HASH_SHA256;
144  /* set constants */
145  mod_q_mask = this->params->q - 1;
146 
147  /* allocate memory for the larger of the two seeds */
148  seed1_len = (this->params->N + 3)/4;
149  seed2_len = 3 + 2*this->params->sec_strength_len + plaintext.len;
150  seed = chunk_alloc(max(seed1_len, seed2_len));
151 
152  /* loop until a message representative with proper weight is achieved */
153  do
154  {
155  if (!this->drbg->generate(this->drbg,
156  this->params->sec_strength_len * BITS_PER_BYTE,
157  this->params->sec_strength_len, b))
158  {
159  goto err;
160  }
161 
162  /* form sData (OID || m || b || hTrunc) */
163  ptr = seed.ptr;
164  memcpy(ptr, this->params->oid, NTRU_OID_LEN);
165  ptr += NTRU_OID_LEN;
166  memcpy(ptr, plaintext.ptr, plaintext.len);
167  ptr += plaintext.len;
168  memcpy(ptr, b, this->params->sec_strength_len);
169  ptr += this->params->sec_strength_len;
170  memcpy(ptr, this->encoding.ptr + 2 + NTRU_OID_LEN,
171  this->params->sec_strength_len);
172  ptr += this->params->sec_strength_len;
173  seed.len = seed2_len;
174 
175  DBG2(DBG_LIB, "generate polynomial r");
176  r_poly = ntru_poly_create_from_seed(hash_algid, seed, this->params->c_bits,
177  this->params->N, this->params->q,
178  this->params->dF_r, this->params->dF_r,
179  this->params->is_product_form);
180  if (!r_poly)
181  {
182  goto err;
183  }
184 
185  /* form R = h * r */
186  r_poly->ring_mult(r_poly, this->pubkey, t1);
187  r_poly->destroy(r_poly);
188 
189  /* form R mod 4 */
190  ntru_coeffs_mod4_2_octets(this->params->N, t1, seed.ptr);
191  seed.len = seed1_len;
192 
193  /* form mask */
194  mask = ntru_trits_create(this->params->N, hash_algid, seed);
195  if (!mask)
196  {
197  DBG1(DBG_LIB, "mask creation failed");
198  goto err;
199  }
200 
201  /* form the padded message M */
202  ptr = M;
203  memcpy(ptr, b, this->params->sec_strength_len);
204  ptr += this->params->sec_strength_len;
205  if (this->params->m_len_len == 2)
206  {
207  *ptr++ = (uint8_t)((plaintext.len >> 8) & 0xff);
208  }
209  *ptr++ = (uint8_t)(plaintext.len & 0xff);
210  memcpy(ptr, plaintext.ptr, plaintext.len);
211  ptr += plaintext.len;
212 
213  /* add an extra zero byte in case without it the bit string
214  * is not a multiple of 3 bits and therefore might not be
215  * able to produce enough trits
216  */
217  memset(ptr, 0, this->params->m_len_max - plaintext.len + 2);
218 
219  /* convert M to trits (Mbin to Mtrin) */
220  mprime_len = this->params->N;
221  if (this->params->is_product_form)
222  {
223  --mprime_len;
224  }
225  ntru_bits_2_trits(M, mprime_len, Mtrin);
226  mask_trits = mask->get_trits(mask);
227 
228 
229  /* form the msg representative m' by adding Mtrin to mask, mod p */
230  if (this->params->is_product_form)
231  {
232  m1 = 0;
233  for (i = 0; i < mprime_len; i++)
234  {
235  t2[i] = mask_trits[i] + Mtrin[i];
236  if (t2[i] >= 3)
237  {
238  t2[i] -= 3;
239  }
240  if (t2[i] == 1)
241  {
242  ++m1;
243  }
244  else if (t2[i] == 2)
245  {
246  --m1;
247  }
248  }
249  }
250  else
251  {
252  for (i = 0; i < mprime_len; i++)
253  {
254  t2[i] = mask_trits[i] + Mtrin[i];
255  if (t2[i] >= 3)
256  {
257  t2[i] -= 3;
258  }
259  }
260  }
261  mask->destroy(mask);
262 
263  /* check that message representative meets minimum weight
264  * requirements
265  */
266  if (this->params->is_product_form)
267  {
268  msg_rep_good = (abs(m1) <= this->params->min_msg_rep_wt);
269  }
270  else
271  {
272  msg_rep_good = ntru_check_min_weight(mprime_len, t2,
273  this->params->min_msg_rep_wt);
274  }
275  }
276  while (!msg_rep_good);
277 
278  /* form ciphertext e by adding m' to R mod q */
279  for (i = 0; i < mprime_len; i++)
280  {
281  if (t2[i] == 1)
282  {
283  t1[i] = (t1[i] + 1) & mod_q_mask;
284  }
285  else if (t2[i] == 2)
286  {
287  t1[i] = (t1[i] - 1) & mod_q_mask;
288  }
289  }
290  if (this->params->is_product_form)
291  {
292  t1[i] = (t1[i] - m1) & mod_q_mask;
293  }
294 
295  /* pack ciphertext */
296  *ciphertext = chunk_alloc((this->params->N * this->params->q_bits + 7) / 8);
297  ntru_elements_2_octets(this->params->N, t1, this->params->q_bits,
298  ciphertext->ptr);
299 
300  memwipe(t, t_len);
301  success = TRUE;
302 
303 err:
304  /* cleanup */
305  chunk_clear(&seed);
306  free(t);
307 
308  return success;
309 }

367  data.ptr[1] != NTRU_OID_LEN)
368  {
369  DBG1(DBG_LIB, "received NTRU public key with invalid header");

371  }
372  params = ntru_param_set_get_by_oid(data.ptr + 2);
373 
374  if (!params)
375  {
376  DBG1(DBG_LIB, "received NTRU public key with unknown OID");
377  return NULL;
378  }
379 
380  pubkey_packed_len = (params->N * params->q_bits + 7) / 8;
381 
382  if (data.len < header_len + pubkey_packed_len)
383  {
384  DBG1(DBG_LIB, "received NTRU public key with wrong packed key size");
385  return NULL;
386  }
387 
388  INIT(this,
389  .public = {
390  .get_id = _get_id,
391  .get_encoding = _get_encoding,
392  .encrypt = _encrypt,
393  .destroy = _destroy,
394  },
395  .params = params,
396  .pubkey = malloc(params->N * sizeof(uint16_t)),
397  .encoding = chunk_clone(data),
398  .drbg = drbg->get_ref(drbg),
399  );
400 
401  /* unpack the encoded public key */
402  ntru_octets_2_elements(pubkey_packed_len, data.ptr + header_len,
403  params->q_bits, this->pubkey);
