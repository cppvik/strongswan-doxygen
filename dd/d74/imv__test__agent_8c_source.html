85 static TNC_Result receive_msg(private_imv_test_agent_t *this, imv_state_t *state,
86  imv_msg_t *in_msg)
87 {
88  imv_msg_t *out_msg;
89  imv_test_state_t *test_state;
90  enumerator_t *enumerator;
91  pa_tnc_attr_t *attr;
92  pen_type_t attr_type;
93  TNC_Result result;
94  int rounds;
95  bool fatal_error = FALSE, received_command = FALSE, retry = FALSE;
96 
97  /* generate an outgoing PA-TNC message - we might need it */
98  out_msg = imv_msg_create_as_reply(in_msg);
99 
100  /* parse received PA-TNC message and handle local and remote errors */
101  result = in_msg->receive(in_msg, out_msg, &fatal_error);
102  if (result != TNC_RESULT_SUCCESS)
103  {
104  out_msg->destroy(out_msg);
105  return result;
106  }
107 
108  /* add any new IMC and set its number of rounds */
109  rounds = lib->settings->get_int(lib->settings,
110  "%s.plugins.imv-test.rounds", 0, lib->ns);
111  test_state = (imv_test_state_t*)state;
112  test_state->add_imc(test_state, in_msg->get_src_id(in_msg), rounds);
113 
114  /* analyze PA-TNC attributes */
115  enumerator = in_msg->create_attribute_enumerator(in_msg);
116  while (enumerator->enumerate(enumerator, &attr))
117  {
118  attr_type = attr->get_type(attr);
119 
120  if (attr_type.vendor_id != PEN_ITA)
121  {
122  continue;
123  }
124  if (attr_type.type == ITA_ATTR_COMMAND)
125  {
126  ita_attr_command_t *ita_attr;
127  char *command;
128 
129  received_command = TRUE;
130  ita_attr = (ita_attr_command_t*)attr;
131  command = ita_attr->get_command(ita_attr);
132 
133  if (streq(command, "allow"))
134  {
135  state->set_recommendation(state,
136  TNC_IMV_ACTION_RECOMMENDATION_ALLOW,
137  TNC_IMV_EVALUATION_RESULT_COMPLIANT);
138  }
139  else if (streq(command, "isolate"))
140  {
141  state->set_recommendation(state,
142  TNC_IMV_ACTION_RECOMMENDATION_ISOLATE,
143  TNC_IMV_EVALUATION_RESULT_NONCOMPLIANT_MINOR);
144  }
145  else if (streq(command, "block") || streq(command, "none"))
146  {
147  state->set_recommendation(state,
148  TNC_IMV_ACTION_RECOMMENDATION_NO_ACCESS,
149  TNC_IMV_EVALUATION_RESULT_NONCOMPLIANT_MAJOR);
150  }
151  else if (streq(command, "retry"))
152  {
153  retry = TRUE;
154  }
155  else
156  {
157  DBG1(DBG_IMV, "unsupported ITA Command '%s'", command);
158  state->set_recommendation(state,
159  TNC_IMV_ACTION_RECOMMENDATION_NO_RECOMMENDATION,
160  TNC_IMV_EVALUATION_RESULT_ERROR);
161  }
162  }
163  else if (attr_type.type == ITA_ATTR_DUMMY)
164  {
165  ita_attr_dummy_t *ita_attr;
166 
167  ita_attr = (ita_attr_dummy_t*)attr;
168  DBG1(DBG_IMV, "received dummy attribute value (%d bytes)",
169  ita_attr->get_size(ita_attr));
170  }
171  }
172  enumerator->destroy(enumerator);
173 
174  if (fatal_error)
175  {
176  state->set_recommendation(state,
177  TNC_IMV_ACTION_RECOMMENDATION_NO_RECOMMENDATION,
178  TNC_IMV_EVALUATION_RESULT_ERROR);
179  result = out_msg->send_assessment(out_msg);
180  if (result == TNC_RESULT_SUCCESS)
181  {
182  result = this->agent->provide_recommendation(this->agent, state);
183  }
184  return result;
185  }
186 
187  /* request a handshake retry ? */
188  if (retry)
189  {
190  test_state->set_rounds(test_state, rounds);
191  return this->agent->request_handshake_retry(
192  this->agent->get_id(this->agent),
193  state->get_connection_id(state),
194  TNC_RETRY_REASON_IMV_SERIOUS_EVENT);
195  }
196 
197  /* repeat the measurement ? */
198  if (test_state->another_round(test_state, in_msg->get_src_id(in_msg)))
199  {
200  attr = ita_attr_command_create("repeat");
201  out_msg->add_attribute(out_msg, attr);
202 
203  /* send PA-TNC message with excl flag set */
204  result = out_msg->send(out_msg, TRUE);
205  out_msg->destroy(out_msg);
206 
207  return result;
208  }
209 
210  if (received_command)
211  {
212  result = out_msg->send_assessment(out_msg);
213  if (result == TNC_RESULT_SUCCESS)
214  {
215  result = this->agent->provide_recommendation(this->agent, state);
216  }
217  }
218  else
219  {
220  /* send PA-TNC message with the EXCL flag set */
221  result = out_msg->send(out_msg, TRUE);
222  }
223  out_msg->destroy(out_msg);
224 
225  return result;
226  }

228 METHOD(imv_agent_if_t, receive_message, TNC_Result,
229  private_imv_test_agent_t *this, TNC_ConnectionID id,
230  TNC_MessageType msg_type, chunk_t msg)
231 {

243
