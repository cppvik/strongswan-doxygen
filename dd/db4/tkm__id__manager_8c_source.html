85  this->locks[kind]->write_lock(this->locks[kind]);
86  for (j = 0; j < this->limits[kind]; j++)

94  }
95  this->locks[kind]->unlock(this->locks[kind]);
96 
97  if (!id)

101  }
102

104 }
