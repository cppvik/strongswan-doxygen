140 plugin_t *gcrypt_plugin_create()
141 {
142  private_gcrypt_plugin_t *this;
143 
144  gcry_control(GCRYCTL_SET_THREAD_CBS, &gcry_threads_pthread);
145 
146  if (!gcry_check_version(GCRYPT_VERSION))
147  {
148  DBG1(DBG_LIB, "libgcrypt version mismatch");
149  return NULL;
150  }
151 
152  /* we currently do not use secure memory */
153  gcry_control(GCRYCTL_DISABLE_SECMEM, 0);
154  if (lib->settings->get_bool(lib->settings, "%s.plugins.gcrypt.quick_random",
155  FALSE, lib->ns))
156  {
157  gcry_control(GCRYCTL_ENABLE_QUICK_RANDOM, 0);
158  }
159  gcry_control(GCRYCTL_INITIALIZATION_FINISHED, 0);
160 
161  /* initialize static allocations we want to exclude from leak-detective */
162  gcry_create_nonce(NULL, 0);
163 
164  INIT(this,
165  .public = {
166  .plugin = {
167  .get_name = _get_name,
168  .get_features = _get_features,
169  .destroy = _destroy,
170  },
171  },
172  );
173 
174  return &this->public.plugin;
175 }
