220 static bool parse(private_x509_crl_t *this)
221 {
222  asn1_parser_t *parser;
223  chunk_t object;
224  chunk_t extnID = chunk_empty;
225  chunk_t userCertificate = chunk_empty;
226  int objectID;
227  int sig_alg = OID_UNKNOWN;
228  bool success = FALSE;
229  bool critical = FALSE;
230  revoked_t *revoked = NULL;
231 
232  parser = asn1_parser_create(crlObjects, this->encoding);
233 
234  while (parser->iterate(parser, &objectID, &object))
235  {
236  u_int level = parser->get_level(parser)+1;
237 
238  switch (objectID)
239  {
240  case CRL_OBJ_TBS_CERT_LIST:
241  this->tbsCertList = object;
242  break;
243  case CRL_OBJ_VERSION:
244  this->version = (object.len) ? (1+(u_int)*object.ptr) : 1;
245  DBG2(DBG_ASN, " v%d", this->version);
246  break;
247  case CRL_OBJ_SIG_ALG:
248  sig_alg = asn1_parse_algorithmIdentifier(object, level, NULL);
249  break;
250  case CRL_OBJ_ISSUER:
251  this->issuer = identification_create_from_encoding(ID_DER_ASN1_DN, object);
252  DBG2(DBG_ASN, " '%Y'", this->issuer);
253  break;
254  case CRL_OBJ_THIS_UPDATE:
255  this->thisUpdate = asn1_parse_time(object, level);
256  break;
257  case CRL_OBJ_NEXT_UPDATE:
258  this->nextUpdate = asn1_parse_time(object, level);
259  break;
260  case CRL_OBJ_USER_CERTIFICATE:
261  userCertificate = object;
262  break;
263  case CRL_OBJ_REVOCATION_DATE:
264  revoked = malloc_thing(revoked_t);
265  revoked->serial = chunk_clone(userCertificate);
266  revoked->date = asn1_parse_time(object, level);
267  revoked->reason = CRL_REASON_UNSPECIFIED;
268  this->revoked->insert_last(this->revoked, (void *)revoked);
269  break;
270  case CRL_OBJ_CRL_ENTRY_EXTN_ID:
271  case CRL_OBJ_EXTN_ID:
272  extnID = object;
273  break;
274  case CRL_OBJ_CRL_ENTRY_CRITICAL:
275  case CRL_OBJ_CRITICAL:
276  critical = object.len && *object.ptr;
277  DBG2(DBG_ASN, " %s", critical ? "TRUE" : "FALSE");
278  break;
279  case CRL_OBJ_CRL_ENTRY_EXTN_VALUE:
280  case CRL_OBJ_EXTN_VALUE:
281  {
282  int extn_oid = asn1_known_oid(extnID);
283 
284  switch (extn_oid)
285  {
286  case OID_CRL_REASON_CODE:
287  if (revoked)
288  {
289  if (object.len && *object.ptr == ASN1_ENUMERATED &&
290  asn1_length(&object) == 1)
291  {
292  revoked->reason = *object.ptr;
293  }
294  DBG2(DBG_ASN, " '%N'", crl_reason_names,
295  revoked->reason);
296  }
297  break;
298  case OID_AUTHORITY_KEY_ID:
299  this->authKeyIdentifier =
300  x509_parse_authorityKeyIdentifier(
301  object, level, &this->authKeySerialNumber);
302  break;
303  case OID_CRL_NUMBER:
304  if (!asn1_parse_simple_object(&object, ASN1_INTEGER,
305  level, "crlNumber"))
306  {
307  goto end;
308  }
309  this->crlNumber = object;
310  break;
311  case OID_FRESHEST_CRL:
312  x509_parse_crlDistributionPoints(object, level,
313  this->crl_uris);
314  break;
315  case OID_DELTA_CRL_INDICATOR:
316  if (!asn1_parse_simple_object(&object, ASN1_INTEGER,
317  level, "deltaCrlIndicator"))
318  {
319  goto end;
320  }
321  this->baseCrlNumber = object;
322  break;
323  case OID_ISSUING_DIST_POINT:
324  /* TODO support of IssuingDistributionPoints */
325  break;
326  default:
327  if (critical && lib->settings->get_bool(lib->settings,
328  "%s.x509.enforce_critical", TRUE, lib->ns))
329  {
330  DBG1(DBG_ASN, "critical '%s' extension not supported",
331  (extn_oid == OID_UNKNOWN) ? "unknown" :
332  (char*)oid_names[extn_oid].name);
333  goto end;
334  }
335  break;
336  }
337  break;
338  }
339  case CRL_OBJ_ALGORITHM:
340  {
341  this->algorithm = asn1_parse_algorithmIdentifier(object, level, NULL);
342  if (this->algorithm != sig_alg)
343  {
344  DBG1(DBG_ASN, " signature algorithms do not agree");
345  goto end;
346  }
347  break;
348  }
349  case CRL_OBJ_SIGNATURE:
350  this->signature = chunk_skip(object, 1);
351  break;
352  default:
353  break;
354  }
355  }
356  success = parser->success(parser);
357 
358 end:
359  parser->destroy(parser);
360  return success;
361 }

447 METHOD(certificate_t, issued_by, bool,
448  private_x509_crl_t *this, certificate_t *issuer, signature_scheme_t *schemep)
449 {
450  public_key_t *key;
451  signature_scheme_t scheme;
452  bool valid;
453  x509_t *x509 = (x509_t*)issuer;
454  chunk_t keyid = chunk_empty;
455 
456  /* check if issuer is an X.509 CA certificate */
457  if (issuer->get_type(issuer) != CERT_X509)
458  {
459  return FALSE;
460  }
461  if (!(x509->get_flags(x509) & (X509_CA | X509_CRL_SIGN)))
462  {
463  return FALSE;
464  }
465 
466  /* compare keyIdentifiers if available, otherwise use DNs */
467  if (this->authKeyIdentifier.ptr)
468  {
469  keyid = x509->get_subjectKeyIdentifier(x509);
470  if (keyid.len && !chunk_equals(keyid, this->authKeyIdentifier))
471  {
472  return FALSE;
473  }
474  }
475  if (!keyid.len)
476  {
477  if (!this->issuer->equals(this->issuer, issuer->get_subject(issuer)))
478  {
479  return FALSE;
480  }
481  }
482 
483  scheme = signature_scheme_from_oid(this->algorithm);
484  if (scheme == SIGN_UNKNOWN)
485  {
486  return FALSE;
487  }
488  key = issuer->get_public_key(issuer);
489  if (!key)
490  {
491  return FALSE;
492  }
493  valid = key->verify(key, scheme, this->tbsCertList, this->signature);
494  key->destroy(key);
495  if (valid && schemep)
496  {
497  *schemep = scheme;
498  }
499  return valid;
500 }

704 static bool generate(private_x509_crl_t *this, certificate_t *cert,
705  private_key_t *key, hash_algorithm_t digest_alg)
706 {
707  chunk_t extensions = chunk_empty, certList = chunk_empty, serial;
708  chunk_t crlDistributionPoints = chunk_empty, baseCrlNumber = chunk_empty;
709  enumerator_t *enumerator;
710  crl_reason_t reason;
711  time_t date;
712  x509_t *x509;
713 
714  x509 = (x509_t*)cert;
715 
716  this->issuer = cert->get_subject(cert);
717  this->issuer = this->issuer->clone(this->issuer);
718 
719  this->authKeyIdentifier = chunk_clone(x509->get_subjectKeyIdentifier(x509));
720 
721  /* select signature scheme */
722  this->algorithm = hasher_signature_algorithm_to_oid(digest_alg,
723  key->get_type(key));
724  if (this->algorithm == OID_UNKNOWN)
725  {
726  return FALSE;
727  }
728 
729  enumerator = create_enumerator(this);
730  while (enumerator->enumerate(enumerator, &serial, &date, &reason))
731  {
732  chunk_t revoked, entry_ext = chunk_empty;
733 
734  if (reason != CRL_REASON_UNSPECIFIED)
735  {
736  entry_ext = asn1_wrap(ASN1_SEQUENCE, "m",
737  asn1_wrap(ASN1_SEQUENCE, "mm",
738  asn1_build_known_oid(OID_CRL_REASON_CODE),
739  asn1_wrap(ASN1_OCTET_STRING, "m",
740  asn1_wrap(ASN1_ENUMERATED, "c",
741  chunk_from_chars(reason)))));
742  }
743  revoked = asn1_wrap(ASN1_SEQUENCE, "mmm",
744  asn1_integer("c", serial),
745  asn1_from_time(&date, ASN1_UTCTIME),
746  entry_ext);
747  certList = chunk_cat("mm", certList, revoked);
748  }
749  enumerator->destroy(enumerator);
750 
751  crlDistributionPoints = x509_build_crlDistributionPoints(this->crl_uris,
752  OID_FRESHEST_CRL);
753 
754  if (this->baseCrlNumber.len)
755  {
756  baseCrlNumber = asn1_wrap(ASN1_SEQUENCE, "mmm",
757  asn1_build_known_oid(OID_DELTA_CRL_INDICATOR),
758  asn1_wrap(ASN1_BOOLEAN, "c",
759  chunk_from_chars(0xFF)),
760  asn1_wrap(ASN1_OCTET_STRING, "m",
761  asn1_integer("c", this->baseCrlNumber)));
