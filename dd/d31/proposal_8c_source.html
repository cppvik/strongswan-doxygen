191 static bool select_algo(private_proposal_t *this, proposal_t *other,
192  proposal_t *selected, transform_type_t type, bool priv)
193 {
194  enumerator_t *e1, *e2;
195  u_int16_t alg1, alg2, ks1, ks2;
196  bool found = FALSE, optional = FALSE;
197 
198  if (type == INTEGRITY_ALGORITHM &&
199  selected->get_algorithm(selected, ENCRYPTION_ALGORITHM, &alg1, NULL) &&
200  encryption_algorithm_is_aead(alg1))
201  {
202  /* no integrity algorithm required, we have an AEAD */
203  return TRUE;
204  }
205  if (type == DIFFIE_HELLMAN_GROUP)
206  {
207  optional = this->protocol == PROTO_ESP || this->protocol == PROTO_AH;
208  }
209 
210  e1 = create_enumerator(this, type);
211  e2 = other->create_enumerator(other, type);
212  if (!e1->enumerate(e1, NULL, NULL))
213  {
214  if (!e2->enumerate(e2, &alg2, NULL))
215  {
216  found = TRUE;
217  }
218  else if (optional)
219  {
220  do
221  { /* if the other peer proposes NONE, we accept the proposal */
222  found = !alg2;
223  }
224  while (!found && e2->enumerate(e2, &alg2, NULL));
225  }
226  }
227 
228  e1->destroy(e1);
229  e1 = create_enumerator(this, type);
230  /* compare algs, order of algs in "first" is preferred */
231  while (!found && e1->enumerate(e1, &alg1, &ks1))
232  {
233  e2->destroy(e2);
234  e2 = other->create_enumerator(other, type);
235  while (e2->enumerate(e2, &alg2, &ks2))
236  {
237  if (alg1 == alg2 && ks1 == ks2)
238  {
239  if (!priv && alg1 >= 1024)
240  {
241  /* accept private use algorithms only if requested */
242  DBG1(DBG_CFG, "an algorithm from private space would match, "
243  "but peer implementation is unknown, skipped");
244  continue;
245  }
246  /* ok, we have an algorithm */
247  selected->add_algorithm(selected, type, alg1, ks1);
248  found = TRUE;
249  break;
250  }
251  }
252  }
253  /* no match in all comparisons */
254  e1->destroy(e1);
255  e2->destroy(e2);
256 
257  if (!found)
258  {
259  DBG2(DBG_CFG, " no acceptable %N found", transform_type_names, type);
260  }
261  return found;
262 }

265  private_proposal_t *this, proposal_t *other, bool private)
266 {

317 static bool algo_list_equals(private_proposal_t *this, proposal_t *other,
318  transform_type_t type)
319 {
320  enumerator_t *e1, *e2;
321  u_int16_t alg1, alg2, ks1, ks2;
322  bool equals = TRUE;
323 
324  e1 = create_enumerator(this, type);
325  e2 = other->create_enumerator(other, type);
326  while (e1->enumerate(e1, &alg1, &ks1))
327  {
328  if (!e2->enumerate(e2, &alg2, &ks2))
329  {
330  /* this has more algs */
331  equals = FALSE;
332  break;
333  }
334  if (alg1 != alg2 || ks1 != ks2)
335  {
336  equals = FALSE;
337  break;
338  }
339  }
340  if (e2->enumerate(e2, &alg2, &ks2))
341  {
342  /* other has more algs */
343  equals = FALSE;
344  }
345  e1->destroy(e1);
346  e2->destroy(e2);
347 
348  return equals;
349 }

416 static void check_proposal(private_proposal_t *this)
417 {
418  enumerator_t *e;
419  entry_t *entry;
420  u_int16_t alg, ks;
421  bool all_aead = TRUE;
422  int i;
423 
424  if (this->protocol == PROTO_IKE)
425  {
426  e = create_enumerator(this, PSEUDO_RANDOM_FUNCTION);
427  if (!e->enumerate(e, &alg, &ks))
428  {
429  /* No explicit PRF found. We assume the same algorithm as used
430  * for integrity checking */
431  e->destroy(e);
432  e = create_enumerator(this, INTEGRITY_ALGORITHM);
433  while (e->enumerate(e, &alg, &ks))
434  {
435  for (i = 0; i < countof(integ_prf_map); i++)
436  {
437  if (alg == integ_prf_map[i].integ)
438  {
439  add_algorithm(this, PSEUDO_RANDOM_FUNCTION,
440  integ_prf_map[i].prf, 0);
441  break;
442  }
443  }
444  }
445  }
446  e->destroy(e);
447  }
448 
449  if (this->protocol == PROTO_ESP)
450  {
451  e = create_enumerator(this, ENCRYPTION_ALGORITHM);
452  while (e->enumerate(e, &alg, &ks))
453  {
454  if (!encryption_algorithm_is_aead(alg))
455  {
456  all_aead = FALSE;
457  break;
458  }
459  }
460  e->destroy(e);
461 
462  if (all_aead)
463  {
464  /* if all encryption algorithms in the proposal are AEADs,
465  * we MUST NOT propose any integrity algorithms */
466  e = array_create_enumerator(this->transforms);
467  while (e->enumerate(e, &entry))
468  {
469  if (entry->type == INTEGRITY_ALGORITHM)
470  {
471  array_remove_at(this->transforms, e);
472  }
473  }
474  e->destroy(e);
475  }
476  }
477 
478  if (this->protocol == PROTO_AH || this->protocol == PROTO_ESP)
479  {
480  e = create_enumerator(this, EXTENDED_SEQUENCE_NUMBERS);
481  if (!e->enumerate(e, NULL, NULL))
482  { /* ESN not specified, assume not supported */
483  add_algorithm(this, EXTENDED_SEQUENCE_NUMBERS, NO_EXT_SEQ_NUMBERS, 0);
484  }
485  e->destroy(e);
486  }
487 
488  array_compress(this->transforms);
489 }

545  const void *const *args)
546 {
547  private_proposal_t *this = *((private_proposal_t**)(args[0]));

632 static bool proposal_add_supported_ike(private_proposal_t *this, bool aead)
633 {
634  enumerator_t *enumerator;
635  encryption_algorithm_t encryption;
636  integrity_algorithm_t integrity;
637  pseudo_random_function_t prf;
638  diffie_hellman_group_t group;
639  const char *plugin_name;
640 
641  if (aead)
642  {
643  enumerator = lib->crypto->create_aead_enumerator(lib->crypto);
644  while (enumerator->enumerate(enumerator, &encryption, &plugin_name))
645  {
646  switch (encryption)
647  {
648  case ENCR_AES_CCM_ICV8:
649  case ENCR_AES_CCM_ICV12:
650  case ENCR_AES_CCM_ICV16:
651  case ENCR_AES_GCM_ICV8:
652  case ENCR_AES_GCM_ICV12:
653  case ENCR_AES_GCM_ICV16:
654  case ENCR_CAMELLIA_CCM_ICV8:
655  case ENCR_CAMELLIA_CCM_ICV12:
656  case ENCR_CAMELLIA_CCM_ICV16:
657  /* we assume that we support all AES/Camellia sizes */
658  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 128);
659  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 192);
660  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 256);
661  break;
662  default:
663  break;
664  }
665  }
666  enumerator->destroy(enumerator);
667 
668  if (!array_count(this->transforms))
669  {
670  return FALSE;
671  }
672  }
673  else
674  {
675  enumerator = lib->crypto->create_crypter_enumerator(lib->crypto);
676  while (enumerator->enumerate(enumerator, &encryption, &plugin_name))
677  {
678  switch (encryption)
679  {
680  case ENCR_AES_CBC:
681  case ENCR_AES_CTR:
682  case ENCR_CAMELLIA_CBC:
683  case ENCR_CAMELLIA_CTR:
684  /* we assume that we support all AES/Camellia sizes */
685  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 128);
686  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 192);
687  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 256);
688  break;
689  case ENCR_3DES:
690  add_algorithm(this, ENCRYPTION_ALGORITHM, encryption, 0);
691  break;
692  case ENCR_DES:
693  /* no, thanks */
694  break;
695  default:
696  break;
697  }
698  }
699  enumerator->destroy(enumerator);
700 
701  if (!array_count(this->transforms))
702  {
703  return FALSE;
704  }
705 
706  enumerator = lib->crypto->create_signer_enumerator(lib->crypto);
707  while (enumerator->enumerate(enumerator, &integrity, &plugin_name))
708  {
709  switch (integrity)
710  {
711  case AUTH_HMAC_SHA1_96:
712  case AUTH_HMAC_SHA2_256_128:
713  case AUTH_HMAC_SHA2_384_192:
714  case AUTH_HMAC_SHA2_512_256:
715  case AUTH_HMAC_MD5_96:
716  case AUTH_AES_XCBC_96:
717  case AUTH_AES_CMAC_96:
718  add_algorithm(this, INTEGRITY_ALGORITHM, integrity, 0);
719  break;
720  default:
721  break;
722  }
723  }
724  enumerator->destroy(enumerator);
725  }
726 
727  enumerator = lib->crypto->create_prf_enumerator(lib->crypto);
728  while (enumerator->enumerate(enumerator, &prf, &plugin_name))
729  {
730  switch (prf)
731  {
732  case PRF_HMAC_SHA1:
733  case PRF_HMAC_SHA2_256:
734  case PRF_HMAC_SHA2_384:
735  case PRF_HMAC_SHA2_512:
736  case PRF_HMAC_MD5:
737  case PRF_AES128_XCBC:
738  case PRF_AES128_CMAC:
739  add_algorithm(this, PSEUDO_RANDOM_FUNCTION, prf, 0);
740  break;
741  default:
742  break;
743  }
744  }
745  enumerator->destroy(enumerator);
746 
747  enumerator = lib->crypto->create_dh_enumerator(lib->crypto);
748  while (enumerator->enumerate(enumerator, &group, &plugin_name))
749  {
750  switch (group)
751  {
752  case MODP_NULL:
753  /* only for testing purposes */
754  break;
755  case MODP_768_BIT:
