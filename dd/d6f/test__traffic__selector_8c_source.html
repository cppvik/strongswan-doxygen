24 static void verify(const char *str, const char *alt, traffic_selector_t *ts)
25 {
26  char buf[512];
27 
28  snprintf(buf, sizeof(buf), "%R", ts);
29  DESTROY_IF(ts);
30  if (!streq(buf, str) && (!alt || !streq(buf, alt)))
31  {
32  fail("%s != %s or %s", buf, str, alt);
33  }
34 }
