50 static pid_t invoke(void *null, guest_t *guest, char *args[], int argc)
51 {
52  pid_t pid;
53 
54  pid = fork();
55  switch (pid)
56  {
57  case 0: /* child */
58  /* create a new process group in order to prevent signals (e.g.
59  * SIGINT) sent to the parent from terminating the child */
60  setpgid(0, 0);
61  dup2(open("/dev/null", 0), 1);
62  dup2(open("/dev/null", 0), 2);
63  execvp(args[0], args);
64  /* FALL */
65  case -1:
66  return 0;
67  default:
68  return pid;
69  }
70 }
