104  struct in_addr *addr, struct in_addr *mask)

292 static bool manage_out(struct iptc_handle *ipth, entry_t *entry, bool add)
293 {
294  struct {
295  struct ipt_entry e;
296  struct ipt_entry_target t;
297  struct xt_mark_tginfo2 m;
298  } ipt = {
299  .e = {
300  .target_offset = XT_ALIGN(sizeof(ipt.e)),
301  .next_offset = sizeof(ipt),
302  },
303  .t = {
304  .u.user.target_size = XT_ALIGN(sizeof(ipt.t) + sizeof(ipt.m)),
305  .u.user.name = "MARK",
306  .u.user.revision = 2,
307  },
308  .m = {
309  .mark = entry->mark,
310  .mask = ~0,
311  },
312  };
313  enumerator_t *enumerator;
314  traffic_selector_t *ts;
315 
316  enumerator = array_create_enumerator(entry->rts);
317  while (enumerator->enumerate(enumerator, &ts))
318  {
319  if (!ts2in(ts, &ipt.e.ip.dst, &ipt.e.ip.dmsk))
320  {
321  continue;
322  }
323  if (ipt.e.ip.dst.s_addr == 0xffffffff ||
324  ipt.e.ip.dst.s_addr == entry->broadcast ||
325  memeq(&ipt.e.ip.dst.s_addr, "\xe0", 1))
326  {
327  /* skip broadcast/multicast selectors, they are shared and the mark
328  * is set by the socket we use for reinjection */
329  continue;
330  }
331  if (!manage_rule(ipth, "PREROUTING", add, &ipt.e) ||
332  !manage_rule(ipth, "OUTPUT", add, &ipt.e))
333  {
334  enumerator->destroy(enumerator);
335  return FALSE;
336  }
337  }
338  enumerator->destroy(enumerator);
339 
340  return TRUE;
341 }

369 static bool add_entry(private_forecast_listener_t *this,
370  struct iptc_handle *ipth, host_t *lhost, host_t *rhost,
371  child_sa_t *child_sa, bool encap)
372 {
373  enumerator_t *enumerator;
374  traffic_selector_t *ts;
375  entry_t *entry;
376 
377  INIT(entry,
378  .lts = array_create(0, 0),
379  .rts = array_create(0, 0),
380  .lhost = lhost->clone(lhost),
381  .rhost = rhost->clone(rhost),
382  .spi = child_sa->get_spi(child_sa, TRUE),
383  .encap = encap,
384  .mark = child_sa->get_mark(child_sa, TRUE).value,
385  .reinject = is_reinject_config(this, child_sa->get_name(child_sa)),
386  .broadcast = this->broadcast,
387  );
388 
389  enumerator = child_sa->create_ts_enumerator(child_sa, TRUE);
390  while (enumerator->enumerate(enumerator, &ts))
391  {
392  array_insert(entry->lts, ARRAY_TAIL, ts->clone(ts));
393  }
394  enumerator->destroy(enumerator);
395 
396  enumerator = child_sa->create_ts_enumerator(child_sa, FALSE);
397  while (enumerator->enumerate(enumerator, &ts))
398  {
399  array_insert(entry->rts, ARRAY_TAIL, ts->clone(ts));
400  }
401  enumerator->destroy(enumerator);
402 
403  if (manage_pre(ipth, entry, TRUE) &&
404  manage_out(ipth, entry, TRUE))
405  {
406  this->lock->write_lock(this->lock);
407  this->entries->insert_last(this->entries, entry);
408  this->lock->unlock(this->lock);
409  return TRUE;
410  }
411  entry_destroy(entry);
412  return FALSE;
413 }
