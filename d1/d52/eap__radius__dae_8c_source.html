382 static bool receive(private_eap_radius_dae_t *this)
383 {
384  struct sockaddr_storage addr;
385  socklen_t addr_len = sizeof(addr);
386  radius_message_t *request;
387  char buf[2048];
388  ssize_t len;
389  host_t *client;
390 
391  len = recvfrom(this->fd, buf, sizeof(buf), MSG_DONTWAIT,
392  (struct sockaddr*)&addr, &addr_len);
393  if (len > 0)
394  {
395  request = radius_message_parse(chunk_create(buf, len));
396  if (request)
397  {
398  client = host_create_from_sockaddr((struct sockaddr*)&addr);
399  if (client)
400  {
401  if (!send_retransmit(this, request, client))
402  {
403  if (request->verify(request, NULL, this->secret,
404  this->hasher, this->signer))
405  {
406  switch (request->get_code(request))
407  {
408  case RMC_DISCONNECT_REQUEST:
409  process_disconnect(this, request, client);
410  break;
411  case RMC_COA_REQUEST:
412  process_coa(this, request, client);
413  break;
414  default:
415  DBG1(DBG_CFG, "ignoring unsupported RADIUS DAE "
416  "%N message from %H",
417  radius_message_code_names,
418  request->get_code(request), client);
419  break;
420  }
421  }
422  }
423  client->destroy(client);
424  }
425  request->destroy(request);
426  }
427  else
428  {
429  DBG1(DBG_NET, "ignoring invalid RADIUS DAE request");
430  }
431  }
432  else if (errno != EWOULDBLOCK)
433  {
434  DBG1(DBG_NET, "receiving RADIUS DAE request failed: %s", strerror(errno));
435  }
436  return TRUE;
437 }
