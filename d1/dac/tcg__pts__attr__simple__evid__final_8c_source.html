169 {
170  bio_writer_t *writer;
171  u_int8_t flags;

176  }
177  flags = this->flags & PTS_SIMPLE_EVID_FINAL_FLAG_MASK;

182  }
183 
184  writer = bio_writer_create(PTS_SIMPLE_EVID_FINAL_SIZE);
185  writer->write_uint8 (writer, flags);
186  writer->write_uint8 (writer, PTS_SIMPLE_EVID_FINAL_RESERVED);
187 
192  writer->write_uint16(writer, this->comp_hash_algorithm);
193 
194  /* Optional fields */
195  if (this->flags != PTS_SIMPLE_EVID_FINAL_NO)
196  {
197  writer->write_uint32 (writer, this->pcr_comp.len);
198  writer->write_data (writer, this->pcr_comp);
199

201  writer->write_data (writer, this->tpm_quote_sig);
202  }
203 
204  if (this->has_evid_sig)
205  {
206  writer->write_data (writer, this->evid_sig);
207  }
208 
209  this->value = writer->extract_buf(writer);
210  this->length = this->value.len;
211  writer->destroy(writer);

214 METHOD(pa_tnc_attr_t, process, status_t,
215  private_tcg_pts_attr_simple_evid_final_t *this, u_int32_t *offset)
216 {
217  bio_reader_t *reader;
218  u_int8_t flags, reserved;
219  u_int16_t algorithm;
220  u_int32_t pcr_comp_len, tpm_quote_sig_len, evid_sig_len;
221  status_t status = FAILED;
222 
223  *offset = 0;
224 
225  if (this->value.len < this->length)
226  {
227  return NEED_MORE;
228  }
229  if (this->value.len < PTS_SIMPLE_EVID_FINAL_SIZE)
230  {
231  DBG1(DBG_TNC, "insufficient data for Simple Evidence Final");
232  return FAILED;
233  }
234  reader = bio_reader_create(this->value);
235 
236  reader->read_uint8(reader, &flags);
237  reader->read_uint8(reader, &reserved);
238 
239  this->flags = flags & PTS_SIMPLE_EVID_FINAL_FLAG_MASK;
240 
241  this->has_evid_sig = (flags & PTS_SIMPLE_EVID_FINAL_EVID_SIG) != 0;
242 
248  reader->read_uint16(reader, &algorithm);
249  this->comp_hash_algorithm = algorithm;
250 
251  /* Optional Composite Hash Algorithm and TPM PCR Composite fields */
252  if (this->flags != PTS_SIMPLE_EVID_FINAL_NO)
253  {
254  if (!reader->read_uint32(reader, &pcr_comp_len))
255  {
256  DBG1(DBG_TNC, "insufficient data for PTS Simple Evidence Final "
257  "PCR Composite Length");
258  goto end;
259  }
260  if (!reader->read_data(reader, pcr_comp_len, &this->pcr_comp))
261  {
262  DBG1(DBG_TNC, "insufficient data for PTS Simple Evidence Final "
263  "PCR Composite");
264  goto end;
265  }
266  this->pcr_comp = chunk_clone(this->pcr_comp);
267 
268  if (!reader->read_uint32(reader, &tpm_quote_sig_len))
269  {
270  DBG1(DBG_TNC, "insufficient data for PTS Simple Evidence Final "
271  "TPM Quote Singature Length");
272  goto end;
273  }
274  if (!reader->read_data(reader, tpm_quote_sig_len, &this->tpm_quote_sig))
275  {
276  DBG1(DBG_TNC, "insufficient data for PTS Simple Evidence Final "
277  "TPM Quote Singature");
278  goto end;
279  }
280  this->tpm_quote_sig = chunk_clone(this->tpm_quote_sig);
281  }
282 
283  /* Optional Evidence Signature field */
284  if (this->has_evid_sig)
285  {
286  evid_sig_len = reader->remaining(reader);
287  reader->read_data(reader, evid_sig_len, &this->evid_sig);
288  this->evid_sig = chunk_clone(this->evid_sig);
289  }
290 
291  reader->destroy(reader);
292  return SUCCESS;
293 
294 end:
295  reader->destroy(reader);
296  return status;
297 }
