116 static bool register_vici(private_vici_plugin_t *this,
117  plugin_feature_t *feature, bool reg, void *data)
118 {
119  if (reg)
120  {
121  char *uri;
122 
123  uri = lib->settings->get_str(lib->settings, "%s.plugins.vici.socket",
124  VICI_DEFAULT_URI, lib->ns);
125  this->dispatcher = vici_dispatcher_create(uri);
126  if (this->dispatcher)
127  {
128  this->query = vici_query_create(this->dispatcher);
129  this->control = vici_control_create(this->dispatcher);
130  this->cred = vici_cred_create(this->dispatcher);
131  this->authority = vici_authority_create(this->dispatcher,
132  this->cred);
133  lib->credmgr->add_set(lib->credmgr, &this->authority->set);
134  this->config = vici_config_create(this->dispatcher, this->authority);
135  this->attrs = vici_attribute_create(this->dispatcher);
136  this->logger = vici_logger_create(this->dispatcher);
137 
138  charon->backends->add_backend(charon->backends,
139  &this->config->backend);
140  charon->attributes->add_provider(charon->attributes,
141  &this->attrs->provider);
142  charon->bus->add_logger(charon->bus, &this->logger->logger);
143  charon->bus->add_listener(charon->bus, &this->query->listener);
144  return TRUE;
145  }
146  return FALSE;
147  }
148  else
149  {
150  charon->bus->remove_listener(charon->bus, &this->query->listener);
151  charon->bus->remove_logger(charon->bus, &this->logger->logger);
152  charon->attributes->remove_provider(charon->attributes,
153  &this->attrs->provider);
154  charon->backends->remove_backend(charon->backends,
155  &this->config->backend);
156 
157  this->logger->destroy(this->logger);
158  this->attrs->destroy(this->attrs);
159  this->config->destroy(this->config);
160  lib->credmgr->remove_set(lib->credmgr, &this->authority->set);
161  this->authority->destroy(this->authority);
162  this->cred->destroy(this->cred);
163  this->control->destroy(this->control);
164  this->query->destroy(this->query);
165  this->dispatcher->destroy(this->dispatcher);
166  }
167  return TRUE;
168 }
