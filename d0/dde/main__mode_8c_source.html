124 static bool has_notify_errors(private_main_mode_t *this, message_t *message)
125 {
126  enumerator_t *enumerator;
127  payload_t *payload;
128  bool err = FALSE;
129 
130  enumerator = message->create_payload_enumerator(message);
131  while (enumerator->enumerate(enumerator, &payload))
132  {
133  if (payload->get_type(payload) == PLV1_NOTIFY)
134  {
135  notify_payload_t *notify;
136  notify_type_t type;
137 
138  notify = (notify_payload_t*)payload;
139  type = notify->get_notify_type(notify);
140  if (type < 16384)
141  {
142  DBG1(DBG_IKE, "received %N error notify",
143  notify_type_names, type);
144  err = TRUE;
145  }
146  else if (type == INITIAL_CONTACT_IKEV1)
147  {
148  if (!this->initiator && this->state == MM_AUTH)
149  {
150  /* If authenticated and received INITIAL_CONTACT,
151  * delete any existing IKE_SAs with that peer.
152  * The delete takes place when the SA is checked in due
153  * to other id not known until the 3rd message.*/
154  this->ike_sa->set_condition(this->ike_sa,
155  COND_INIT_CONTACT_SEEN, TRUE);
156  }
157  }
158  else
159  {
160  DBG1(DBG_IKE, "received %N notify", notify_type_names, type);
161  }
162  }
163  }
164  enumerator->destroy(enumerator);
165 
166  return err;
167 }

172 static status_t send_notify(private_main_mode_t *this, notify_type_t type)
173 {
174  notify_payload_t *notify;
175  ike_sa_id_t *ike_sa_id;
176  u_int64_t spi_i, spi_r;
177  chunk_t spi;
178 
179  notify = notify_payload_create_from_protocol_and_type(PLV1_NOTIFY,
180  PROTO_IKE, type);
181  ike_sa_id = this->ike_sa->get_id(this->ike_sa);
182  spi_i = ike_sa_id->get_initiator_spi(ike_sa_id);
183  spi_r = ike_sa_id->get_responder_spi(ike_sa_id);
184  spi = chunk_cata("cc", chunk_from_thing(spi_i), chunk_from_thing(spi_r));
185  notify->set_spi_data(notify, spi);
186 
187  this->ike_sa->queue_task(this->ike_sa,
188  (task_t*)informational_create(this->ike_sa, notify));
189  /* cancel all active/passive tasks in favour of informational */
190  this->ike_sa->flush_queue(this->ike_sa,
191  this->initiator ? TASK_QUEUE_ACTIVE : TASK_QUEUE_PASSIVE);
192  return ALREADY_DONE;
193 }

211 static void add_initial_contact(private_main_mode_t *this, message_t *message,
212  identification_t *idi)
213 {
214  identification_t *idr;
215  host_t *host;
216  notify_payload_t *notify;
217  ike_sa_id_t *ike_sa_id;
218  u_int64_t spi_i, spi_r;
219  chunk_t spi;
220 
221  idr = this->ph1->get_id(this->ph1, this->peer_cfg, FALSE);
222  if (idr && !idr->contains_wildcards(idr))
223  {
224  if (this->peer_cfg->get_unique_policy(this->peer_cfg) != UNIQUE_NO &&
225  this->peer_cfg->get_unique_policy(this->peer_cfg) != UNIQUE_NEVER)
226  {
227  host = this->ike_sa->get_other_host(this->ike_sa);
228  if (!charon->ike_sa_manager->has_contact(charon->ike_sa_manager,
229  idi, idr, host->get_family(host)))
230  {
231  notify = notify_payload_create_from_protocol_and_type(
232  PLV1_NOTIFY, PROTO_IKE, INITIAL_CONTACT_IKEV1);
233  ike_sa_id = this->ike_sa->get_id(this->ike_sa);
234  spi_i = ike_sa_id->get_initiator_spi(ike_sa_id);
235  spi_r = ike_sa_id->get_responder_spi(ike_sa_id);
236  spi = chunk_cata("cc", chunk_from_thing(spi_i),
237  chunk_from_thing(spi_r));
238  notify->set_spi_data(notify, spi);
239  message->add_payload(message, (payload_t*)notify);
240  }

245 METHOD(task_t, build_i, status_t,
246  private_main_mode_t *this, message_t *message)
247 {
248  switch (this->state)
249  {
250  case MM_INIT:
251  {
252  sa_payload_t *sa_payload;
253  linked_list_t *proposals;
254  packet_t *packet;
255 
256  DBG0(DBG_IKE, "initiating Main Mode IKE_SA %s[%d] to %H",
257  this->ike_sa->get_name(this->ike_sa),
258  this->ike_sa->get_unique_id(this->ike_sa),
259  this->ike_sa->get_other_host(this->ike_sa));
260  this->ike_sa->set_state(this->ike_sa, IKE_CONNECTING);
261 
262  this->ike_cfg = this->ike_sa->get_ike_cfg(this->ike_sa);
263  this->peer_cfg = this->ike_sa->get_peer_cfg(this->ike_sa);
264  this->peer_cfg->get_ref(this->peer_cfg);
265 
266  this->method = this->ph1->get_auth_method(this->ph1, this->peer_cfg);
267  if (this->method == AUTH_NONE)
268  {
269  DBG1(DBG_CFG, "configuration uses unsupported authentication");
270  return FAILED;
271  }
272  this->lifetime = this->peer_cfg->get_reauth_time(this->peer_cfg,
273  FALSE);
274  if (!this->lifetime)
275  { /* fall back to rekey time of no rekey time configured */
276  this->lifetime = this->peer_cfg->get_rekey_time(this->peer_cfg,
277  FALSE);
278  }
279  this->lifetime += this->peer_cfg->get_over_time(this->peer_cfg);
280  proposals = this->ike_cfg->get_proposals(this->ike_cfg);
281  sa_payload = sa_payload_create_from_proposals_v1(proposals,
282  this->lifetime, 0, this->method, MODE_NONE,
283  ENCAP_NONE, 0);
284  proposals->destroy_offset(proposals, offsetof(proposal_t, destroy));
285 
286  message->add_payload(message, &sa_payload->payload_interface);
287 
288  /* pregenerate message to store SA payload */
289  if (this->ike_sa->generate_message(this->ike_sa, message,
290  &packet) != SUCCESS)
291  {
292  DBG1(DBG_IKE, "pregenerating SA payload failed");
293  return FAILED;
294  }
295  packet->destroy(packet);
296  if (!this->ph1->save_sa_payload(this->ph1, message))
297  {
298  return FAILED;
299  }
300 
301  this->state = MM_SA;
302  return NEED_MORE;
303  }
304  case MM_SA:
305  {
306  u_int16_t group;
307 
308  if (!this->ph1->create_hasher(this->ph1))
309  {
310  return send_notify(this, NO_PROPOSAL_CHOSEN);
311  }
312  if (!this->proposal->get_algorithm(this->proposal,
313  DIFFIE_HELLMAN_GROUP, &group, NULL))
314  {
315  DBG1(DBG_IKE, "DH group selection failed");
316  return send_notify(this, NO_PROPOSAL_CHOSEN);
317  }
318  if (!this->ph1->create_dh(this->ph1, group))
319  {
320  DBG1(DBG_IKE, "negotiated DH group not supported");
321  return send_notify(this, INVALID_KEY_INFORMATION);
322  }
323  if (!this->ph1->add_nonce_ke(this->ph1, message))
324  {
325  return send_notify(this, INVALID_KEY_INFORMATION);
326  }
327  this->state = MM_KE;
328  return NEED_MORE;
329  }
330  case MM_KE:
331  {
332  id_payload_t *id_payload;
333  identification_t *id;
334 
335  id = this->ph1->get_id(this->ph1, this->peer_cfg, TRUE);
336  if (!id)
337  {
338  DBG1(DBG_CFG, "own identity not known");
339  return send_notify(this, INVALID_ID_INFORMATION);
340  }
341  this->ike_sa->set_my_id(this->ike_sa, id->clone(id));
342  id_payload = id_payload_create_from_identification(PLV1_ID, id);
343  message->add_payload(message, &id_payload->payload_interface);
344 
345  if (!this->ph1->build_auth(this->ph1, this->method, message,
346  id_payload->get_encoded(id_payload)))
347  {
348  return send_notify(this, AUTHENTICATION_FAILED);
349  }
350 
351  add_initial_contact(this, message, id);
352 
353  this->state = MM_AUTH;
354  return NEED_MORE;
355  }
356  default:
357  return FAILED;
358  }

361 METHOD(task_t, process_r, status_t,
362  private_main_mode_t *this, message_t *message)
363 {
364  switch (this->state)
365  {
366  case MM_INIT:
367  {
368  linked_list_t *list;
369  sa_payload_t *sa_payload;
370  bool private;
371 
372  this->ike_cfg = this->ike_sa->get_ike_cfg(this->ike_sa);
373  DBG0(DBG_IKE, "%H is initiating a Main Mode IKE_SA",
374  message->get_source(message));
375  this->ike_sa->set_state(this->ike_sa, IKE_CONNECTING);
376 
377  this->ike_sa->update_hosts(this->ike_sa,
378  message->get_destination(message),
379  message->get_source(message), TRUE);
380 
381  sa_payload = (sa_payload_t*)message->get_payload(message,
382  PLV1_SECURITY_ASSOCIATION);
383  if (!sa_payload)
384  {
385  DBG1(DBG_IKE, "SA payload missing");
386  return send_notify(this, INVALID_PAYLOAD_TYPE);
387  }
388  if (!this->ph1->save_sa_payload(this->ph1, message))
389  {
390  return send_notify(this, INVALID_PAYLOAD_TYPE);
391  }
392 
393  list = sa_payload->get_proposals(sa_payload);
394  private = this->ike_sa->supports_extension(this->ike_sa,
395  EXT_STRONGSWAN);
396  this->proposal = this->ike_cfg->select_proposal(this->ike_cfg,
397  list, private);
398  list->destroy_offset(list, offsetof(proposal_t, destroy));
399  if (!this->proposal)
400  {
401  DBG1(DBG_IKE, "no proposal found");
402  return send_notify(this, NO_PROPOSAL_CHOSEN);
403  }
404  this->ike_sa->set_proposal(this->ike_sa, this->proposal);
405 
406  this->method = sa_payload->get_auth_method(sa_payload);
407  this->lifetime = sa_payload->get_lifetime(sa_payload);
408 
409  this->state = MM_SA;
410  return NEED_MORE;
411  }
412  case MM_SA:
413  {
414  u_int16_t group;
415 
416  if (!this->ph1->create_hasher(this->ph1))
417  {
418  return send_notify(this, INVALID_KEY_INFORMATION);
419  }
420  if (!this->proposal->get_algorithm(this->proposal,
421  DIFFIE_HELLMAN_GROUP, &group, NULL))
422  {
423  DBG1(DBG_IKE, "DH group selection failed");
424  return send_notify(this, INVALID_KEY_INFORMATION);
425  }
426  if (!this->ph1->create_dh(this->ph1, group))
427  {
428  DBG1(DBG_IKE, "negotiated DH group not supported");
429  return send_notify(this, INVALID_KEY_INFORMATION);
430  }
431  if (!this->ph1->get_nonce_ke(this->ph1, message))
432  {
433  return send_notify(this, INVALID_PAYLOAD_TYPE);
434  }
435  this->state = MM_KE;
436  return NEED_MORE;
437  }
438  case MM_KE:
439  {
440  id_payload_t *id_payload;
441  identification_t *id;
442 
443  id_payload = (id_payload_t*)message->get_payload(message, PLV1_ID);
444  if (!id_payload)
445  {
446  DBG1(DBG_IKE, "IDii payload missing");
447  return send_notify(this, INVALID_PAYLOAD_TYPE);
448  }
449  id = id_payload->get_identification(id_payload);
450  this->ike_sa->set_other_id(this->ike_sa, id);
451 
452  while (TRUE)
453  {
454  DESTROY_IF(this->peer_cfg);
455  this->peer_cfg = this->ph1->select_config(this->ph1,
456  this->method, FALSE, id);
457  if (!this->peer_cfg)
458  {
459  return send_notify(this, AUTHENTICATION_FAILED);
460  }
461  this->ike_sa->set_peer_cfg(this->ike_sa, this->peer_cfg);
462 
463  if (this->ph1->verify_auth(this->ph1, this->method, message,
464  id_payload->get_encoded(id_payload)))
465  {
466  break;
467  }
468  }
469 
470  if (!charon->bus->authorize(charon->bus, FALSE))
471  {
472  DBG1(DBG_IKE, "Main Mode authorization hook forbids IKE_SA, "
473  "cancelling");
474  return send_notify(this, AUTHENTICATION_FAILED);
475  }
476 
477  this->state = MM_AUTH;
478  if (has_notify_errors(this, message))
479  {
480  return FAILED;
481  }
482  return NEED_MORE;
483  }
484  default:
485  return FAILED;
486  }

582  if (job)
583  {
584  job->queue_task(job, (task_t*)

596  }

603  }
604  default:
