108 static bool do_netstat(ietf_attr_port_filter_t *attr)
109 {
110  FILE *file;
111  char buf[BUF_LEN];
112  chunk_t line, token;
113  int n = 0;
114  bool success = FALSE;
115  const char system_v4[] = "127.0.1.1";
116  const char loopback_v4[] = "127.0.0.1";
117  const char loopback_v6[] = "::1";
118 
119  /* Open a pipe stream for reading the output of the netstat commmand */
120  file = popen("/bin/netstat -n -l -p -4 -6 --inet", "r");
121  if (!file)
122  {
123  DBG1(DBG_IMC, "failed to run netstat command");
124  return FALSE;
125  }
126 
127  /* Read the output a line at a time */
128  while (fgets(buf, sizeof(buf), file))
129  {
130  u_char *pos;
131  u_int8_t new_protocol, protocol;
132  u_int16_t new_port, port;
133  int i;
134  enumerator_t *enumerator;
135  bool allowed, found = FALSE;
136 
137  DBG2(DBG_IMC, "%.*s", (int)(strlen(buf)-1), buf);
138 
139  if (n++ < 2)
140  {
141  /* skip the first two header lines */
142  continue;
143  }
144  line = chunk_create(buf, strlen(buf));
145 
146  /* Extract the IP protocol type */
147  if (!extract_token(&token, ' ', &line))
148  {
149  DBG1(DBG_IMC, "protocol field in netstat output not found");
150  goto end;
151  }
152  if (match("tcp", &token) || match("tcp6", &token))
153  {
154  new_protocol = IPPROTO_TCP;
155  }
156  else if (match("udp", &token) || match("udp6", &token))
157  {
158  new_protocol = IPPROTO_UDP;
159  }
160  else
161  {
162  DBG1(DBG_IMC, "skipped unknown IP protocol in netstat output");
163  continue;
164  }
165 
166  /* Skip the Recv-Q and Send-Q fields */
167  for (i = 0; i < 3; i++)
168  {
169  if (!eat_whitespace(&line) || !extract_token(&token, ' ', &line))
170  {
171  token = chunk_empty;
172  break;
173  }
174  }
175  if (token.len == 0)
176  {
177  DBG1(DBG_IMC, "local address field in netstat output not found");
178  goto end;
179  }
180 
181  /* Find the local port appended to the local address */
182  pos = token.ptr + token.len;
183  while (*--pos != ':' && --token.len);
184  if (*pos != ':')
185  {
186  DBG1(DBG_IMC, "local port field in netstat output not found");
187  goto end;
188  }
189  token.len--;
190 
191  /* ignore ports of IPv4 and IPv6 loopback interfaces
192  and the internal system IPv4 address */
193  if ((token.len == strlen(system_v4) &&
194  memeq(system_v4, token.ptr, token.len)) ||
195  (token.len == strlen(loopback_v4) &&
196  memeq(loopback_v4, token.ptr, token.len)) ||
197  (token.len == strlen(loopback_v6) &&
198  memeq(loopback_v6, token.ptr, token.len)))
199  {
200  continue;
201  }
202 
203  /* convert the port string to an integer */
204  new_port = atoi(pos+1);
205 
206  /* check if the there is already a port entry */
207  enumerator = attr->create_port_enumerator(attr);
208  while (enumerator->enumerate(enumerator, &allowed, &protocol, &port))
209  {
210  if (new_port == port && new_protocol == protocol)
211  {
212  found = TRUE;
213  }
214  }
215  enumerator->destroy(enumerator);
216 
217  /* Skip the duplicate port entry */
218  if (found)
219  {
220  continue;
221  }
222 
223  /* Add new port entry */
224  attr->add_port(attr, FALSE, new_protocol, new_port);
225  }
226 
227  /* Successfully completed the parsing of the netstat output */
228  success = TRUE;
229 
230 end:
231  /* Close the pipe stream */
232  pclose(file);
233  return success;
234 }

239 static TNC_Result add_port_filter(imc_msg_t *msg)
240 {
241  pa_tnc_attr_t *attr;
242  ietf_attr_port_filter_t *attr_port_filter;
243 
244  attr = ietf_attr_port_filter_create(pen_type_create(PEN_IETF,
245  IETF_ATTR_PORT_FILTER));
246  attr->set_noskip_flag(attr, TRUE);
247  attr_port_filter = (ietf_attr_port_filter_t*)attr;
248  if (!do_netstat(attr_port_filter))
249  {
250  attr->destroy(attr);
251  return TNC_RESULT_FATAL;
252  }
253  msg->add_attribute(msg, attr);
254 
255  return TNC_RESULT_SUCCESS;

294 static TNC_Result receive_message(imc_msg_t *in_msg)
295 {
296  imc_msg_t *out_msg;
297  enumerator_t *enumerator;
298  pa_tnc_attr_t *attr;
299  pen_type_t attr_type;
300  TNC_Result result = TNC_RESULT_SUCCESS;
301  bool fatal_error = FALSE;
302 
303  /* generate an outgoing PA-TNC message - we might need it */
304  out_msg = imc_msg_create_as_reply(in_msg);
305 
306  /* parse received PA-TNC message and handle local and remote errors */
307  result = in_msg->receive(in_msg, out_msg, &fatal_error);
308  if (result != TNC_RESULT_SUCCESS)
309  {
310  out_msg->destroy(out_msg);
311  return result;
312  }
313 
314  /* analyze PA-TNC attributes */
315  enumerator = in_msg->create_attribute_enumerator(in_msg);
316  while (enumerator->enumerate(enumerator, &attr))
317  {
318  attr_type = attr->get_type(attr);
319 
320  if (attr_type.vendor_id != PEN_IETF)
321  {
322  continue;
323  }
324  if (attr_type.type == IETF_ATTR_ATTRIBUTE_REQUEST)
325  {
326  ietf_attr_attr_request_t *attr_cast;
327  pen_type_t *entry;
328  enumerator_t *e;
329 
330  attr_cast = (ietf_attr_attr_request_t*)attr;
331 
332  e = attr_cast->create_enumerator(attr_cast);
333  while (e->enumerate(e, &entry))
334  {
335  if (entry->vendor_id != PEN_IETF)
336  {
337  continue;
338  }
339  switch (entry->type)
340  {
341  case IETF_ATTR_PORT_FILTER:
342  result = add_port_filter(out_msg);
343  break;
344  default:
345  break;
346  }
347  }
348  e->destroy(e);
349  }
350  }
351  enumerator->destroy(enumerator);
352 
353  if (fatal_error)
354  {
355  result = TNC_RESULT_FATAL;
356  }
357  else if (result == TNC_RESULT_SUCCESS)
358  {
359  /* send PA-TNC message with the EXCL flag set */
360  result = out_msg->send(out_msg, TRUE);
361  }
362  out_msg->destroy(out_msg);
363 
364  return result;
365 }
