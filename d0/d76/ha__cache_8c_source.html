112 METHOD(ha_cache_t, cache, void,
113  private_ha_cache_t *this, ike_sa_t *ike_sa, ha_message_t *message)
114 {
115  entry_t *entry;
116 
117  this->mutex->lock(this->mutex);
118  switch (message->get_type(message))
119  {
120  case HA_IKE_ADD:
121  entry = entry_create(message);
122  entry = this->cache->put(this->cache, ike_sa, entry);
123  if (entry)
124  {
125  entry_destroy(entry);
126  }
127  break;
128  case HA_IKE_UPDATE:
129  entry = this->cache->get(this->cache, ike_sa);
130  if (entry)
131  {
132  entry->segment = this->kernel->get_segment(this->kernel,
133  ike_sa->get_other_host(ike_sa));
134  entry->updates->insert_last(entry->updates, message);
135  break;
136  }
137  message->destroy(message);
138  break;
139  case HA_IKE_MID_INITIATOR:
140  entry = this->cache->get(this->cache, ike_sa);
141  if (entry)
142  {
143  DESTROY_IF(entry->midi);
144  entry->midi = message;
145  break;
146  }
147  message->destroy(message);
148  break;
149  case HA_IKE_MID_RESPONDER:
150  entry = this->cache->get(this->cache, ike_sa);
151  if (entry)
152  {
153  DESTROY_IF(entry->midr);
154  entry->midr = message;
155  break;
156  }
157  message->destroy(message);
158  break;
159  case HA_IKE_IV:
160  entry = this->cache->get(this->cache, ike_sa);
161  if (entry)
162  {
163  DESTROY_IF(entry->iv);
164  entry->iv = message;
165  break;
166  }
167  message->destroy(message);
168  break;
169  case HA_IKE_DELETE:
170  entry = this->cache->remove(this->cache, ike_sa);
171  if (entry)
172  {
173  entry_destroy(entry);
174  }
175  message->destroy(message);
176  break;
177  default:
178  message->destroy(message);
179  break;
180  }
181  this->mutex->unlock(this->mutex);
182 }

199 static status_t rekey_children(ike_sa_t *ike_sa)
200 {
201  enumerator_t *enumerator;
202  child_sa_t *child_sa;
203  status_t status = SUCCESS;
204  linked_list_t *children;
205  struct {
206  protocol_id_t protocol;
207  u_int32_t spi;
208  } *info;
209 
210  children = linked_list_create();
211  enumerator = ike_sa->create_child_sa_enumerator(ike_sa);
212  while (enumerator->enumerate(enumerator, &child_sa))
213  {
214  INIT(info,
215  .protocol = child_sa->get_protocol(child_sa),
216  .spi = child_sa->get_spi(child_sa, TRUE),
217  );
218  children->insert_last(children, info);
219  }
220  enumerator->destroy(enumerator);
221 
222  enumerator = children->create_enumerator(children);
223  while (enumerator->enumerate(enumerator, &info))
224  {
225  if (ike_sa->supports_extension(ike_sa, EXT_MS_WINDOWS) &&
226  ike_sa->has_condition(ike_sa, COND_NAT_THERE))
227  {
228  /* NATed Windows clients don't accept CHILD_SA rekeying, but fail
229  * with an "invalid situation" error. We just close the CHILD_SA,
230  * Windows will reestablish it immediately if required. */
231  DBG1(DBG_CFG, "resyncing CHILD_SA using a delete");
232  status = ike_sa->delete_child_sa(ike_sa, info->protocol, info->spi,
233  FALSE);
234  }
235  else
236  {
237  DBG1(DBG_CFG, "resyncing CHILD_SA using a rekey");
238  status = ike_sa->rekey_child_sa(ike_sa, info->protocol, info->spi);
239  }
240  if (status == DESTROY_ME)
241  {
242  break;
243  }
244  }
245  enumerator->destroy(enumerator);
246  children->destroy_function(children, free);
247 
248  return status;
249 }

254 static void rekey_segment(private_ha_cache_t *this, u_int segment)
255 {
256  ike_sa_t *ike_sa;
257  enumerator_t *enumerator;
258  linked_list_t *list;
259  ike_sa_id_t *id;
260 
261  list = linked_list_create();
262 
263  enumerator = charon->ike_sa_manager->create_enumerator(
264  charon->ike_sa_manager, TRUE);
265  while (enumerator->enumerate(enumerator, &ike_sa))
266  {
267  if (this->tunnel && this->tunnel->is_sa(this->tunnel, ike_sa))
268  {
269  continue;
270  }
271  if (ike_sa->get_state(ike_sa) == IKE_ESTABLISHED &&
272  this->kernel->get_segment(this->kernel,
273  ike_sa->get_other_host(ike_sa)) == segment)
274  {
275  id = ike_sa->get_id(ike_sa);
276  list->insert_last(list, id->clone(id));
277  }
278  }
279  enumerator->destroy(enumerator);
280 
281  while (list->remove_last(list, (void**)&id) == SUCCESS)
282  {
283  ike_sa = charon->ike_sa_manager->checkout(charon->ike_sa_manager, id);
284  if (ike_sa)
285  {
286  if (rekey_children(ike_sa) != DESTROY_ME)
287  {
288  charon->ike_sa_manager->checkin(
289  charon->ike_sa_manager, ike_sa);
290  }
291  else
292  {
293  charon->ike_sa_manager->checkin_and_destroy(
294  charon->ike_sa_manager, ike_sa);
295  }
296  }
297  id->destroy(id);
298  }
299  list->destroy(list);
300 }

302 METHOD(ha_cache_t, resync, void,
303  private_ha_cache_t *this, u_int segment)
304 {
305  enumerator_t *enumerator, *updates;
306  ike_sa_t *ike_sa;
307  entry_t *entry;
308  ha_message_t *message;
309 
310  DBG1(DBG_CFG, "resyncing HA segment %d", segment);
311 
312  this->mutex->lock(this->mutex);
313  enumerator = this->cache->create_enumerator(this->cache);
314  while (enumerator->enumerate(enumerator, &ike_sa, &entry))
315  {
316  if (entry->segment == segment)
317  {
318  this->socket->push(this->socket, entry->add);
319  updates = entry->updates->create_enumerator(entry->updates);
320  while (updates->enumerate(updates, &message))
321  {
322  this->socket->push(this->socket, message);
323  }
324  updates->destroy(updates);
325  if (entry->midi)
326  {
327  this->socket->push(this->socket, entry->midi);
328  }
329  if (entry->midr)
330  {
331  this->socket->push(this->socket, entry->midr);
332  }
333  if (entry->iv)
334  {
335  this->socket->push(this->socket, entry->iv);
336  }
337  }
338  }
339  enumerator->destroy(enumerator);
340  this->mutex->unlock(this->mutex);
341 
342  rekey_segment(this, segment);
343 }
