185 static bool parse_directoryName(chunk_t blob, int level, bool implicit,
186  identification_t **name)
187 {
188  identification_t *directoryName;
189  enumerator_t *enumerator;
190  bool first = TRUE;
191  linked_list_t *list;
192 
193  list = linked_list_create();
194  x509_parse_generalNames(blob, level, implicit, list);
195 
196  enumerator = list->create_enumerator(list);
197  while (enumerator->enumerate(enumerator, &directoryName))
198  {
199  if (first)
200  {
201  *name = directoryName;
202  first = FALSE;
203  }
204  else
205  {
206  DBG1(DBG_ASN, "more than one directory name - first selected");
207  directoryName->destroy(directoryName);
208  break;
209  }
210  }
211  enumerator->destroy(enumerator);
212  list->destroy(list);
213 
214  if (first)
215  {
216  DBG1(DBG_ASN, "no directoryName found");
217  return FALSE;
218  }
219  return TRUE;
220 }

287 static bool parse_groups(private_x509_ac_t *this, chunk_t encoded, int level0)
288 {
289  ac_group_type_t type;
290  group_t *group;
291  asn1_parser_t *parser;
292  chunk_t object;
293  int objectID;
294  bool success;
295 
296  parser = asn1_parser_create(ietfAttrSyntaxObjects, encoded);
297  parser->set_top_level(parser, level0);
298  while (parser->iterate(parser, &objectID, &object))
299  {
300  switch (objectID)
301  {

309  type = AC_GROUP_TYPE_STRING;
310  break;
311  default:
312  continue;
313  }
314  INIT(group,
315  .type = type,
316  .value = chunk_clone(object),
317  );
318  this->groups->insert_last(this->groups, group);
319  }
320  success = parser->success(parser);
321  parser->destroy(parser);
322 
323  return success;
324 }

418 static bool parse_certificate(private_x509_ac_t *this)
419 {
420  asn1_parser_t *parser;
421  chunk_t object;
422  int objectID;
423  int type = OID_UNKNOWN;
424  int extn_oid = OID_UNKNOWN;
425  int sig_alg = OID_UNKNOWN;
426  bool success = FALSE;
427  bool critical;
428 
429  parser = asn1_parser_create(acObjects, this->encoding);
430 
431  while (parser->iterate(parser, &objectID, &object))
432  {
433  u_int level = parser->get_level(parser)+1;
434 
435  switch (objectID)
436  {
437  case AC_OBJ_CERTIFICATE_INFO:
438  this->certificateInfo = object;
439  break;
440  case AC_OBJ_VERSION:
441  this->version = (object.len) ? (1 + (u_int)*object.ptr) : 1;
442  DBG2(DBG_ASN, " v%d", this->version);
443  if (this->version != 2)
444  {
445  DBG1(DBG_ASN, "v%d attribute certificates are not "
446  "supported", this->version);
447  goto end;
448  }
449  break;
450  case AC_OBJ_HOLDER_ISSUER:
451  if (!parse_directoryName(object, level, FALSE,
452  &this->holderIssuer))
453  {
454  goto end;
455  }
456  break;
457  case AC_OBJ_HOLDER_SERIAL:
458  this->holderSerial = identification_create_from_encoding(
459  ID_KEY_ID, object);
460  break;
461  case AC_OBJ_ENTITY_NAME:
462  if (!parse_directoryName(object, level, TRUE,
463  &this->entityName))
464  {
465  goto end;
466  }
467  break;
468  case AC_OBJ_ISSUER_NAME:
469  if (!parse_directoryName(object, level, FALSE,
470  &this->issuerName))
471  {
472  goto end;
473  }
474  break;
475  case AC_OBJ_SIG_ALG:
476  sig_alg = asn1_parse_algorithmIdentifier(object, level, NULL);
477  break;
478  case AC_OBJ_SERIAL_NUMBER:
479  this->serialNumber = chunk_clone(object);
480  break;
481  case AC_OBJ_NOT_BEFORE:
482  this->notBefore = asn1_to_time(&object, ASN1_GENERALIZEDTIME);
483  break;
484  case AC_OBJ_NOT_AFTER:
485  this->notAfter = asn1_to_time(&object, ASN1_GENERALIZEDTIME);
486  break;
487  case AC_OBJ_ATTRIBUTE_TYPE:
488  type = asn1_known_oid(object);
489  break;
490  case AC_OBJ_ATTRIBUTE_VALUE:
491  {
492  switch (type)
493  {
494  case OID_AUTHENTICATION_INFO:
495  DBG2(DBG_ASN, " need to parse authenticationInfo");
496  break;
497  case OID_ACCESS_IDENTITY:
498  DBG2(DBG_ASN, " need to parse accessIdentity");
499  break;
500  case OID_CHARGING_IDENTITY:
501  DBG2(DBG_ASN, " need to parse chargingIdentity");
502  break;
503  case OID_GROUP:
504  DBG2(DBG_ASN, "-- > --");
505  if (!parse_groups(this, object, level))
506  {
507  goto end;
508  }
509  DBG2(DBG_ASN, "-- < --");
510  break;
511  case OID_ROLE:
512  parse_roleSyntax(object, level);
513  break;
514  default:
515  break;
516  }
517  break;
518  }
519  case AC_OBJ_EXTN_ID:
520  extn_oid = asn1_known_oid(object);
521  break;
522  case AC_OBJ_CRITICAL:
523  critical = object.len && *object.ptr;
524  DBG2(DBG_ASN, " %s",(critical)?"TRUE":"FALSE");
525  break;
526  case AC_OBJ_EXTN_VALUE:
527  {
528  switch (extn_oid)
529  {
530  case OID_CRL_DISTRIBUTION_POINTS:
531  DBG2(DBG_ASN, " need to parse crlDistributionPoints");
532  break;
533  case OID_AUTHORITY_KEY_ID:
534  this->authKeyIdentifier =
535  x509_parse_authorityKeyIdentifier(object,
536  level, &this->authKeySerialNumber);
537  break;
538  case OID_TARGET_INFORMATION:
539  DBG2(DBG_ASN, " need to parse targetInformation");
540  break;
541  case OID_NO_REV_AVAIL:
542  this->noRevAvail = TRUE;
543  break;
544  default:
545  break;
546  }
547  break;
548  }
549  case AC_OBJ_ALGORITHM:
550  this->algorithm = asn1_parse_algorithmIdentifier(object, level,
551  NULL);
552  if (this->algorithm != sig_alg)
553  {
554  DBG1(DBG_ASN, " signature algorithms do not agree");
555  success = FALSE;
556  goto end;
557  }
558  break;
559  case AC_OBJ_SIGNATURE:
560  this->signature = chunk_skip(object, 1);
561  break;
562  default:
563  break;
564  }
565  }
566  success = parser->success(parser);
567 
568 end:
569  parser->destroy(parser);
570  return success;
571 }

636 static chunk_t build_attributes(private_x509_ac_t *this)
637 {
638  enumerator_t *enumerator;
639  group_t *group;
640  chunk_t values;
641  size_t size = 0, len;
642  u_char *pos;
643 
644  /* precalculate the total size of all values */
645  enumerator = this->groups->create_enumerator(this->groups);
646  while (enumerator->enumerate(enumerator, &group))
647  {
648  len = group->value.len;
649  size += 1 + (len > 0) + (len >= 128) +
650  (len >= 256) + (len >= 65536) + len;
651  }
652  enumerator->destroy(enumerator);
653 
654  pos = asn1_build_object(&values, ASN1_SEQUENCE, size);
655 
656  enumerator = this->groups->create_enumerator(this->groups);
657  while (enumerator->enumerate(enumerator, &group))
658  {
659  chunk_t attr;
660  asn1_t type;
661 
662  switch (group->type)
663  {
664  case AC_GROUP_TYPE_OCTETS:
665  type = ASN1_OCTET_STRING;
666  break;
667  case AC_GROUP_TYPE_STRING:
668  type = ASN1_UTF8STRING;
669  break;
670  case AC_GROUP_TYPE_OID:
671  type = ASN1_OID;
672  break;
673  default:
674  continue;
675  }
676  attr = asn1_simple_object(type, group->value);
677 
678  memcpy(pos, attr.ptr, attr.len);
679  pos += attr.len;
680  free(attr.ptr);
681  }
682  enumerator->destroy(enumerator);
683 
684  return asn1_wrap(ASN1_SEQUENCE, "m",
685  build_attribute_type(OID_GROUP,
686  asn1_wrap(ASN1_SEQUENCE, "m", values)));
687 }

692 static chunk_t build_authorityKeyIdentifier(private_x509_ac_t *this)
693 {

720  keyIdentifier,
