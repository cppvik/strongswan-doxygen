199 METHOD(tnccs_manager_t, create_connection, TNC_ConnectionID,

207  entry->type = type;
208  entry->tnccs = tnccs;
209  entry->send_message = send_message;
210  entry->request_handshake_retry = request_handshake_retry;
211  entry->max_msg_len = max_msg_len;
212  if (recs)
213  {
214  /* we assume a TNC Server needing recommendations from IMVs */

227  if (!tnc->imcs)

230  free(entry);
231  return 0;
232  }

234  }

236  entry->id = ++this->connection_id;

461 static TNC_Result identity_attribute(TNC_UInt32 buffer_len,
462  TNC_BufferReference buffer,
463  TNC_UInt32 *value_len,
464  linked_list_t *list)
465 {
466  bio_writer_t *writer;
467  enumerator_t *enumerator;
468  u_int32_t count;
469  chunk_t value;
470  tncif_identity_t *tnc_id;
471  TNC_Result result = TNC_RESULT_INVALID_PARAMETER;
472 
473  count = list->get_count(list);
474  writer = bio_writer_create(4 + TNCIF_IDENTITY_MIN_SIZE * count);
475  writer->write_uint32(writer, count);
476 
477  enumerator = list->create_enumerator(list);
478  while (enumerator->enumerate(enumerator, &tnc_id))
479  {
480  tnc_id->build(tnc_id, writer);
481  }
482  enumerator->destroy(enumerator);
483 
484  value = writer->get_buf(writer);
485  *value_len = value.len;
486  if (buffer && buffer_len >= value.len)
487  {
488  memcpy(buffer, value.ptr, value.len);
489  result = TNC_RESULT_SUCCESS;
490  }
491  writer->destroy(writer);
492 
493  return result;
494 }

496 METHOD(tnccs_manager_t, get_attribute, TNC_Result,
497  private_tnc_tnccs_manager_t *this, bool is_imc,
498  TNC_UInt32 imcv_id,
499  TNC_ConnectionID id,
500  TNC_AttributeID attribute_id,
501  TNC_UInt32 buffer_len,
502  TNC_BufferReference buffer,
503  TNC_UInt32 *value_len)
504 {
505  enumerator_t *enumerator;
506  tnccs_connection_entry_t *entry;
507  bool attribute_match = FALSE, entry_found = FALSE;
508 
509  if (is_imc)
510  {
511  switch (attribute_id)
512  {
513  /* these attributes are unsupported */
514  case TNC_ATTRIBUTEID_SOHR:
515  case TNC_ATTRIBUTEID_SSOHR:
516  return TNC_RESULT_INVALID_PARAMETER;
517 
518  /* these attributes are supported */
519  case TNC_ATTRIBUTEID_PRIMARY_IMC_ID:
520  attribute_match = TRUE;
521  break;
522 
523  /* these attributes are yet to be matched */
524  default:
525  break;
526  }
527  }
528  else
529  {
530  switch (attribute_id)
531  {
532  /* these attributes are unsupported or invalid */
533  case TNC_ATTRIBUTEID_REASON_STRING:
534  case TNC_ATTRIBUTEID_REASON_LANGUAGE:
535  case TNC_ATTRIBUTEID_SOH:
536  case TNC_ATTRIBUTEID_SSOH:
537  return TNC_RESULT_INVALID_PARAMETER;
538 
539  /* these attributes are supported */
540  case TNC_ATTRIBUTEID_PRIMARY_IMV_ID:
541  case TNC_ATTRIBUTEID_AR_IDENTITIES:
542  attribute_match = TRUE;
543  break;
544 
545  /* these attributes are yet to be matched */
546  default:
547  break;
548  }
549  }
550 
551  if (!attribute_match)
552  {
553  switch (attribute_id)
554  {
555  /* these attributes are supported */
556  case TNC_ATTRIBUTEID_PREFERRED_LANGUAGE:
557  case TNC_ATTRIBUTEID_MAX_ROUND_TRIPS:
558  case TNC_ATTRIBUTEID_MAX_MESSAGE_SIZE:
559  case TNC_ATTRIBUTEID_HAS_LONG_TYPES:
560  case TNC_ATTRIBUTEID_HAS_EXCLUSIVE:
561  case TNC_ATTRIBUTEID_HAS_SOH:
562  case TNC_ATTRIBUTEID_IFTNCCS_PROTOCOL:
563  case TNC_ATTRIBUTEID_IFTNCCS_VERSION:
564  case TNC_ATTRIBUTEID_IFT_PROTOCOL:
565  case TNC_ATTRIBUTEID_IFT_VERSION:
566  break;
567 
568  /* these attributes are unsupported or unknown */
569  case TNC_ATTRIBUTEID_DHPN:
570  case TNC_ATTRIBUTEID_TLS_UNIQUE:
571  default:
572  return TNC_RESULT_INVALID_PARAMETER;
573  }
574  }
575 
576  /* attributes specific to the TNCC or TNCS are unsupported */
577  if (id == TNC_CONNECTIONID_ANY)
578  {
579  return TNC_RESULT_INVALID_PARAMETER;
580  }
581 
582  this->connection_lock->read_lock(this->connection_lock);
583  enumerator = this->connections->create_enumerator(this->connections);
584  while (enumerator->enumerate(enumerator, &entry))
585  {
586  if (id == entry->id)
587  {
588  entry_found = TRUE;
589  break;
590  }
591  }
592  enumerator->destroy(enumerator);
593  this->connection_lock->unlock(this->connection_lock);
594 
595  if (!entry_found)
596  {
597  return TNC_RESULT_INVALID_PARAMETER;
598  }
599 
600  switch (attribute_id)
601  {
602  case TNC_ATTRIBUTEID_PREFERRED_LANGUAGE:
603  {
604  recommendations_t *recs;
605  chunk_t pref_lang;
606 
607  recs = entry->recs;
608  if (!recs)
609  {
610  return TNC_RESULT_INVALID_PARAMETER;
611  }
612  pref_lang = recs->get_preferred_language(recs);
613  if (pref_lang.len == 0)
614  {
615  return TNC_RESULT_INVALID_PARAMETER;
616  }
617  *value_len = pref_lang.len;
618  if (buffer && buffer_len >= pref_lang.len)
619  {
620  memcpy(buffer, pref_lang.ptr, pref_lang.len);
621  }
622  return TNC_RESULT_SUCCESS;
623  }
624  case TNC_ATTRIBUTEID_MAX_ROUND_TRIPS:
625  return uint_attribute(buffer_len, buffer, value_len,
626  0xffffffff);
627  case TNC_ATTRIBUTEID_MAX_MESSAGE_SIZE:
628  return uint_attribute(buffer_len, buffer, value_len,
629  entry->max_msg_len);
630  case TNC_ATTRIBUTEID_HAS_LONG_TYPES:
631  case TNC_ATTRIBUTEID_HAS_EXCLUSIVE:
632  return bool_attribute(buffer_len, buffer, value_len,

648  protocol = "IF-TNCCS-SOH";

654  }

658

662  version = "1.1";
663  break;
664  case TNCCS_2_0:
665  version = "2.0";
666  break;
667  case TNCCS_SOH:
668  version = "1.0";
669  break;
670  default:
671  return TNC_RESULT_INVALID_PARAMETER;
672  }
673  return str_attribute(buffer_len, buffer, value_len, version);
674  }
675  case TNC_ATTRIBUTEID_IFT_PROTOCOL:
676  {
677  char *protocol;
678

680  {
681  case TNC_IFT_EAP_1_0:

685  break;

687  case TNC_IFT_TLS_2_0:
688  protocol = "IF-T for TLS";
689  break;
690  default:
691  return TNC_RESULT_INVALID_PARAMETER;
692  }
693  return str_attribute(buffer_len, buffer, value_len, protocol);
694  }
695  case TNC_ATTRIBUTEID_IFT_VERSION:
696  {
697  char *version;
698 
699  switch (entry->tnccs->get_transport(entry->tnccs))
700  {
701  case TNC_IFT_EAP_1_0:
702  case TNC_IFT_TLS_1_0:
703  version = "1.0";
704  break;
705  case TNC_IFT_EAP_1_1:
706  version = "1.1";
707  break;
708  case TNC_IFT_EAP_2_0:
709  case TNC_IFT_TLS_2_0:
710  version = "2.0";
711  break;
712  default:
713  return TNC_RESULT_INVALID_PARAMETER;
714  }
715  return str_attribute(buffer_len, buffer, value_len, version);
716  }
717  case TNC_ATTRIBUTEID_AR_IDENTITIES:
718  {
719  linked_list_t *list;
720  identification_t *peer_id;
721  host_t *peer_ip;
722  tnccs_t *tnccs;
723  tncif_identity_t *tnc_id;
724  u_int32_t id_type, subject_type;
725  chunk_t id_value;
726  char *id_str;
727  TNC_Result result;
728 
729  list = linked_list_create();
730  tnccs = entry->tnccs;
731 
732  peer_id = tnccs->tls.is_server(&tnccs->tls) ?
733  tnccs->tls.get_peer_id(&tnccs->tls) :
734  tnccs->tls.get_server_id(&tnccs->tls);
735  if (peer_id)
736  {
737  switch (peer_id->get_type(peer_id))
738  {
739  case ID_IPV4_ADDR:
740  id_type = TNC_ID_IPV4_ADDR;
741  subject_type = TNC_SUBJECT_MACHINE;
742  break;
743  case ID_IPV6_ADDR:
744  id_type = TNC_ID_IPV6_ADDR;
745  subject_type = TNC_SUBJECT_MACHINE;
746  break;
747  case ID_FQDN:
748  id_type = TNC_ID_USERNAME;
749  subject_type = TNC_SUBJECT_USER;

802  }
