56 static bool push_env(char *envp[], u_int count, char *fmt, ...)
57 {
58  int i = 0;
59  char *str;
60  va_list args;
61 
62  while (envp[i])
63  {
64  if (++i + 1 >= count)
65  {
66  return FALSE;
67  }
68  }
69  va_start(args, fmt);
70  if (vasprintf(&str, fmt, args) >= 0)

123  ike_sa->get_other_eap_id(ike_sa));

142  }
