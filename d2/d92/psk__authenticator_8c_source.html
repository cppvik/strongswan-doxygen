112  recv_auth_data = auth_payload->get_data(auth_payload);
113  my_id = this->ike_sa->get_my_id(this->ike_sa);
114  other_id = this->ike_sa->get_other_id(this->ike_sa);
115  enumerator = lib->credmgr->create_shared_enumerator(lib->credmgr,
116  SHARED_IKE, my_id, other_id);
117  while (!authenticated && enumerator->enumerate(enumerator, &key, NULL, NULL))
118  {
119  keys_found++;
120 
121  if (!keymat->get_psk_sig(keymat, TRUE, this->ike_sa_init, this->nonce,
122  key->get_key(key), other_id, this->reserved, &auth_data))
123  {
124  continue;
125  }
126  if (auth_data.len && chunk_equals_const(auth_data, recv_auth_data))
127  {
128  DBG1(DBG_IKE, "authentication of '%Y' with %N successful",
129  other_id, auth_method_names, AUTH_PSK);
130  authenticated = TRUE;
131  }
132  chunk_free(&auth_data);
133  }
134  enumerator->destroy(enumerator);
135 
136  if (!authenticated)
137  {
138  if (keys_found == 0)
139  {
140  DBG1(DBG_IKE, "no shared key found for '%Y' - '%Y'", my_id, other_id);
141  return NOT_FOUND;
142  }
143  DBG1(DBG_IKE, "tried %d shared key%s for '%Y' - '%Y', but MAC mismatched",
144  keys_found, keys_found == 1 ? "" : "s", my_id, other_id);
145  return FAILED;
146  }
147 
148  auth = this->ike_sa->get_auth_cfg(this->ike_sa, FALSE);
149  auth->add(auth, AUTH_RULE_AUTH_CLASS, AUTH_CLASS_PSK);
150  return SUCCESS;
151 }
