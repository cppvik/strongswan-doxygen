96 METHOD(mac_t, get_mac, bool,
97  private_mac_t *this, chunk_t data, u_int8_t *out)
98 {
99  if (!this->key_set)
100  {
101  return FALSE;
102  }
103 #if OPENSSL_VERSION_NUMBER >= 0x10000000L
104  if (!HMAC_Update(&this->hmac, data.ptr, data.len))
105  {
106  return FALSE;
107  }
108  if (out == NULL)
109  {
110  return TRUE;
111  }
112  if (!HMAC_Final(&this->hmac, out, NULL))
113  {
114  return FALSE;
115  }
116 #else /* OPENSSL_VERSION_NUMBER < 1.0 */
117  HMAC_Update(&this->hmac, data.ptr, data.len);
118  if (out == NULL)
119  {
120  return TRUE;
121  }
122  HMAC_Final(&this->hmac, out, NULL);
123 #endif
124  return set_key(this, chunk_empty);
125 }
