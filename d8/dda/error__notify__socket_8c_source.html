78 
79  this->mutex->lock(this->mutex);
80  enumerator = this->connected->create_enumerator(this->connected);
81  while (enumerator->enumerate(enumerator, &stream))
82  {
83  if (!stream->write_all(stream, msg, sizeof(*msg)))
84  {
85  switch (errno)
