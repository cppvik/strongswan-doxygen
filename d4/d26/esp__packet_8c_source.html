170 static bool remove_padding(private_esp_packet_t *this, chunk_t plaintext)
171 {
172  u_int8_t next_header, pad_length;
173  chunk_t padding, payload;
174  bio_reader_t *reader;
175 
176  reader = bio_reader_create(plaintext);
177  if (!reader->read_uint8_end(reader, &next_header) ||
178  !reader->read_uint8_end(reader, &pad_length))
179  {
180  DBG1(DBG_ESP, "parsing ESP payload failed: invalid length");
181  goto failed;
182  }
183  if (!reader->read_data_end(reader, pad_length, &padding) ||
184  !check_padding(padding))
185  {
186  DBG1(DBG_ESP, "parsing ESP payload failed: invalid padding");
187  goto failed;
188  }
189  this->payload = ip_packet_create(reader->peek(reader));
190  reader->destroy(reader);
191  if (!this->payload)
192  {
193  DBG1(DBG_ESP, "parsing ESP payload failed: unsupported payload");
194  return FALSE;
195  }
196  this->next_header = next_header;
197  payload = this->payload->get_encoding(this->payload);
198 
199  DBG3(DBG_ESP, "ESP payload:\n payload %B\n padding %B\n "
200  "padding length = %hhu, next header = %hhu", &payload, &padding,
201  pad_length, this->next_header);
202  return TRUE;
203 
204 failed:
205  reader->destroy(reader);
206  chunk_free(&plaintext);
207  return FALSE;
208 }

210 METHOD(esp_packet_t, decrypt, status_t,
211  private_esp_packet_t *this, esp_context_t *esp_context)
212 {
213  bio_reader_t *reader;
214  u_int32_t spi, seq;
215  chunk_t data, iv, icv, aad, ciphertext, plaintext;
216  aead_t *aead;
217 
218  DESTROY_IF(this->payload);
219  this->payload = NULL;
220 
221  data = this->packet->get_data(this->packet);
222  aead = esp_context->get_aead(esp_context);
223 
224  reader = bio_reader_create(data);
225  if (!reader->read_uint32(reader, &spi) ||
226  !reader->read_uint32(reader, &seq) ||
227  !reader->read_data(reader, aead->get_iv_size(aead), &iv) ||
228  !reader->read_data_end(reader, aead->get_icv_size(aead), &icv) ||
229  reader->remaining(reader) % aead->get_block_size(aead))
230  {
231  DBG1(DBG_ESP, "ESP decryption failed: invalid length");
232  return PARSE_ERROR;
233  }
234  ciphertext = reader->peek(reader);
235  reader->destroy(reader);
236 
237  if (!esp_context->verify_seqno(esp_context, seq))
238  {
239  DBG1(DBG_ESP, "ESP sequence number verification failed:\n "
240  "src %H, dst %H, SPI %.8x [seq %u]",
241  get_source(this), get_destination(this), spi, seq);
242  return VERIFY_ERROR;
243  }
244  DBG3(DBG_ESP, "ESP decryption:\n SPI %.8x [seq %u]\n IV %B\n "
245  "encrypted %B\n ICV %B", spi, seq, &iv, &ciphertext, &icv);
246 
247  /* include ICV in ciphertext for decryption/verification */
248  ciphertext.len += icv.len;
249  /* aad = spi + seq */
250  aad = chunk_create(data.ptr, 8);
251 
252  if (!aead->decrypt(aead, ciphertext, aad, iv, &plaintext))
253  {
254  DBG1(DBG_ESP, "ESP decryption or ICV verification failed");
255  return FAILED;
256  }
257  esp_context->set_authenticated_seqno(esp_context, seq);
258 
259  if (!remove_padding(this, plaintext))
260  {
261  return PARSE_ERROR;
262  }
263  return SUCCESS;
264 }

279 METHOD(esp_packet_t, encrypt, status_t,
280  private_esp_packet_t *this, esp_context_t *esp_context, u_int32_t spi)
281 {
282  chunk_t iv, icv, aad, padding, payload, ciphertext;
283  bio_writer_t *writer;
284  u_int32_t next_seqno;
285  size_t blocksize, plainlen;
286  aead_t *aead;
287  iv_gen_t *iv_gen;
288 
289  this->packet->set_data(this->packet, chunk_empty);
290 
291  if (!esp_context->next_seqno(esp_context, &next_seqno))
292  {
293  DBG1(DBG_ESP, "ESP encapsulation failed: sequence numbers cycled");
294  return FAILED;
295  }
296 
297  aead = esp_context->get_aead(esp_context);
298  iv_gen = aead->get_iv_gen(aead);
299  if (!iv_gen)
300  {
301  DBG1(DBG_ESP, "ESP encryption failed: no IV generator");
302  return NOT_FOUND;
303  }
304 
305  blocksize = aead->get_block_size(aead);
306  iv.len = aead->get_iv_size(aead);
307  icv.len = aead->get_icv_size(aead);
308 
309  /* plaintext = payload, padding, pad_length, next_header */
310  payload = this->payload ? this->payload->get_encoding(this->payload)
311  : chunk_empty;
312  plainlen = payload.len + 2;
313  padding.len = pad_len(plainlen, blocksize);
314  /* ICV must be on a 4-byte boundary */
315  padding.len += pad_len(iv.len + plainlen + padding.len, 4);
316  plainlen += padding.len;
317 
318  /* len = spi, seq, IV, plaintext, ICV */
319  writer = bio_writer_create(2 * sizeof(u_int32_t) + iv.len + plainlen +
320  icv.len);
321  writer->write_uint32(writer, ntohl(spi));
322  writer->write_uint32(writer, next_seqno);
323 
324  iv = writer->skip(writer, iv.len);
325  if (!iv_gen->get_iv(iv_gen, next_seqno, iv.len, iv.ptr))
326  {
327  DBG1(DBG_ESP, "ESP encryption failed: could not generate IV");
328  writer->destroy(writer);
329  return FAILED;
330  }
331 
332  /* plain-/ciphertext will start here */
333  ciphertext = writer->get_buf(writer);
334  ciphertext.ptr += ciphertext.len;
335  ciphertext.len = plainlen;
336 
337  writer->write_data(writer, payload);
338 
339  padding = writer->skip(writer, padding.len);
340  generate_padding(padding);
341 
342  writer->write_uint8(writer, padding.len);
343  writer->write_uint8(writer, this->next_header);
344 
345  /* aad = spi + seq */
346  aad = writer->get_buf(writer);
347  aad.len = 8;
348  icv = writer->skip(writer, icv.len);
349 
350  DBG3(DBG_ESP, "ESP before encryption:\n payload = %B\n padding = %B\n "
351  "padding length = %hhu, next header = %hhu", &payload, &padding,
352  (u_int8_t)padding.len, this->next_header);
353 
354  /* encrypt/authenticate the content inline */
355  if (!aead->encrypt(aead, ciphertext, aad, iv, NULL))
356  {
357  DBG1(DBG_ESP, "ESP encryption or ICV generation failed");
358  writer->destroy(writer);
359  return FAILED;
360  }
361 
362  DBG3(DBG_ESP, "ESP packet:\n SPI %.8x [seq %u]\n IV %B\n "
363  "encrypted %B\n ICV %B", ntohl(spi), next_seqno, &iv,
364  &ciphertext, &icv);
365 
366  this->packet->set_data(this->packet, writer->extract_buf(writer));
367  writer->destroy(writer);
368  return SUCCESS;
369 }
