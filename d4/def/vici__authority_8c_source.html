337  request_data_t *request, vici_message_t *message,

464 CALLBACK(list_authorities, vici_message_t*,
465  private_vici_authority_t *this, char *name, u_int id, vici_message_t *request)
466 {
467  enumerator_t *enumerator, *e;
468  authority_t *authority;
469  vici_builder_t *b;
470  char *str, *uri;
471 
472  str = request->get_str(request, NULL, "name");
473 
474  this->lock->read_lock(this->lock);
475  enumerator = this->authorities->create_enumerator(this->authorities);
476  while (enumerator->enumerate(enumerator, &authority))
477  {
478  if (str && !streq(str, authority->name))
479  {
480  continue;
481  }
482  b = vici_builder_create();
483 
484  /* open authority section */
485  b->begin_section(b, authority->name);
486 
487  /* subject DN of cacert */
488  b->add_kv(b, "cacert", "%Y",
489  authority->cert->get_subject(authority->cert));
490 
491  /* list of crl_uris */
492  b->begin_list(b, "crl_uris");
493  e = authority->crl_uris->create_enumerator(authority->crl_uris);
494  while (e->enumerate(e, &uri))
495  {
496  b->add_li(b, "%s", uri);
497  }
498  e->destroy(e);
499  b->end_list(b);
500 
501  /* list of ocsp_uris */
502  b->begin_list(b, "ocsp_uris");
503  e = authority->ocsp_uris->create_enumerator(authority->ocsp_uris);
504  while (e->enumerate(e, &uri))
505  {
506  b->add_li(b, "%s", uri);
507  }
508  e->destroy(e);
509  b->end_list(b);
510 
511  /* cert_uri_base */
512  if (authority->cert_uri_base)
513  {
514  b->add_kv(b, "cert_uri_base", "%s", authority->cert_uri_base);
515  }
516 
517  /* close authority and raise event */
518  b->end_section(b);
519  this->dispatcher->raise_event(this->dispatcher, "list-authority", id,
520  b->finalize(b));
521  }
522  enumerator->destroy(enumerator);
523  this->lock->unlock(this->lock);
524 
525  b = vici_builder_create();
526  return b->finalize(b);
527 }

607 static enumerator_t *create_inner_cdp_hashandurl(authority_t *authority,
