333 {
334  static bool once = FALSE;
335  malloc_zone_t *zone;
336  void *page;
337 
338  if (once)
339  {
340  return TRUE;
341  }

629 static int print_traces(private_leak_detective_t *this,
630  leak_detective_report_cb_t cb, void *user,
631  int thresh, int thresh_count,
632  bool detailed, int *whitelisted, size_t *sum)
633 {
634  int leaks = 0;
635  memory_header_t *hdr;
636  enumerator_t *enumerator;
637  hashtable_t *entries;
638  struct {
640  backtrace_t *backtrace;
642  size_t bytes;
644  u_int count;
645  } *entry;
646  bool before;
647 
648  before = enable_thread(FALSE);
649 
650  entries = hashtable_create((hashtable_hash_t)hash,
651  (hashtable_equals_t)equals, 1024);
652  lock->lock(lock);
653  for (hdr = first_header.next; hdr != NULL; hdr = hdr->next)
654  {
655  if (whitelisted &&
656  hdr->backtrace->contains_function(hdr->backtrace,
657  whitelist, countof(whitelist)))
658  {
659  (*whitelisted)++;
660  continue;
661  }
662  entry = entries->get(entries, hdr->backtrace);
663  if (entry)
664  {
665  entry->bytes += hdr->bytes;
666  entry->count++;
667  }
668  else
669  {
670  INIT(entry,
671  .backtrace = hdr->backtrace->clone(hdr->backtrace),
672  .bytes = hdr->bytes,
673  .count = 1,
674  );
675  entries->put(entries, entry->backtrace, entry);
676  }
677  if (sum)
678  {
679  *sum += hdr->bytes;
680  }
681  leaks++;
682  }
683  lock->unlock(lock);
684 
685  enumerator = entries->create_enumerator(entries);
686  while (enumerator->enumerate(enumerator, NULL, &entry))
687  {
688  if (cb)
689  {
690  if (!thresh || entry->bytes >= thresh)
691  {
692  if (!thresh_count || entry->count >= thresh_count)
693  {
694  cb(user, entry->count, entry->bytes, entry->backtrace,
695  detailed);
696  }
697  }
698  }
699  entry->backtrace->destroy(entry->backtrace);
700  free(entry);
701  }
702  enumerator->destroy(enumerator);
703  entries->destroy(entries);
704 
705  enable_thread(before);
706  return leaks;
707 }

831 HOOK(void, free, void *ptr)
832 {
833  memory_header_t *hdr;
834  memory_tail_t *tail;
835  backtrace_t *backtrace;
836  bool before;
837 
838  if (!enabled || thread_disabled->get(thread_disabled))
839  {
840  real_free(ptr);
841  return;
842  }
843  /* allow freeing of NULL */
844  if (ptr == NULL)
845  {
846  return;
847  }
848  hdr = ptr - sizeof(memory_header_t);
849  tail = ptr + hdr->bytes;
850 
851  before = enable_thread(FALSE);
852  if (hdr->magic != MEMORY_HEADER_MAGIC ||
853  tail->magic != MEMORY_TAIL_MAGIC)
854  {
855  if (has_hdr(hdr))
856  {
857  /* memory was allocated by our hooks but is corrupted */
858  fprintf(stderr, "freeing corrupted memory (%p): "
859  "header magic 0x%x, tail magic 0x%x:\n",
860  ptr, hdr->magic, tail->magic);
861  }
862  else
863  {
864  /* memory was not allocated by our hooks */
865  fprintf(stderr, "freeing invalid memory (%p)\n", ptr);
866  }
867  backtrace = backtrace_create(2);
868  backtrace->log(backtrace, stderr, TRUE);
869  backtrace->destroy(backtrace);
870  }
871  else
872  {
873  remove_hdr(hdr);
874 
875  hdr->backtrace->destroy(hdr->backtrace);
876 
877  /* clear MAGIC, set mem to something remarkable */
878  memset(hdr, MEMORY_FREE_PATTERN,
879  sizeof(memory_header_t) + hdr->bytes + sizeof(memory_tail_t));
880 
881  real_free(hdr);
882  }
883  enable_thread(before);
884 }

918  tail->magic != MEMORY_TAIL_MAGIC)

923  backtrace->log(backtrace, stderr, TRUE);
924  backtrace->destroy(backtrace);

929  memset(&tail->magic, MEMORY_ALLOC_PATTERN, sizeof(tail->magic));
