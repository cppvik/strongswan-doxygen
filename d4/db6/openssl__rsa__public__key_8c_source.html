56 static bool verify_emsa_pkcs1_signature(private_openssl_rsa_public_key_t *this,
57  int type, chunk_t data, chunk_t signature)
58 {
59  bool valid = FALSE;
60  int rsa_size = RSA_size(this->rsa);
61 
62  /* OpenSSL expects a signature of exactly RSA size (no leading 0x00) */
63  if (signature.len > rsa_size)
64  {
65  signature = chunk_skip(signature, signature.len - rsa_size);
66  }
67 
68  if (type == NID_undef)
69  {
70  char *buf;
71  int len;
72 
73  buf = malloc(rsa_size);
74  len = RSA_public_decrypt(signature.len, signature.ptr, buf, this->rsa,
75  RSA_PKCS1_PADDING);
76  if (len != -1)
77  {
78  valid = chunk_equals_const(data, chunk_create(buf, len));
79  }
80  free(buf);
81  }
82  else
83  {
84  EVP_MD_CTX *ctx;
85  EVP_PKEY *key;
86  const EVP_MD *hasher;
87 
88  hasher = EVP_get_digestbynid(type);
89  if (!hasher)
90  {
91  return FALSE;
92  }
93 
94  ctx = EVP_MD_CTX_create();
95  key = EVP_PKEY_new();
96 
97  if (!ctx || !key)
98  {
99  goto error;
100  }
101  if (!EVP_PKEY_set1_RSA(key, this->rsa))
102  {
103  goto error;
104  }
105  if (!EVP_VerifyInit_ex(ctx, hasher, NULL))
106  {
107  goto error;
108  }
109  if (!EVP_VerifyUpdate(ctx, data.ptr, data.len))
110  {
111  goto error;
112  }
113  valid = (EVP_VerifyFinal(ctx, signature.ptr, signature.len, key) == 1);
114 
115 error:
116  if (key)
117  {
118  EVP_PKEY_free(key);
119  }
120  if (ctx)
121  {
122  EVP_MD_CTX_destroy(ctx);
123  }
124  }
125  return valid;
126 }

182  len = RSA_public_encrypt(plain.len, plain.ptr, encrypted,

203 bool openssl_rsa_fingerprint(RSA *rsa, cred_encoding_type_t type, chunk_t *fp)
204 {
205  hasher_t *hasher;
206  chunk_t key;
207  u_char *p;
208 
209  if (lib->encoding->get_cache(lib->encoding, type, rsa, fp))
210  {
211  return TRUE;
212  }
213  switch (type)
214  {
215  case KEYID_PUBKEY_SHA1:
216  key = chunk_alloc(i2d_RSAPublicKey(rsa, NULL));
217  p = key.ptr;
218  i2d_RSAPublicKey(rsa, &p);
219  break;
220  case KEYID_PUBKEY_INFO_SHA1:
221  key = chunk_alloc(i2d_RSA_PUBKEY(rsa, NULL));
222  p = key.ptr;
223  i2d_RSA_PUBKEY(rsa, &p);
224  break;
225  default:
226  {
227  chunk_t n = chunk_empty, e = chunk_empty;
228  bool success = FALSE;
229 
230  if (openssl_bn2chunk(rsa->n, &n) &&
231  openssl_bn2chunk(rsa->e, &e))
232  {
233  success = lib->encoding->encode(lib->encoding, type, rsa, fp,
234  CRED_PART_RSA_MODULUS, n,
235  CRED_PART_RSA_PUB_EXP, e, CRED_PART_END);
236  }
237  chunk_free(&n);
238  chunk_free(&e);
239  return success;
240  }
241  }
242  hasher = lib->crypto->create_hasher(lib->crypto, HASH_SHA1);
243  if (!hasher || !hasher->allocate_hash(hasher, key, fp))
244  {
245  DBG1(DBG_LIB, "SHA1 hash algorithm not supported, fingerprinting failed");
246  DESTROY_IF(hasher);
247  free(key.ptr);
248  return FALSE;
249  }
250  free(key.ptr);
251  hasher->destroy(hasher);
252  lib->encoding->cache(lib->encoding, type, rsa, *fp);
253  return TRUE;
254 }

368 openssl_rsa_public_key_t *openssl_rsa_public_key_load(key_type_t type,
369  va_list args)
370 {
371  private_openssl_rsa_public_key_t *this;
372  chunk_t blob, n, e;
373 
374  n = e = blob = chunk_empty;
375  while (TRUE)
376  {
377  switch (va_arg(args, builder_part_t))
378  {
379  case BUILD_BLOB_ASN1_DER:
380  blob = va_arg(args, chunk_t);
381  continue;
382  case BUILD_RSA_MODULUS:
383  n = va_arg(args, chunk_t);
384  continue;
385  case BUILD_RSA_PUB_EXP:
386  e = va_arg(args, chunk_t);
387  continue;
388  case BUILD_END:
389  break;
390  default:
391  return NULL;
392  }
393  break;
394  }
395 
396  this = create_empty();
397  if (blob.ptr)
398  {
399  switch (type)
400  {
401  case KEY_ANY:
402  this->rsa = d2i_RSA_PUBKEY(NULL, (const u_char**)&blob.ptr,
403  blob.len);
404  break;
405  case KEY_RSA:
406  this->rsa = d2i_RSAPublicKey(NULL, (const u_char**)&blob.ptr,
407  blob.len);
408  break;
409  default:
410  break;
411  }
412  if (this->rsa)
413  {
414  return &this->public;
415  }
416  }
417  else if (n.ptr && e.ptr && type == KEY_RSA)
418  {
419  this->rsa = RSA_new();
420  this->rsa->n = BN_bin2bn((const u_char*)n.ptr, n.len, NULL);
421  this->rsa->e = BN_bin2bn((const u_char*)e.ptr, e.len, NULL);
422  return &this->public;
423  }
424  destroy(this);
425  return NULL;
426 }
