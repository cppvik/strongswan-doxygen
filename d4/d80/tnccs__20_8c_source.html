171 METHOD(tls_t, process, status_t,
172  private_tnccs_20_t *this, void *buf, size_t buflen)
173 {
174  pb_tnc_batch_t *batch;
175  bool from_server, fatal_header_error = FALSE;
176  status_t status;
177  chunk_t data;
178 
179  /* On arrival of first batch from TNC client create TNC server */
180  if (this->is_server && !this->tnc_server)
181  {
182  this->tnc_server = tnccs_20_server_create(&this->public, _send_msg,
183  this->max_batch_len, this->max_msg_len,
184  this->eap_transport);
185  if (!this->tnc_server)
186  {
187  return FAILED;
188  }
189  this->tnccs_handler = this->tnc_server;
190  this->tnccs_handler->begin_handshake(this->tnccs_handler, FALSE);
191  }
192 
193  data = chunk_create(buf, buflen);
194  DBG1(DBG_TNC, "received TNCCS batch (%u bytes)", data.len);
195  DBG3(DBG_TNC, "%B", &data);
196 
197  /* Parse the header of the received PB-TNC batch */
198  batch = pb_tnc_batch_create_from_data(data);
199  status = batch->process_header(batch, !this->mutual, this->is_server,
200  &from_server);
201  if (status == FAILED)
202  {
203  fatal_header_error = TRUE;
204  status = VERIFY_ERROR;
205  }
206  this->to_server = this->mutual ? from_server : !this->is_server;
207 
208  /* In the mutual case, first batch from TNC server requires a TNC client */
209  if (this->to_server && !this->tnc_client)
210  {
211  this->tnc_client = tnccs_20_client_create(&this->public, _send_msg,
212  this->max_batch_len, this->max_msg_len);
213  if (!this->tnc_client)
214  {
215  batch->destroy(batch);
216  return FAILED;
217  }
218  this->tnccs_handler = this->tnc_client;
219  this->tnccs_handler->begin_handshake(this->tnccs_handler, this->mutual);
220  }
221  else
222  {
223  /* Set active TNCCS handler for processing */
224  this->tnccs_handler = this->to_server ? this->tnc_client :
225  this->tnc_server;
226  }
227  DBG2(DBG_TNC, "TNC %s is handling inbound connection",
228  this->to_server ? "client" : "server");
229 
230  if (status == SUCCESS)
231  {
232  status = this->tnccs_handler->process(this->tnccs_handler, batch);
233  }
234  if (status == VERIFY_ERROR)
235  {
236  this->tnccs_handler->handle_errors(this->tnccs_handler, batch,
237  fatal_header_error);
238  status = NEED_MORE;
239  }
240  batch->destroy(batch);
241 
242  /* Has a mutual connection been established? */
243  this->mutual = this->is_server ?
244  this->tnc_server->get_mutual(this->tnc_server) :
245  this->tnc_client->get_mutual(this->tnc_client);
246 
247  if (this->mutual && !this->is_server)
248  {
249  pb_tnc_state_t client_state, server_state;
250 
251  client_state = !this->tnc_client ? PB_STATE_INIT :
252  this->tnc_client->get_state(this->tnc_client);
253  server_state = !this->tnc_server ? PB_STATE_INIT :
254  this->tnc_server->get_state(this->tnc_server);
255 
256  /* In half-duplex mutual mode toggle the direction on the client side */
257  if ((!this->to_server && client_state != PB_STATE_DECIDED) ||
258  ( this->to_server && server_state != PB_STATE_END))
259  {
260  this->to_server = !this->to_server;
261  }
262  else if (client_state == PB_STATE_DECIDED &&
263  server_state == PB_STATE_END)
264  {
265  /* Cause the final CLOSE batch to be sent to the TNC server */
266  this->to_server = TRUE;
267  }
268 
269  /* Suppress a successful CLOSE batch coming from the TNC server */
270  if (status == SUCCESS)
271  {
272  status = NEED_MORE;
273  }
274  }
275 
276  return status;
277 }
