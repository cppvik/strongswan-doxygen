111 METHOD(payload_t, verify, status_t,
112  private_certreq_payload_t *this)
113 {
114  if (this->type == PLV2_CERTREQ &&
115  this->encoding == ENC_X509_SIGNATURE)
116  {
117  if (this->data.len % HASH_SIZE_SHA1)
118  {
119  DBG1(DBG_ENC, "invalid X509 hash length (%d) in certreq",
120  this->data.len);
121  return FAILED;
122  }
123  }
124  return SUCCESS;
125 }

192 METHOD(enumerator_t, keyid_enumerate, bool,
193  keyid_enumerator_t *this, chunk_t *chunk)
194 {
195  if (this->pos == NULL)
196  {
197  this->pos = this->full.ptr;
198  }
199  else
200  {
201  this->pos += HASH_SIZE_SHA1;
202  if (this->pos > (this->full.ptr + this->full.len - HASH_SIZE_SHA1))
203  {
204  this->pos = NULL;
205  }
206  }
207  if (this->pos)
208  {
209  chunk->ptr = this->pos;
210  chunk->len = HASH_SIZE_SHA1;
211  return TRUE;
212  }
213  return FALSE;
214 }
