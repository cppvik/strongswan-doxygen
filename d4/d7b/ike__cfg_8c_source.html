146 static host_t* resolve(linked_list_t *hosts, int family, u_int16_t port)
147 {
148  enumerator_t *enumerator;

150  bool tried = FALSE;
151  char *str;
152 
153  enumerator = hosts->create_enumerator(hosts);
154  while (enumerator->enumerate(enumerator, &str))
155  {
156  host = host_create_from_dns(str, family, port);

189 static u_int match(linked_list_t *hosts, linked_list_t *ranges, host_t *cand)
190 {
191  enumerator_t *enumerator;
192  traffic_selector_t *ts;
193  char *str;
194  host_t *host;
195  u_int8_t mask;
196  u_int quality = 0;
197 
198  /* try single hosts first */
199  enumerator = hosts->create_enumerator(hosts);
200  while (enumerator->enumerate(enumerator, &str))
201  {
202  host = host_create_from_dns(str, cand->get_family(cand), 0);
203  if (host)
204  {
205  if (host->ip_equals(host, cand))
206  {
207  quality = max(quality, 128 + 1);
208  }
209  if (host->is_anyaddr(host))
210  {
211  quality = max(quality, 1);
212  }
213  host->destroy(host);
214  }
215  }
216  enumerator->destroy(enumerator);
217 
218  /* then ranges/subnets */
219  enumerator = ranges->create_enumerator(ranges);
220  while (enumerator->enumerate(enumerator, &ts))
221  {
222  if (ts->includes(ts, cand))
223  {
224  if (ts->to_subnet(ts, &host, &mask))
225  {
226  quality = max(quality, mask + 1);
227  host->destroy(host);
228  }
229  else
230  {
231  quality = max(quality, 1);
232  }
233  }
234  }
235  enumerator->destroy(enumerator);
236 
237  return quality;
238 }

312 METHOD(ike_cfg_t, select_proposal, proposal_t*,
313  private_ike_cfg_t *this, linked_list_t *proposals, bool private)
314 {
315  enumerator_t *stored_enum, *supplied_enum;
316  proposal_t *stored, *supplied, *selected;
317 
318  stored_enum = this->proposals->create_enumerator(this->proposals);
319  supplied_enum = proposals->create_enumerator(proposals);
320 
321 
322  /* compare all stored proposals with all supplied. Stored ones are preferred.*/
323  while (stored_enum->enumerate(stored_enum, (void**)&stored))
324  {
325  proposals->reset_enumerator(proposals, supplied_enum);
326 
327  while (supplied_enum->enumerate(supplied_enum, (void**)&supplied))
328  {
329  selected = stored->select(stored, supplied, private);
330  if (selected)
331  {
332  /* they match, return */
333  stored_enum->destroy(stored_enum);
334  supplied_enum->destroy(supplied_enum);
335  DBG2(DBG_CFG, "received proposals: %#P", proposals);
336  DBG2(DBG_CFG, "configured proposals: %#P", this->proposals);
337  DBG2(DBG_CFG, "selected proposal: %P", selected);
338  return selected;
339  }
340  }
341  }
342  /* no proposal match :-(, will result in a NO_PROPOSAL_CHOSEN... */
343  stored_enum->destroy(stored_enum);
344  supplied_enum->destroy(supplied_enum);
345  DBG1(DBG_CFG, "received proposals: %#P", proposals);
346  DBG1(DBG_CFG, "configured proposals: %#P", this->proposals);
347 
348  return NULL;

371  private_ike_cfg_t *this, ike_cfg_t *other_public)
372 {
373  private_ike_cfg_t *other = (private_ike_cfg_t*)other_public;
374  enumerator_t *e1, *e2;
375  proposal_t *p1, *p2;
376  bool eq = TRUE;
377 
378  if (this == other)
379  {
380  return TRUE;
381  }

384  return FALSE;
385  }
386  if (this->proposals->get_count(this->proposals) !=
387  other->proposals->get_count(other->proposals))
388  {
389  return FALSE;
390  }
391  e1 = this->proposals->create_enumerator(this->proposals);
392  e2 = other->proposals->create_enumerator(other->proposals);
393  while (e1->enumerate(e1, &p1) && e2->enumerate(e2, &p2))
394  {
395  if (!p1->equals(p1, p2))
396  {
397  eq = FALSE;
398  break;
399  }
400  }
401  e1->destroy(e1);
402  e2->destroy(e2);
403 
404  return (eq &&
405  this->version == other->version &&
406  this->certreq == other->certreq &&
407  this->force_encap == other->force_encap &&
408  this->fragmentation == other->fragmentation &&
409  streq(this->me, other->me) &&
410  streq(this->other, other->other) &&
411  this->my_port == other->my_port &&
412  this->other_port == other->other_port);

488 static void parse_addresses(char *str, linked_list_t *hosts,
489  linked_list_t *ranges)
490 {
491  enumerator_t *enumerator;
492  traffic_selector_t *ts;
493 
494  enumerator = enumerator_create_token(str, ",", " ");
495  while (enumerator->enumerate(enumerator, &str))
496  {
497  ts = make_subnet(str);
498  if (ts)
499  {
500  ranges->insert_last(ranges, ts);
501  continue;
502  }
503  ts = make_range(str);
504  if (ts)
505  {
506  ranges->insert_last(ranges, ts);
507  continue;
508  }
509  hosts->insert_last(hosts, strdup(str));
510  }
511  enumerator->destroy(enumerator);
512 }

534  {

537  continue;

540  if (host)

544  family = host->get_family(host);

549  family = AF_UNSPEC;

552  }
553  }
554  DESTROY_IF(host);
555  }
556  enumerator->destroy(enumerator);
557  return family;
