589 static enumerator_t *create_lease_query(char *filter)
590 {
591  enumerator_t *query;
592  identification_t *id = NULL;
593  host_t *addr = NULL;
594  u_int tstamp = 0;
595  bool online = FALSE, valid = FALSE, expired = FALSE;
596  char *value, *pos, *pool = NULL;
597  enum {
598  FIL_POOL = 0,
599  FIL_ID,
600  FIL_ADDR,
601  FIL_TSTAMP,
602  FIL_STATE,
603  };
604  char *const token[] = {
605  [FIL_POOL] = "pool",
606  [FIL_ID] = "id",
607  [FIL_ADDR] = "addr",
608  [FIL_TSTAMP] = "tstamp",
609  [FIL_STATE] = "status",
610  NULL
611  };
612 
613  /* if the filter string contains a distinguished name as a ID, we replace
614  * ", " by "/ " in order to not confuse the getsubopt parser */
615  pos = filter;
616  while ((pos = strchr(pos, ',')))
617  {
618  if (pos[1] == ' ')
619  {
620  pos[0] = '/';
621  }
622  pos++;
623  }
624 
625  while (filter && *filter != '\0')
626  {
627  switch (getsubopt(&filter, token, &value))
628  {
629  case FIL_POOL:
630  if (value)
631  {
632  pool = value;
633  }
634  break;
635  case FIL_ID:
636  if (value)
637  {
638  id = identification_create_from_string(value);
639  }
640  break;
641  case FIL_ADDR:
642  if (value)
643  {
644  addr = host_create_from_string(value, 0);
645  }
646  if (!addr)
647  {
648  fprintf(stderr, "invalid 'addr' in filter string.\n");
649  exit(EXIT_FAILURE);
650  }
651  break;
652  case FIL_TSTAMP:
653  if (value)
654  {
655  tstamp = atoi(value);
656  }
657  if (tstamp == 0)
658  {
659  online = TRUE;
660  }
661  break;
662  case FIL_STATE:
663  if (value)
664  {
665  if (streq(value, "online"))
666  {
667  online = TRUE;
668  }
669  else if (streq(value, "valid"))
670  {
671  valid = TRUE;
672  }
673  else if (streq(value, "expired"))
674  {
675  expired = TRUE;
676  }
677  else
678  {
679  fprintf(stderr, "invalid 'state' in filter string.\n");
680  exit(EXIT_FAILURE);
681  }
682  }
683  break;
684  default:
685  fprintf(stderr, "invalid filter string.\n");
686  exit(EXIT_FAILURE);
687  break;
688  }
689  }
690  query = db->query(db,
691  "SELECT name, addresses.address, identities.type, "
692  "identities.data, leases.acquired, leases.released, timeout "
693  "FROM leases JOIN addresses ON leases.address = addresses.id "
694  "JOIN pools ON addresses.pool = pools.id "
695  "JOIN identities ON leases.identity = identities.id "
696  "WHERE (? OR name = ?) "
697  "AND (? OR (identities.type = ? AND identities.data = ?)) "
698  "AND (? OR addresses.address = ?) "
699  "AND (? OR (? >= leases.acquired AND (? <= leases.released))) "
700  "AND (? OR leases.released > ? - timeout) "
701  "AND (? OR leases.released < ? - timeout) "
702  "AND ? "
703  "UNION "
704  "SELECT name, address, identities.type, identities.data, "
705  "acquired, released, timeout FROM addresses "
706  "JOIN pools ON addresses.pool = pools.id "
707  "JOIN identities ON addresses.identity = identities.id "
708  "WHERE ? AND released = 0 "
709  "AND (? OR name = ?) "
710  "AND (? OR (identities.type = ? AND identities.data = ?)) "
711  "AND (? OR address = ?)",
712  DB_INT, pool == NULL, DB_TEXT, pool,
713  DB_INT, id == NULL,
714  DB_INT, id ? id->get_type(id) : 0,
715  DB_BLOB, id ? id->get_encoding(id) : chunk_empty,
716  DB_INT, addr == NULL,
717  DB_BLOB, addr ? addr->get_address(addr) : chunk_empty,
718  DB_INT, tstamp == 0, DB_UINT, tstamp, DB_UINT, tstamp,
719  DB_INT, !valid, DB_INT, time(NULL),
720  DB_INT, !expired, DB_INT, time(NULL),
721  DB_INT, !online,
722  /* union */
723  DB_INT, !(valid || expired),
724  DB_INT, pool == NULL, DB_TEXT, pool,
725  DB_INT, id == NULL,
726  DB_INT, id ? id->get_type(id) : 0,
727  DB_BLOB, id ? id->get_encoding(id) : chunk_empty,
728  DB_INT, addr == NULL,
729  DB_BLOB, addr ? addr->get_address(addr) : chunk_empty,
730  /* res */
731  DB_TEXT, DB_BLOB, DB_INT, DB_BLOB, DB_UINT, DB_UINT, DB_UINT);
732  /* id and addr leak but we can't destroy them until query is destroyed. */
733  return query;
734 }
