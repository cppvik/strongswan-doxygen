306 static status_t create_plugin(private_plugin_loader_t *this, void *handle,
307  char *name, bool integrity, bool critical,
308  plugin_entry_t **entry)
309 {
310  char create[128];
311  plugin_t *plugin;
312  plugin_constructor_t constructor;
313 
314  if (snprintf(create, sizeof(create), "%s_plugin_create",
315  name) >= sizeof(create))
316  {
317  return FAILED;
318  }
319  translate(create, "-", "_");
320  constructor = dlsym(handle, create);
321  if (constructor == NULL)
322  {
323  return NOT_FOUND;
324  }
325  if (integrity && lib->integrity)
326  {
327  if (!lib->integrity->check_segment(lib->integrity, name, constructor))
328  {
329  DBG1(DBG_LIB, "plugin '%s': failed segment integrity test", name);
330  return FAILED;
331  }
332  DBG1(DBG_LIB, "plugin '%s': passed file and segment integrity tests",
333  name);
334  }
335  plugin = constructor();
336  if (plugin == NULL)
337  {
338  DBG1(DBG_LIB, "plugin '%s': failed to load - %s returned NULL", name,
339  create);
340  return FAILED;
341  }
342  INIT(*entry,
343  .plugin = plugin,
344  .critical = critical,
345  .features = linked_list_create(),
346  );
347  DBG2(DBG_LIB, "plugin '%s': loaded successfully", name);
348  return SUCCESS;
349 }

354 static plugin_entry_t *load_plugin(private_plugin_loader_t *this, char *name,
355  char *file, bool critical)
356 {
357  plugin_entry_t *entry;
358  void *handle;
359  int flag = RTLD_LAZY;
360 
361  switch (create_plugin(this, RTLD_DEFAULT, name, FALSE, critical, &entry))
362  {
363  case SUCCESS:
364  this->plugins->insert_last(this->plugins, entry);
365  return entry;
366  case NOT_FOUND:
367  if (file)
368  { /* try to load the plugin from a file */
369  break;
370  }
371  /* fall-through */
372  default:
373  return NULL;
374  }
375  if (lib->integrity)
376  {
377  if (!lib->integrity->check_file(lib->integrity, name, file))
378  {
379  DBG1(DBG_LIB, "plugin '%s': failed file integrity test of '%s'",
380  name, file);
381  return NULL;
382  }
383  }
384  if (lib->settings->get_bool(lib->settings, "%s.dlopen_use_rtld_now",
385  lib->ns, FALSE))
386  {
387  flag = RTLD_NOW;
388  }
389 #ifdef RTLD_NODELETE
390  /* If supported, do not unload the library when unloading a plugin. It
391  * really doesn't matter in productive systems, but causes many (dependency)
392  * library reloads during unit tests. Some libraries can't handle that, e.g.
393  * GnuTLS leaks file descriptors in its library load/unload functions. */
394  flag |= RTLD_NODELETE;
395 #endif
396  handle = dlopen(file, flag);
397  if (handle == NULL)
398  {
399  DBG1(DBG_LIB, "plugin '%s' failed to load: %s", name, dlerror());
400  return NULL;
401  }
402  if (create_plugin(this, handle, name, TRUE, critical, &entry) != SUCCESS)
403  {
404  dlclose(handle);
405  return NULL;
406  }
407  entry->handle = handle;
408  this->plugins->insert_last(this->plugins, entry);
409  return entry;
410 }

492  while (enumerator->enumerate(enumerator, &plugin, NULL))
493  {

496  {

504  {
505  buf[len - 1] = '\0';
506  }
507  return buf;

633 static bool load_dependencies(private_plugin_loader_t *this,
634  provided_feature_t *provided,
635  int level)
636 {
637  registered_feature_t *registered, lookup;
638  int indent = level * 2;
639  int i;
640 
641  /* first entry is provided feature, followed by dependencies */
642  for (i = 1; i < provided->dependencies; i++)
643  {
644  if (provided->feature[i].kind != FEATURE_DEPENDS &&
645  provided->feature[i].kind != FEATURE_SDEPEND)
646  { /* end of dependencies */
647  break;
648  }
649 
650  /* we load the feature even if a compatible one is already loaded,
651  * otherwise e.g. a specific database implementation loaded before
652  * another might cause a plugin feature loaded in-between to fail */
653  lookup.feature = &provided->feature[i];
654  do
655  { /* prefer an exactly matching feature, could be omitted but
656  * results in a more predictable behavior */
657  registered = this->features->get_match(this->features,
658  &lookup,
659  (void*)loadable_feature_equals);
660  if (!registered)
661  { /* try fuzzy matching */
662  registered = this->features->get_match(this->features,
663  &lookup,
664  (void*)loadable_feature_matches);
665  }
666  if (registered)
667  {
668  load_registered(this, registered, level);
669  }
670  /* we could stop after finding one but for dependencies like
671  * DB_ANY it might be needed to load all matching features */
672  }
673  while (registered);
674 
675  if (!find_compatible_feature(this, &provided->feature[i]))
676  {
677  char *name, *provide, *depend;
678  bool soft = provided->feature[i].kind == FEATURE_SDEPEND;
679 
680  name = provided->entry->plugin->get_name(provided->entry->plugin);
681  provide = plugin_feature_get_string(&provided->feature[0]);
682  depend = plugin_feature_get_string(&provided->feature[i]);
683  if (soft)
684  {
685  DBG3(DBG_LIB, "%*sfeature %s in plugin '%s' has unmet soft "
686  "dependency: %s", indent, "", provide, name, depend);
687  }
688  else if (provided->entry->critical)
689  {
690  DBG1(DBG_LIB, "feature %s in critical plugin '%s' has unmet "
691  "dependency: %s", provide, name, depend);
692  }
693  else
694  {
695  DBG2(DBG_LIB, "feature %s in plugin '%s' has unmet dependency: "
696  "%s", provide, name, depend);
697  }
698  free(provide);
699  free(depend);
700  if (soft)
701  { /* it's ok if we can't resolve soft dependencies */
702  continue;
703  }
704  return FALSE;
705  }
706  }
707  return TRUE;
708 }

713 static void load_feature(private_plugin_loader_t *this,
714  provided_feature_t *provided,
715  int level)
716 {
717  if (load_dependencies(this, provided, level))
718  {
719  char *name, *provide;
720 
721  if (plugin_feature_load(provided->entry->plugin, provided->feature,
722  provided->reg))
723  {
724  provided->loaded = TRUE;
725  /* insert first so we can unload the features in reverse order */
726  this->loaded->insert_first(this->loaded, provided);
727  return;
728  }
729 
730  name = provided->entry->plugin->get_name(provided->entry->plugin);
731  provide = plugin_feature_get_string(&provided->feature[0]);
732  if (provided->entry->critical)
733  {
734  DBG1(DBG_LIB, "feature %s in critical plugin '%s' failed to load",
735  provide, name);
736  }
737  else
738  {
739  DBG2(DBG_LIB, "feature %s in plugin '%s' failed to load",
740  provide, name);
741  }
742  free(provide);
743  }
744  else
745  { /* TODO: we could check the current level and set a different flag when
746  * being loaded as dependency. If there are loops there is a chance the
747  * feature can be loaded later when loading the feature directly. */
748  this->stats.depends++;
749  }
750  provided->failed = TRUE;
751  this->stats.critical += provided->entry->critical ? 1 : 0;
752  this->stats.failed++;
753 }

758 static void load_provided(private_plugin_loader_t *this,
759  provided_feature_t *provided,
760  int level)
761 {
762  char *name, *provide;
763  int indent = level * 2;
764 
765  if (provided->loaded || provided->failed)
766  {
767  return;
768  }
769  name = provided->entry->plugin->get_name(provided->entry->plugin);
770  provide = plugin_feature_get_string(provided->feature);
771  if (provided->loading)
772  { /* prevent loop */
773  DBG3(DBG_LIB, "%*sloop detected while loading %s in plugin '%s'",
774  indent, "", provide, name);
775  free(provide);
776  return;
777  }
778  DBG3(DBG_LIB, "%*sloading feature %s in plugin '%s'",
779  indent, "", provide, name);
780  free(provide);
781 
782  provided->loading = TRUE;
783  load_feature(this, provided, level + 1);
784  provided->loading = FALSE;
785 }

790 static void load_features(private_plugin_loader_t *this)
791 {
792  enumerator_t *enumerator, *inner;
793  plugin_entry_t *plugin;
794  provided_feature_t *provided;
795 
796  /* we do this in plugin order to allow implicit dependencies to be resolved
797  * by reordering plugins */
798  enumerator = this->plugins->create_enumerator(this->plugins);
799  while (enumerator->enumerate(enumerator, &plugin))

814 static void register_features(private_plugin_loader_t *this,
815  plugin_entry_t *entry)
816 {
817  plugin_feature_t *feature, *reg;
818  registered_feature_t *registered, lookup;
819  provided_feature_t *provided;
820  int count, i;
821 
822  if (!entry->plugin->get_features)
823  { /* feature interface not supported */
824  DBG1(DBG_LIB, "plugin '%s' does not provide features, deprecated",
825  entry->plugin->get_name(entry->plugin));
826  return;
827  }
828  reg = NULL;
829  count = entry->plugin->get_features(entry->plugin, &feature);
830  for (i = 0; i < count; i++)
831  {
832  switch (feature->kind)
833  {
834  case FEATURE_PROVIDE:
835  lookup.feature = feature;
836  registered = this->features->get(this->features, &lookup);
837  if (!registered)
838  {
839  INIT(registered,
840  .feature = feature,
841  .plugins = linked_list_create(),
842  );
843  this->features->put(this->features, registered, registered);
844  }
845  INIT(provided,
846  .entry = entry,
847  .feature = feature,
848  .reg = reg,
849  .dependencies = count - i,
850  );
851  registered->plugins->insert_last(registered->plugins,
852  provided);
853  entry->features->insert_last(entry->features, provided);
854  break;
855  case FEATURE_REGISTER:
856  case FEATURE_CALLBACK:
857  reg = feature;
858  break;
859  default:
860  break;
861  }
862  feature++;
863  }
864 }

869 static void unregister_feature(private_plugin_loader_t *this,
870  provided_feature_t *provided)
871 {
872  registered_feature_t *registered, lookup;
873 
874  lookup.feature = provided->feature;
875  registered = this->features->get(this->features, &lookup);
876  if (registered)
877  {
878  registered->plugins->remove(registered->plugins, provided, NULL);
879  if (registered->plugins->get_count(registered->plugins) == 0)

882  registered->plugins->destroy(registered->plugins);

887  provided_feature_t *first;
888

890  registered->feature = first->feature;
891  }
892  }

1032 static char *modular_pluginlist(char *list)
1033 {
1034  enumerator_t *enumerator;
1035  array_t *given, *final;
1036  plugin_priority_t item, *current, found;
1037  char *plugin, *plugins = NULL;
1038  int i = 0, max_prio;
1039 
1040  if (!lib->settings->get_bool(lib->settings, "%s.load_modular", FALSE,
1041  lib->ns))
1042  {
1043  return list;
1044  }
1045 
1046  given = array_create(sizeof(plugin_priority_t), 0);
1047  final = array_create(sizeof(plugin_priority_t), 0);
1048 
1049  enumerator = enumerator_create_token(list, " ", " ");
1050  while (enumerator->enumerate(enumerator, &plugin))
1051  {
1052  item.name = strdup(plugin);
1053  item.prio = i++;
1054  array_insert(given, ARRAY_TAIL, &item);
1055  }
1056  enumerator->destroy(enumerator);
1057  array_sort(given, (void*)plugin_priority_cmp_name, NULL);
1058  /* the maximum priority used for plugins not found in this list */
1059  max_prio = i + 1;
1060 
1061  enumerator = lib->settings->create_section_enumerator(lib->settings,
1062  "%s.plugins", lib->ns);
1063  while (enumerator->enumerate(enumerator, &plugin))
1064  {
1065  item.prio = lib->settings->get_int(lib->settings,
1066  "%s.plugins.%s.load", 0, lib->ns, plugin);
1067  if (!item.prio)
1068  {
1069  if (!lib->settings->get_bool(lib->settings,
1070  "%s.plugins.%s.load", FALSE, lib->ns, plugin))
1071  {
1072  continue;
1073  }
1074  item.prio = 1;
1075  }
1076  item.name = plugin;
1077  item.def = max_prio;
1078  if (array_bsearch(given, &item, (void*)plugin_priority_cmp_name,
1079  &found) != -1)
1080  {
1081  item.def = max_prio - found.prio;
1082  }
1083  array_insert(final, ARRAY_TAIL, &item);
1084  }
1085  enumerator->destroy(enumerator);
1086  array_destroy_function(given, (void*)plugin_priority_free, NULL);
1087 
1088  array_sort(final, (void*)plugin_priority_cmp, NULL);
1089 
1090  plugins = strdup("");
1091  enumerator = array_create_enumerator(final);
1092  while (enumerator->enumerate(enumerator, &current))
1093  {
1094  char *prev = plugins;
1095  if (asprintf(&plugins, "%s %s", plugins ?: "", current->name) < 0)
1096  {
1097  plugins = prev;
1098  break;
1099  }
1100  free(prev);
1101  }
1102  enumerator->destroy(enumerator);
1103  array_destroy(final);
1104  return plugins;
1105 }

1108  private_plugin_loader_t *this, char *list)
1109 {

1112  bool critical_failed = FALSE;

1116 #endif /* PLUGINDIR */

1122  {
1123  plugin_entry_t *entry;
1124  bool critical = FALSE;
1125  char buf[PATH_MAX], *file = NULL;
1126  int len;
1127 
1128  token = strdup(token);

1133  token[len-1] = '\0';

1156  critical_failed = TRUE;

1158  }
1159  free(token);

1172  purge_plugins(this);
1173  }

1176  free(this->loaded_plugins);
1177  this->loaded_plugins = loaded_plugins_list(this);
1178  }

1182  }
