84

89  }
90  if (bits_needed > 32)
91  {
92  /* too many bits requested */
93  return FALSE;
94  }
95 
96  while (bits_needed)
97  {
98  if (this->bits_left == 0)
99  {
100  if (this->octets_left == 0)
101  {
102  /* get another block from MGF1 */
103  if (!this->mgf1->get_mask(this->mgf1, this->hash_len,
104  this->octets))
105  {
106  /* no block available */
107  return FALSE;
108  }
109  this->octets_left = this->hash_len;
110  this->octets_count += this->hash_len;
111  }
112  this->bits = untoh32(this->octets + this->hash_len -
113  this->octets_left);
114  this->bits_left = 32;
115  this->octets_left -= 4;
116  }
117  if (bits_needed > this->bits_left)
118  {
119  bits_now = this->bits_left;
120  this->bits_left = 0;
121  bits_needed -= bits_now;
122  }
123  else
124  {
125  bits_now = bits_needed;
126  this->bits_left -= bits_needed;
127  bits_needed = 0;
128  }
129  if (bits_now == 32)
130  {
131  *bits = this->bits;
132  }
133  else
134  {

136  *bits |= this->bits >> this->bits_left;
137  if (this->bits_left)

140  }
