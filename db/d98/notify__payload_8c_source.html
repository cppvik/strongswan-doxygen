415 METHOD(payload_t, verify, status_t,
416  private_notify_payload_t *this)
417 {
418  bool bad_length = FALSE;
419 
420  switch (this->protocol_id)
421  {
422  case PROTO_NONE:
423  case PROTO_IKE:
424  case PROTO_AH:
425  case PROTO_ESP:
426  break;
427  default:
428  DBG1(DBG_ENC, "Unknown protocol (%d)", this->protocol_id);
429  return FAILED;
430  }
431 
432  switch (this->notify_type)
433  {
434  case INVALID_KE_PAYLOAD:
435  {
436  if (this->type == PLV2_NOTIFY && this->notify_data.len != 2)
437  {
438  bad_length = TRUE;
439  }
440  break;
441  }
442  case NAT_DETECTION_SOURCE_IP:
443  case NAT_DETECTION_DESTINATION_IP:
444  case ME_CONNECTAUTH:
445  {
446  if (this->notify_data.len != HASH_SIZE_SHA1)
447  {
448  bad_length = TRUE;
449  }
450  break;
451  }
452  case INVALID_SYNTAX:
453  case INVALID_MAJOR_VERSION:
454  case NO_PROPOSAL_CHOSEN:
455  {
456  if (this->type == PLV2_NOTIFY && this->notify_data.len != 0)
457  {
458  bad_length = TRUE;
459  }
460  break;
461  }
462  case ADDITIONAL_IP4_ADDRESS:
463  {
464  if (this->notify_data.len != 4)
465  {
466  bad_length = TRUE;
467  }
468  break;
469  }
470  case ADDITIONAL_IP6_ADDRESS:
471  {
472  if (this->notify_data.len != 16)
473  {
474  bad_length = TRUE;
475  }
476  break;
477  }
478  case SIGNATURE_HASH_ALGORITHMS:
479  {
480  if (this->notify_data.len % 2)
481  {
482  bad_length = TRUE;
483  }
484  break;
485  }
486  case AUTH_LIFETIME:
487  {
488  if (this->notify_data.len != 4)
489  {
490  bad_length = TRUE;
491  }
492  break;
493  }
494  case IPCOMP_SUPPORTED:
495  {
496  if (this->notify_data.len != 3)
497  {
498  bad_length = TRUE;
499  }
500  break;
501  }
502  case ME_ENDPOINT:
503  if (this->notify_data.len != 8 &&
504  this->notify_data.len != 12 &&
505  this->notify_data.len != 24)
506  {
507  bad_length = TRUE;
508  }
509  break;
510  case ME_CONNECTID:
511  if (this->notify_data.len < 4 ||
512  this->notify_data.len > 16)
513  {
514  bad_length = TRUE;
515  }
516  break;
517  case ME_CONNECTKEY:
518  if (this->notify_data.len < 16 ||
519  this->notify_data.len > 32)
520  {
521  bad_length = TRUE;
522  }
523  break;
524  case DPD_R_U_THERE:
525  case DPD_R_U_THERE_ACK:
526  if (this->notify_data.len != 4)
527  {
528  bad_length = TRUE;
529  }
530  break;
531  default:
532  /* TODO: verify */
533  break;
534  }
535  if (bad_length)
536  {
537  DBG1(DBG_ENC, "invalid notify data length for %N (%d)",
538  notify_type_names, this->notify_type,
539  this->notify_data.len);
540  return FAILED;
541  }
542  return SUCCESS;
543 }

640 METHOD(notify_payload_t, set_spi, void,
641  private_notify_payload_t *this, u_int32_t spi)
642 {
643  chunk_free(&this->spi);
644  switch (this->protocol_id)
645  {
646  case PROTO_AH:
