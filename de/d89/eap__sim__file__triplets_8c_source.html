85 static void enumerator_destroy(triplet_enumerator_t *e)
86 {
87  if (e->current)
88  {
89  /* We assume that the current element is used on invocation if destroy.
90  * We move that triplet to the end to avoid handout of the same triplet
91  * next time. */
92  e->this->triplets->remove_at(e->this->triplets, e->inner);
93  e->this->triplets->insert_last(e->this->triplets, e->current);
94  }
95  e->inner->destroy(e->inner);
96  e->this->mutex->unlock(e->this->mutex);
97  free(e);
98 }

153 static bool read_triplets(private_eap_sim_file_triplets_t *this, char *path)
154 {
155  char line[512];
156  FILE *file;
157  int i, nr = 0;
158 
159  file = fopen(path, "r");
160  if (file == NULL)
161  {
162  DBG1(DBG_CFG, "opening triplet file %s failed: %s",
163  path, strerror(errno));
164  return FALSE;
165  }
166 
167  /* read line by line */
168  while (fgets(line, sizeof(line), file))
169  {
170  triplet_t *triplet;
171  enumerator_t *enumerator;
172  char *token;
173 
174  nr++;
175  /* skip comments, empty lines */
176  switch (line[0])
177  {
178  case '\n':
179  case '\r':
180  case '#':
181  case '\0':
182  continue;
183  default:
184  break;
185  }
186  triplet = malloc_thing(triplet_t);
187  memset(triplet, 0, sizeof(triplet_t));
188 
189  i = 0;
190  enumerator = enumerator_create_token(line, ",", " \n\r#");
191  while (enumerator->enumerate(enumerator, &token))
192  {
193  switch (i++)
194  {
195  case 0: /* IMSI */
196  triplet->imsi = identification_create_from_string(token);
197  continue;
198  case 1: /* rand */
199  parse_token(triplet->rand, token, SIM_RAND_LEN);
200  continue;
201  case 2: /* sres */
202  parse_token(triplet->sres, token, SIM_SRES_LEN);
203  continue;
204  case 3: /* kc */
205  parse_token(triplet->kc, token, SIM_KC_LEN);
206  continue;
207  default:
208  break;;
209  }
210  break;
211  }
212  enumerator->destroy(enumerator);
213  if (i < 4)
214  {
215  DBG1(DBG_CFG, "error in triplet file, line %d", nr);
216  triplet_destroy(triplet);
217  continue;
218  }
219 
220  DBG2(DBG_CFG, "triplet: imsi %Y\nrand %b\nsres %b\nkc %b",
221  triplet->imsi, triplet->rand, SIM_RAND_LEN,
222  triplet->sres, SIM_SRES_LEN, triplet->kc, SIM_KC_LEN);
223 
224  this->triplets->insert_last(this->triplets, triplet);
225  }
226  fclose(file);
227 
228  DBG1(DBG_CFG, "read %d triplets from %s",
229  this->triplets->get_count(this->triplets), path);
230  return TRUE;
231 }
