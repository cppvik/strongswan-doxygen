160 static bool wait_remove_entry(private_ipsec_sa_mgr_t *this,
161  ipsec_sa_entry_t *entry)
162 {
163  if (entry->awaits_deletion)
164  {
165  /* this will be deleted by another thread already */
166  return FALSE;
167  }
168  entry->awaits_deletion = TRUE;
169  while (entry->locked)
170  {
171  entry->condvar->wait(entry->condvar, this->mutex);
172  }
173  while (entry->waiting_threads > 0)
174  {
175  entry->condvar->broadcast(entry->condvar);
176  entry->condvar->wait(entry->condvar, this->mutex);
177  }
178  return TRUE;
179 }

185 static bool wait_for_entry(private_ipsec_sa_mgr_t *this,
186  ipsec_sa_entry_t *entry)
187 {
188  while (entry->locked && !entry->awaits_deletion)
189  {
190  entry->waiting_threads++;
191  entry->condvar->wait(entry->condvar, this->mutex);
192  entry->waiting_threads--;
193  }
194  if (entry->awaits_deletion)
195  {
196  /* others may still be waiting, */
197  entry->condvar->signal(entry->condvar);
198  return FALSE;
199  }
200  entry->locked = TRUE;
201  return TRUE;
202 }

294 static job_requeue_t sa_expired(ipsec_sa_expired_t *expired)
295 {
296  private_ipsec_sa_mgr_t *this = expired->manager;
297 
298  this->mutex->lock(this->mutex);
299  if (this->sas->find_first(this->sas, (void*)match_entry_by_ptr,
300  NULL, expired->entry) == SUCCESS)
301  {
302  u_int32_t hard_offset;
303 
304  hard_offset = expired->hard_offset;
305  expired->entry->sa->expire(expired->entry->sa, hard_offset == 0);
306  if (hard_offset)
307  { /* soft limit reached, schedule hard expire */
308  expired->hard_offset = 0;
309  this->mutex->unlock(this->mutex);
310  return JOB_RESCHEDULE(hard_offset);
311  }
312  /* hard limit reached */
313  if (remove_entry(this, expired->entry))
314  {
315  destroy_entry(expired->entry);
316  }
317  }
318  this->mutex->unlock(this->mutex);
319  return JOB_REQUEUE_NONE;
320 }

325 static void schedule_expiration(private_ipsec_sa_mgr_t *this,
326  ipsec_sa_entry_t *entry)
327 {
328  lifetime_cfg_t *lifetime = entry->sa->get_lifetime(entry->sa);
329  ipsec_sa_expired_t *expired;
330  callback_job_t *job;
331  u_int32_t timeout;
332 
333  if (!lifetime->time.life)
334  { /* no expiration at all */
335  return;
336  }
337 
338  INIT(expired,
339  .manager = this,
340  .entry = entry,
341  );
342 
343  /* schedule a rekey first, a hard timeout will be scheduled then, if any */
344  expired->hard_offset = lifetime->time.life - lifetime->time.rekey;
345  timeout = lifetime->time.rekey;
346 
347  if (lifetime->time.life <= lifetime->time.rekey ||
348  lifetime->time.rekey == 0)
349  { /* no rekey, schedule hard timeout */
350  expired->hard_offset = 0;
351  timeout = lifetime->time.life;
352  }
353 
354  job = callback_job_create((callback_job_cb_t)sa_expired, expired,
355  (callback_job_cleanup_t)free, NULL);
356  lib->scheduler->schedule_job(lib->scheduler, (job_t*)job, timeout);
357 }

397 METHOD(ipsec_sa_mgr_t, get_spi, status_t,
398  private_ipsec_sa_mgr_t *this, host_t *src, host_t *dst, u_int8_t protocol,
399  u_int32_t *spi)
400 {
401  u_int32_t spi_new;
402 
403  this->mutex->lock(this->mutex);
404  if (!this->rng)
405  {
406  this->rng = lib->crypto->create_rng(lib->crypto, RNG_WEAK);
407  if (!this->rng)
408  {
409  this->mutex->unlock(this->mutex);
410  DBG1(DBG_ESP, "failed to create RNG for SPI generation");
411  return FAILED;
412  }
413  }
414 
415  do
416  {
417  if (!this->rng->get_bytes(this->rng, sizeof(spi_new),
418  (u_int8_t*)&spi_new))
419  {
420  this->mutex->unlock(this->mutex);
421  DBG1(DBG_ESP, "failed to allocate SPI");
422  return FAILED;
423  }
424  /* make sure the SPI is valid (not in range 0-255) */
425  spi_new |= 0x00000100;
426  spi_new = htonl(spi_new);
427  }
428  while (!allocate_spi(this, spi_new));
429  this->mutex->unlock(this->mutex);
430 
431  *spi = spi_new;
432 
433  DBG2(DBG_ESP, "allocated SPI %.8x", ntohl(*spi));
434  return SUCCESS;
435 }

437 METHOD(ipsec_sa_mgr_t, add_sa, status_t,
438  private_ipsec_sa_mgr_t *this, host_t *src, host_t *dst, u_int32_t spi,
439  u_int8_t protocol, u_int32_t reqid, mark_t mark, u_int32_t tfc,
440  lifetime_cfg_t *lifetime, u_int16_t enc_alg, chunk_t enc_key,
441  u_int16_t int_alg, chunk_t int_key, ipsec_mode_t mode, u_int16_t ipcomp,
442  u_int16_t cpi, bool initiator, bool encap, bool esn, bool inbound,
443  bool update)
444 {
445  ipsec_sa_entry_t *entry;
446  ipsec_sa_t *sa_new;
447 
448  DBG2(DBG_ESP, "adding SAD entry with SPI %.8x and reqid {%u}",
449  ntohl(spi), reqid);
450  DBG2(DBG_ESP, " using encryption algorithm %N with key size %d",
451  encryption_algorithm_names, enc_alg, enc_key.len * 8);
452  DBG2(DBG_ESP, " using integrity algorithm %N with key size %d",
453  integrity_algorithm_names, int_alg, int_key.len * 8);
454 
455  sa_new = ipsec_sa_create(spi, src, dst, protocol, reqid, mark, tfc,
456  lifetime, enc_alg, enc_key, int_alg, int_key, mode,
457  ipcomp, cpi, encap, esn, inbound);
458  if (!sa_new)
459  {
460  DBG1(DBG_ESP, "failed to create SAD entry");
461  return FAILED;
462  }
463 
464  this->mutex->lock(this->mutex);
465 
466  if (update)
467  { /* remove any pre-allocated SPIs */
468  u_int32_t *spi_alloc;
469 
470  spi_alloc = this->allocated_spis->remove(this->allocated_spis, &spi);
471  free(spi_alloc);
472  }
473 
474  if (this->sas->find_first(this->sas, (void*)match_entry_by_spi_src_dst,
475  NULL, &spi, src, dst) == SUCCESS)
476  {
477  this->mutex->unlock(this->mutex);
478  DBG1(DBG_ESP, "failed to install SAD entry: already installed");
479  sa_new->destroy(sa_new);
480  return FAILED;
481  }
482 
483  entry = create_entry(sa_new);
484  schedule_expiration(this, entry);
485  this->sas->insert_first(this->sas, entry);
486 
487  this->mutex->unlock(this->mutex);
488  return SUCCESS;
489 }
