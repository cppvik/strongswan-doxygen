125 METHOD(pb_tnc_msg_t, process, status_t,
126  private_pb_remediation_parameters_msg_t *this, u_int32_t *offset)
127 {
128  bio_reader_t *reader;
129  u_int8_t reserved;
130  status_t status = SUCCESS;
131  u_char *pos;
132 
133  *offset = 0;
134 
135  /* process message */
136  reader = bio_reader_create(this->encoding);
137  reader->read_uint8 (reader, &reserved);
138  reader->read_uint24(reader, &this->parameters_type.vendor_id);
139  reader->read_uint32(reader, &this->parameters_type.type);
140  reader->read_data (reader, reader->remaining(reader), &this->parameters);
141 
142  this->parameters = chunk_clone(this->parameters);
143  reader->destroy(reader);
144 
145  if (this->parameters_type.vendor_id == PEN_IETF &&
146  this->parameters_type.type == PB_REMEDIATION_STRING)
147  {
148  reader = bio_reader_create(this->parameters);
149  status = FAILED;
150  *offset = 8;
151 
152  if (!reader->read_data32(reader, &this->string))
153  {
154  DBG1(DBG_TNC, "insufficient data for remediation string");
155  goto end;
156  };
157  *offset += 4;
158 
159  pos = memchr(this->string.ptr, '\0', this->string.len);
160  if (pos)
161  {
162  DBG1(DBG_TNC, "nul termination in remediation string");
163  *offset += (pos - this->string.ptr);
164  goto end;
165  }
166  *offset += this->string.len;
167 
168  if (!reader->read_data8(reader, &this->lang_code))
169  {
170  DBG1(DBG_TNC, "insufficient data for remediation string lang code");
171  goto end;
172  };
173  *offset += 1;
174 
175  pos = memchr(this->lang_code.ptr, '\0', this->lang_code.len);
176 
177  if (pos)
178  {
179  DBG1(DBG_TNC, "nul termination in remediation string lang code");
180  *offset += (pos - this->lang_code.ptr);
181  goto end;
182  }
183  status = SUCCESS;
184 
185 end:
186  reader->destroy(reader);
187  }

189 }
