50 static traffic_selector_t *build_traffic_selector(private_sql_config_t *this,
51  enumerator_t *e, bool *local)
52 {
53  int type, protocol, start_port, end_port;
54  chunk_t start_addr, end_addr;
55  traffic_selector_t *ts;
56  enum {
57  TS_LOCAL = 0,
58  TS_REMOTE = 1,
59  TS_LOCAL_DYNAMIC = 2,
60  TS_REMOTE_DYNAMIC = 3,
61  } kind;
62 
63  while (e->enumerate(e, &kind, &type, &protocol,
64  &start_addr, &end_addr, &start_port, &end_port))
65  {
66  *local = FALSE;
67  switch (kind)
68  {
69  case TS_LOCAL:
70  *local = TRUE;
71  /* FALL */
72  case TS_REMOTE:
73  ts = traffic_selector_create_from_bytes(protocol, type,
74  start_addr, start_port, end_addr, end_port);
75  break;
76  case TS_LOCAL_DYNAMIC:
77  *local = TRUE;
78  /* FALL */
79  case TS_REMOTE_DYNAMIC:
80  ts = traffic_selector_create_dynamic(protocol,
81  start_port, end_port);
82  break;
83  default:
84  continue;
85  }
86  if (ts)
87  {
88  return ts;
89  }
90  }
91  return NULL;
92 }

337 static peer_cfg_t *build_peer_cfg(private_sql_config_t *this, enumerator_t *e,
338  identification_t *me, identification_t *other)
339 {
340  int id, ike_cfg, l_type, r_type,
341  cert_policy, uniqueid, auth_method, eap_type, eap_vendor, keyingtries,
342  rekeytime, reauthtime, jitter, overtime, mobike, dpd_delay,
343  mediation, mediated_by, p_type;
344  chunk_t l_data, r_data, p_data;
345  char *name, *virtual, *pool;
346  enumerator_t *enumerator;
347 
348  while (e->enumerate(e,
349  &id, &name, &ike_cfg, &l_type, &l_data, &r_type, &r_data,
350  &cert_policy, &uniqueid, &auth_method, &eap_type, &eap_vendor,
351  &keyingtries, &rekeytime, &reauthtime, &jitter, &overtime, &mobike,
352  &dpd_delay, &virtual, &pool,
353  &mediation, &mediated_by, &p_type, &p_data))
354  {
355  identification_t *local_id, *remote_id, *peer_id = NULL;
356  peer_cfg_t *peer_cfg, *mediated_cfg;
357  ike_cfg_t *ike;
358  host_t *vip = NULL;
359  auth_cfg_t *auth;
360 
361  local_id = identification_create_from_encoding(l_type, l_data);
362  remote_id = identification_create_from_encoding(r_type, r_data);
363  if (!id_matches(me, local_id) || !id_matches(other, remote_id))
364  {
365  local_id->destroy(local_id);
366  remote_id->destroy(remote_id);
367  continue;
368  }
369  ike = get_ike_cfg_by_id(this, ike_cfg);
370  mediated_cfg = mediated_by ? get_peer_cfg_by_id(this, mediated_by) : NULL;
371  if (p_type)
372  {
373  peer_id = identification_create_from_encoding(p_type, p_data);
374  }
375  if (virtual)
376  {
377  vip = host_create_from_string(virtual, 0);
378  }
379  if (ike)
380  {
381  peer_cfg = peer_cfg_create(
382  name, ike, cert_policy, uniqueid,
383  keyingtries, rekeytime, reauthtime, jitter, overtime,
384  mobike, FALSE, TRUE, dpd_delay, 0,
385  mediation, mediated_cfg, peer_id);
386  if (vip)
387  {
388  peer_cfg->add_virtual_ip(peer_cfg, vip);
389  }
390  if (pool)
391  {
392  /* attr-sql used comma separated pools, but we now completely
393  * support multiple pools directly. Support old SQL configs: */
394  enumerator = enumerator_create_token(pool, ",", " ");
395  while (enumerator->enumerate(enumerator, &pool))
396  {
397  peer_cfg->add_pool(peer_cfg, pool);
398  }
399  enumerator->destroy(enumerator);
400  }
401  auth = auth_cfg_create();
402  auth->add(auth, AUTH_RULE_AUTH_CLASS, auth_method);
403  auth->add(auth, AUTH_RULE_IDENTITY, local_id);
404  peer_cfg->add_auth_cfg(peer_cfg, auth, TRUE);
405  auth = auth_cfg_create();
406  auth->add(auth, AUTH_RULE_IDENTITY, remote_id);
407  if (eap_type)
408  {
409  auth->add(auth, AUTH_RULE_AUTH_CLASS, AUTH_CLASS_EAP);
410  auth->add(auth, AUTH_RULE_EAP_TYPE, eap_type);
411  if (eap_vendor)
412  {
413  auth->add(auth, AUTH_RULE_EAP_VENDOR, eap_vendor);
414  }
415  }
416  peer_cfg->add_auth_cfg(peer_cfg, auth, FALSE);
417  add_child_cfgs(this, peer_cfg, id);
418  return peer_cfg;
419  }
420  DESTROY_IF(ike);
421  DESTROY_IF(mediated_cfg);
422  DESTROY_IF(peer_id);
423  DESTROY_IF(local_id);
424  DESTROY_IF(remote_id);
425  }
426  return NULL;
427 }
