86 static sqlite3_stmt* run(private_sqlite_database_t *this, char *sql,
87  va_list *args)
88 {
89  sqlite3_stmt *stmt = NULL;
90  int params, i, res = SQLITE_OK;
91 
92 #ifdef HAVE_SQLITE3_PREPARE_V2
93  if (sqlite3_prepare_v2(this->db, sql, -1, &stmt, NULL) == SQLITE_OK)
94 #else
95  if (sqlite3_prepare(this->db, sql, -1, &stmt, NULL) == SQLITE_OK)
96 #endif
97  {
98  params = sqlite3_bind_parameter_count(stmt);
99  for (i = 1; i <= params; i++)
100  {
101  switch (va_arg(*args, db_type_t))
102  {
103  case DB_INT:
104  {
105  res = sqlite3_bind_int(stmt, i, va_arg(*args, int));
106  break;
107  }
108  case DB_UINT:
109  {
110  res = sqlite3_bind_int64(stmt, i, va_arg(*args, u_int));
111  break;
112  }
113  case DB_TEXT:
114  {
115  const char *text = va_arg(*args, const char*);
116  res = sqlite3_bind_text(stmt, i, text, -1,
117  SQLITE_TRANSIENT);
118  break;
119  }
120  case DB_BLOB:
121  {
122  chunk_t c = va_arg(*args, chunk_t);
123  res = sqlite3_bind_blob(stmt, i, c.ptr, c.len,
124  SQLITE_TRANSIENT);
125  break;
126  }
127  case DB_DOUBLE:
128  {
129  res = sqlite3_bind_double(stmt, i, va_arg(*args, double));
130  break;
131  }
132  case DB_NULL:
133  {
134  res = sqlite3_bind_null(stmt, i);
135  break;
136  }
137  default:
138  {
139  res = SQLITE_MISUSE;
140  break;
141  }
142  }
143  if (res != SQLITE_OK)
144  {
145  break;
146  }
147  }
148  }
149  else
150  {
151  DBG1(DBG_LIB, "preparing sqlite statement failed: %s",
152  sqlite3_errmsg(this->db));
153  }
154  if (res != SQLITE_OK)
155  {
156  DBG1(DBG_LIB, "binding sqlite statement failed: %s",
157  sqlite3_errmsg(this->db));
158  sqlite3_finalize(stmt);
159  return NULL;
160  }
161  return stmt;
162 }

194 static bool sqlite_enumerator_enumerate(sqlite_enumerator_t *this, ...)
195 {
196  int i;
197  va_list args;
198 
199  switch (sqlite3_step(this->stmt))
200  {
201  case SQLITE_ROW:
202  break;
203  default:
204  DBG1(DBG_LIB, "stepping sqlite statement failed: %s",
205  sqlite3_errmsg(this->database->db));
206  /* fall */
207  case SQLITE_DONE:
208  return FALSE;
209  }
210  va_start(args, this);
211  for (i = 0; i < this->count; i++)
212  {
213  switch (this->columns[i])
214  {
215  case DB_INT:
216  {
217  int *value = va_arg(args, int*);
218  *value = sqlite3_column_int(this->stmt, i);
219  break;
220  }
221  case DB_UINT:
222  {
223  u_int *value = va_arg(args, u_int*);
224  *value = (u_int)sqlite3_column_int64(this->stmt, i);
225  break;
226  }
227  case DB_TEXT:
228  {
229  const unsigned char **value = va_arg(args, const unsigned char**);
230  *value = sqlite3_column_text(this->stmt, i);
231  break;
232  }
233  case DB_BLOB:
234  {
235  chunk_t *chunk = va_arg(args, chunk_t*);
236  chunk->len = sqlite3_column_bytes(this->stmt, i);
237  chunk->ptr = (u_char*)sqlite3_column_blob(this->stmt, i);
238  break;
239  }
240  case DB_DOUBLE:
241  {
242  double *value = va_arg(args, double*);
243  *value = sqlite3_column_double(this->stmt, i);
244  break;
245  }
246  default:
247  DBG1(DBG_LIB, "invalid result type supplied");
248  va_end(args);
249  return FALSE;
250  }
251  }
252  va_end(args);
253  return TRUE;
254 }

291 {

294  va_list args;
295

297  this->mutex->lock(this->mutex);

350 static bool finalize_transaction(private_sqlite_database_t *this,
351  bool rollback)
352 {
353  transaction_t *trans;
354  char *command = "COMMIT TRANSACTION";
355  bool success;
356 
357  trans = this->transaction->get(this->transaction);
358  if (!trans)
359  {
360  DBG1(DBG_LIB, "no database transaction found");
361  return FALSE;
362  }
363 
364  if (ref_put(&trans->refs))
365  {
366  if (trans->rollback)
367  {
368  command = "ROLLBACK TRANSACTION";
369  }
370  success = execute(this, NULL, command) != -1;
371 
372  this->transaction->set(this->transaction, NULL);
373  free(trans);
374  return success;
375  }
376  else
377  { /* set flag, can't be unset */
378  trans->rollback |= rollback;
379  }
380  return TRUE;
381 }
