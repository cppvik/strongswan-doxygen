24 static void verify_netmask(chunk_t addr, int mask)
25 {
26  int byte, bit;
27 
28  for (byte = 0; byte < addr.len; byte++)
29  {
30  for (bit = 7; bit >= 0; bit--)
31  {
32  int val = (addr.ptr[byte] >> bit) & 0x01;
33  if (mask-- > 0)
34  {
35  ck_assert_int_eq(val, 1);
36  }
37  else
38  {
39  ck_assert_int_eq(val, 0);
40  }
41  }
42  }
43 }
