267 METHOD(chapoly_drv_t, poly, bool,
268  private_chapoly_drv_portable_t *this, u_char *data, u_int blocks)
269 {
270  u_int32_t r0, r1, r2, r3, r4;
271  u_int32_t s1, s2, s3, s4;

275 
276  r0 = this->r[0];
277  r1 = this->r[1];
278  r2 = this->r[2];
279  r3 = this->r[3];
280  r4 = this->r[4];
281 
282  s1 = r1 * 5;
283  s2 = r2 * 5;
284  s3 = r3 * 5;
285  s4 = r4 * 5;
286 
287  h0 = this->h[0];
288  h1 = this->h[1];
289  h2 = this->h[2];
290  h3 = this->h[3];
291  h4 = this->h[4];
292 
293  for (i = 0; i < blocks; i++)
294  {
295  /* h += m[i] */
296  h0 += (uletoh32(data + 0) >> 0) & 0x3ffffff;
297  h1 += (uletoh32(data + 3) >> 2) & 0x3ffffff;
298  h2 += (uletoh32(data + 6) >> 4) & 0x3ffffff;
299  h3 += (uletoh32(data + 9) >> 6) & 0x3ffffff;
300  h4 += (uletoh32(data + 12) >> 8) | (1 << 24);
301 
302  /* h *= r */
303  d0 = mlt(h0, r0) + mlt(h1, s4) + mlt(h2, s3) + mlt(h3, s2) + mlt(h4, s1);
304  d1 = mlt(h0, r1) + mlt(h1, r0) + mlt(h2, s4) + mlt(h3, s3) + mlt(h4, s2);
305  d2 = mlt(h0, r2) + mlt(h1, r1) + mlt(h2, r0) + mlt(h3, s4) + mlt(h4, s3);
306  d3 = mlt(h0, r3) + mlt(h1, r2) + mlt(h2, r1) + mlt(h3, r0) + mlt(h4, s4);
307  d4 = mlt(h0, r4) + mlt(h1, r3) + mlt(h2, r2) + mlt(h3, r1) + mlt(h4, r0);
308 
309  /* (partial) h %= p */
310  d1 += sr(d0, 26); h0 = and(d0, 0x3ffffff);
311  d2 += sr(d1, 26); h1 = and(d1, 0x3ffffff);
312  d3 += sr(d2, 26); h2 = and(d2, 0x3ffffff);
313  d4 += sr(d3, 26); h3 = and(d3, 0x3ffffff);
314  h0 += sr(d4, 26) * 5; h4 = and(d4, 0x3ffffff);
315  h1 += h0 >> 26; h0 = h0 & 0x3ffffff;
316 
317  data += POLY_BLOCK_SIZE;
318  }
319 
320  this->h[0] = h0;
321  this->h[1] = h1;
322  this->h[2] = h2;
323  this->h[3] = h3;
324  this->h[4] = h4;
325 
326  return TRUE;
327 }

366 METHOD(chapoly_drv_t, finish, bool,
367  private_chapoly_drv_portable_t *this, u_char *mac)
368 {
369  u_int32_t h0, h1, h2, h3, h4;
370  u_int32_t g0, g1, g2, g3, g4;
371  u_int32_t mask;
372  u_int64_t f = 0;
373 
374  /* fully carry h */
375  h0 = this->h[0];
376  h1 = this->h[1];
377  h2 = this->h[2];
378  h3 = this->h[3];
379  h4 = this->h[4];
380 
381  h2 += (h1 >> 26); h1 = h1 & 0x3ffffff;
382  h3 += (h2 >> 26); h2 = h2 & 0x3ffffff;
383  h4 += (h3 >> 26); h3 = h3 & 0x3ffffff;
384  h0 += (h4 >> 26) * 5; h4 = h4 & 0x3ffffff;
385  h1 += (h0 >> 26); h0 = h0 & 0x3ffffff;
386 
387  /* compute h + -p */
388  g0 = h0 + 5;
389  g1 = h1 + (g0 >> 26); g0 &= 0x3ffffff;
390  g2 = h2 + (g1 >> 26); g1 &= 0x3ffffff;
391  g3 = h3 + (g2 >> 26); g2 &= 0x3ffffff;
392  g4 = h4 + (g3 >> 26) - (1 << 26); g3 &= 0x3ffffff;
393 
394  /* select h if h < p, or h + -p if h >= p */
395  mask = (g4 >> ((sizeof(u_int32_t) * 8) - 1)) - 1;
396  g0 &= mask;
397  g1 &= mask;
398  g2 &= mask;
399  g3 &= mask;
400  g4 &= mask;
401  mask = ~mask;
402  h0 = (h0 & mask) | g0;
403  h1 = (h1 & mask) | g1;
404  h2 = (h2 & mask) | g2;
405  h3 = (h3 & mask) | g3;
406  h4 = (h4 & mask) | g4;
407 
408  /* h = h % (2^128) */
409  h0 = (h0 >> 0) | (h1 << 26);
410  h1 = (h1 >> 6) | (h2 << 20);
411  h2 = (h2 >> 12) | (h3 << 14);
412  h3 = (h3 >> 18) | (h4 << 8);
413 
414  /* mac = (h + s) % (2^128) */
415  f = (f >> 32) + h0 + this->s[0]; htoule32(mac + 0, f);
416  f = (f >> 32) + h1 + this->s[1]; htoule32(mac + 4, f);
417  f = (f >> 32) + h2 + this->s[2]; htoule32(mac + 8, f);
418  f = (f >> 32) + h3 + this->s[3]; htoule32(mac + 12, f);
419 
420  return TRUE;
421 }
