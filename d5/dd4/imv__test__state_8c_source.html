242  enumerator_t *enumerator;

246  enumerator = this->imcs->create_enumerator(this->imcs);
247  while (enumerator->enumerate(enumerator, &imc_entry))
248  {

250  {
251  found = TRUE;
252  break;
253  }
254  }
255  enumerator->destroy(enumerator);
256 
257  if (!found)
258  {
259  imc_entry = malloc_thing(imc_entry_t);
260  imc_entry->imc_id = imc_id;
