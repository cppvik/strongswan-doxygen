75  this->db->commit(this->db);
76  return row;
77  }
78  this->db->rollback(this->db);

131 static host_t* check_lease(private_attr_sql_provider_t *this, char *name,
132  u_int pool, u_int identity)
133 {
134  while (TRUE)
135  {
136  u_int id;
137  chunk_t address;
138  enumerator_t *e;
139  time_t now = time(NULL);
140 
141  e = this->db->query(this->db,
142  "SELECT id, address FROM addresses "
143  "WHERE pool = ? AND identity = ? AND released != 0 LIMIT 1",
144  DB_UINT, pool, DB_UINT, identity, DB_UINT, DB_BLOB);
145  if (!e || !e->enumerate(e, &id, &address))
146  {
147  DESTROY_IF(e);
148  break;
149  }
150  address = chunk_clonea(address);
151  e->destroy(e);
152 
153  if (this->db->execute(this->db, NULL,
154  "UPDATE addresses SET acquired = ?, released = 0 "
155  "WHERE id = ? AND identity = ? AND released != 0",
156  DB_UINT, now, DB_UINT, id, DB_UINT, identity) > 0)
157  {
158  host_t *host;
159 
160  host = host_create_from_chunk(AF_UNSPEC, address, 0);
161  if (host)
162  {
163  DBG1(DBG_CFG, "acquired existing lease for address %H in"
164  " pool '%s'", host, name);
165  return host;
166  }
167  }
168  }
169  return NULL;

177 static host_t* get_lease(private_attr_sql_provider_t *this, char *name,
178  u_int pool, u_int timeout, u_int identity)
179 {
180  while (TRUE)
181  {
182  u_int id;
183  chunk_t address;
184  enumerator_t *e;
185  time_t now = time(NULL);
186  int hits;
187 
188  if (timeout)
189  {
190  /* check for an expired lease */
191  e = this->db->query(this->db,
192  "SELECT id, address FROM addresses "
193  "WHERE pool = ? AND released != 0 AND released < ? LIMIT 1",
194  DB_UINT, pool, DB_UINT, now - timeout, DB_UINT, DB_BLOB);
195  }
196  else
197  {
198  /* with static leases, check for an unallocated address */
199  e = this->db->query(this->db,
200  "SELECT id, address FROM addresses "
201  "WHERE pool = ? AND identity = 0 LIMIT 1",
202  DB_UINT, pool, DB_UINT, DB_BLOB);
203 
204  }
205 
206  if (!e || !e->enumerate(e, &id, &address))
207  {
208  DESTROY_IF(e);
209  break;
210  }
211  address = chunk_clonea(address);
212  e->destroy(e);
213 
214  if (timeout)
215  {
216  hits = this->db->execute(this->db, NULL,
217  "UPDATE addresses SET "
218  "acquired = ?, released = 0, identity = ? "
219  "WHERE id = ? AND released != 0 AND released < ?",
220  DB_UINT, now, DB_UINT, identity,
221  DB_UINT, id, DB_UINT, now - timeout);
222  }
223  else
224  {
225  hits = this->db->execute(this->db, NULL,
226  "UPDATE addresses SET "
227  "acquired = ?, released = 0, identity = ? "
228  "WHERE id = ? AND identity = 0",
229  DB_UINT, now, DB_UINT, identity, DB_UINT, id);
230  }
231  if (hits > 0)
232  {
233  host_t *host;
234 
235  host = host_create_from_chunk(AF_UNSPEC, address, 0);
236  if (host)
237  {
238  DBG1(DBG_CFG, "acquired new lease for address %H in pool '%s'",
239  host, name);
240  return host;
241  }
242  }
243  }
244  DBG1(DBG_CFG, "no available address found in pool '%s'", name);
245  return NULL;
246 }

249  private_attr_sql_provider_t *this, linked_list_t *pools, ike_sa_t *ike_sa,
250  host_t *requested)
251 {
252  enumerator_t *enumerator;
253  host_t *address = NULL;
254  u_int identity, pool, timeout;
255  char *name;
256  int family;
257 
258  identity = get_identity(this, ike_sa);
259  if (identity)
260  {
261  family = requested->get_family(requested);
262  /* check for an existing lease in all pools */
263  enumerator = pools->create_enumerator(pools);
264  while (enumerator->enumerate(enumerator, &name))
265  {
266  pool = get_pool(this, name, family, &timeout);
267  if (pool)
268  {
269  address = check_lease(this, name, pool, identity);
270  if (address)
271  {
272  break;
273  }
274  }
275  }
276  enumerator->destroy(enumerator);
277 
278  if (!address)
279  {
280  /* get an unallocated address or expired lease */
281  enumerator = pools->create_enumerator(pools);
282  while (enumerator->enumerate(enumerator, &name))
283  {
284  pool = get_pool(this, name, family, &timeout);
285  if (pool)
286  {
287  address = get_lease(this, name, pool, timeout, identity);
288  if (address)
289  {
290  break;
291  }
292  }
293  }
294  enumerator->destroy(enumerator);
295  }
296  }
297  return address;
298 }

344  private_attr_sql_provider_t *this, linked_list_t *pools, ike_sa_t *ike_sa,

352  u_int count;
353  char *name;
354

356  if (ike_sa)

358  u_int identity = get_identity(this, ike_sa);
359 
360  pool_enumerator = pools->create_enumerator(pools);
361  while (pool_enumerator->enumerate(pool_enumerator, &name))
362  {
363  u_int attr_pool = get_attr_pool(this, name);
364  if (!attr_pool)
365  {
366  continue;
367  }
368 
369  attr_enumerator = this->db->query(this->db,
370  "SELECT count(*) FROM attributes "

372  DB_UINT, attr_pool, DB_UINT, identity, DB_UINT);
373 
374  if (attr_enumerator &&
375  attr_enumerator->enumerate(attr_enumerator, &count) &&
376  count != 0)
377  {
378  attr_enumerator->destroy(attr_enumerator);
379  attr_enumerator = this->db->query(this->db,
380  "SELECT type, value FROM attributes "
381  "WHERE pool = ? AND identity = ?", DB_UINT,

383  break;
384  }
385  DESTROY_IF(attr_enumerator);
386  attr_enumerator = NULL;
387  }
388  pool_enumerator->destroy(pool_enumerator);
389  }
390 
391  /* in a second step check for attributes that match name */
392  if (!attr_enumerator)
393  {
394  pool_enumerator = pools->create_enumerator(pools);
395  while (pool_enumerator->enumerate(pool_enumerator, &name))
396  {
397  u_int attr_pool = get_attr_pool(this, name);
398  if (!attr_pool)

400  continue;
401  }
402 
403  attr_enumerator = this->db->query(this->db,
404  "SELECT count(*) FROM attributes "
405  "WHERE pool = ? AND identity = 0",
406  DB_UINT, attr_pool, DB_UINT);
407 
408  if (attr_enumerator &&
409  attr_enumerator->enumerate(attr_enumerator, &count) &&
410  count != 0)
411  {
412  attr_enumerator->destroy(attr_enumerator);
413  attr_enumerator = this->db->query(this->db,
414  "SELECT type, value FROM attributes "
415  "WHERE pool = ? AND identity = 0",
416  DB_UINT, attr_pool, DB_INT, DB_BLOB);
417  break;
418  }
419  DESTROY_IF(attr_enumerator);
420  attr_enumerator = NULL;
421  }
422  pool_enumerator->destroy(pool_enumerator);
423  }
424 
425  /* lastly try to find global attributes */
426  if (!attr_enumerator)
427  {
428  attr_enumerator = this->db->query(this->db,
429  "SELECT type, value FROM attributes "
430  "WHERE pool = 0 AND identity = 0",
431  DB_INT, DB_BLOB);
432  }
433  }
434 
435  return (attr_enumerator ? attr_enumerator : enumerator_create_empty());
436 }
