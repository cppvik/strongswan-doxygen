236 METHOD(stroke_attribute_t, add_pool, void,
237  private_stroke_attribute_t *this, mem_pool_t *pool)
238 {
239  enumerator_t *enumerator;
240  mem_pool_t *current;
241  host_t *base;
242  int size;
243 
244  base = pool->get_base(pool);
245  size = pool->get_size(pool);
246 
247  this->lock->write_lock(this->lock);
248 
249  enumerator = this->pools->create_enumerator(this->pools);
250  while (enumerator->enumerate(enumerator, &current))
251  {
252  if (base && current->get_base(current) &&
253  base->ip_equals(base, current->get_base(current)) &&
254  size == current->get_size(current))
255  {
256  DBG1(DBG_CFG, "reusing virtual IP address pool %s",
257  current->get_name(current));
258  pool->destroy(pool);
259  pool = NULL;
260  break;
261  }
262  }
263  enumerator->destroy(enumerator);
264 
265  if (pool)
266  {
267  if (base)
268  {
269  DBG1(DBG_CFG, "adding virtual IP address pool %s",
270  pool->get_name(pool));
271  }
272  this->pools->insert_last(this->pools, pool);
273  }
274 
275  this->lock->unlock(this->lock);
276 }
