98 static chunk_t generate_natd_hash(private_ike_natd_t *this,
99  ike_sa_id_t *ike_sa_id, host_t *host)
100 {
101  chunk_t natd_chunk, spi_i_chunk, spi_r_chunk, addr_chunk, port_chunk;
102  chunk_t natd_hash;
103  u_int64_t spi_i, spi_r;
104  u_int16_t port;
105 
106  /* prepare all required chunks */
107  spi_i = ike_sa_id->get_initiator_spi(ike_sa_id);
108  spi_r = ike_sa_id->get_responder_spi(ike_sa_id);
109  spi_i_chunk.ptr = (void*)&spi_i;
110  spi_i_chunk.len = sizeof(spi_i);
111  spi_r_chunk.ptr = (void*)&spi_r;
112  spi_r_chunk.len = sizeof(spi_r);
113  port = htons(host->get_port(host));
114  port_chunk.ptr = (void*)&port;
115  port_chunk.len = sizeof(port);
116  addr_chunk = host->get_address(host);
117 
118  /* natd_hash = SHA1( spi_i | spi_r | address | port ) */
119  natd_chunk = chunk_cat("cccc", spi_i_chunk, spi_r_chunk, addr_chunk, port_chunk);
120  if (!this->hasher->allocate_hash(this->hasher, natd_chunk, &natd_hash))
121  {
122  natd_hash = chunk_empty;
123  }
124  DBG3(DBG_IKE, "natd_chunk %B", &natd_chunk);
125  DBG3(DBG_IKE, "natd_hash %B", &natd_hash);
126 
127  chunk_free(&natd_chunk);
128  return natd_hash;
129 }

134 static notify_payload_t *build_natd_payload(private_ike_natd_t *this,
135  notify_type_t type, host_t *host)
136 {
137  chunk_t hash;
138  notify_payload_t *notify;
139  ike_sa_id_t *ike_sa_id;
140  ike_cfg_t *config;
141 
142  ike_sa_id = this->ike_sa->get_id(this->ike_sa);
143  config = this->ike_sa->get_ike_cfg(this->ike_sa);
144  if (force_encap(config) && type == NAT_DETECTION_SOURCE_IP)
145  {
146  u_int32_t addr;
147 
148  /* chunk_hash() is randomly keyed so this produces a random IPv4 address
149  * that changes with every restart but otherwise stays the same */
150  addr = chunk_hash(chunk_from_chars(0x00, 0x00, 0x00, 0x00));
151  host = host_create_from_chunk(AF_INET, chunk_from_thing(addr), 0);
152  hash = generate_natd_hash(this, ike_sa_id, host);
153  host->destroy(host);
154  }
155  else
156  {
157  hash = generate_natd_hash(this, ike_sa_id, host);
158  }
159  if (!hash.len)
160  {
161  return NULL;
162  }
163  notify = notify_payload_create(PLV2_NOTIFY);
164  notify->set_notify_type(notify, type);
165  notify->set_notification_data(notify, hash);
166  chunk_free(&hash);
167 
168  return notify;
169 }

174 static void process_payloads(private_ike_natd_t *this, message_t *message)
175 {
176  enumerator_t *enumerator;
177  payload_t *payload;
178  notify_payload_t *notify;
179  chunk_t hash, src_hash, dst_hash;
180  ike_sa_id_t *ike_sa_id;
181  host_t *me, *other;
182  ike_cfg_t *config;
183 
184  /* Precompute NAT-D hashes for incoming NAT notify comparison */
185  ike_sa_id = message->get_ike_sa_id(message);
186  me = message->get_destination(message);
187  other = message->get_source(message);
188  dst_hash = generate_natd_hash(this, ike_sa_id, me);
189  src_hash = generate_natd_hash(this, ike_sa_id, other);
190 
191  DBG3(DBG_IKE, "precalculated src_hash %B", &src_hash);
192  DBG3(DBG_IKE, "precalculated dst_hash %B", &dst_hash);
193 
194  enumerator = message->create_payload_enumerator(message);
195  while (enumerator->enumerate(enumerator, &payload))
196  {
197  if (payload->get_type(payload) != PLV2_NOTIFY)
198  {
199  continue;
200  }
201  notify = (notify_payload_t*)payload;
202  switch (notify->get_notify_type(notify))
203  {
204  case NAT_DETECTION_DESTINATION_IP:
205  {
206  this->dst_seen = TRUE;
207  hash = notify->get_notification_data(notify);
208  if (!this->dst_matched)
209  {
210  DBG3(DBG_IKE, "received dst_hash %B", &hash);
211  if (chunk_equals(hash, dst_hash))
212  {
213  this->dst_matched = TRUE;
214  }
215  }
216  /* RFC4555 says we should also compare against IKE_SA_INIT
217  * NATD payloads, but this does not work: We are running
218  * there at port 500, but use 4500 afterwards... */
219  if (message->get_exchange_type(message) == INFORMATIONAL &&
220  this->initiator && !this->dst_matched)
221  {
222  this->mapping_changed = this->ike_sa->has_mapping_changed(
223  this->ike_sa, hash);
224  }
225  break;
226  }
227  case NAT_DETECTION_SOURCE_IP:
228  {
229  this->src_seen = TRUE;
230  if (!this->src_matched)
231  {
232  hash = notify->get_notification_data(notify);
233  DBG3(DBG_IKE, "received src_hash %B", &hash);
234  if (chunk_equals(hash, src_hash))
235  {
236  this->src_matched = TRUE;
237  }
238  }
239  break;
240  }
241  default:
242  break;
243  }
244  }
245  enumerator->destroy(enumerator);
246 
247  chunk_free(&src_hash);
248  chunk_free(&dst_hash);
249 
250  if (this->src_seen && this->dst_seen)
251  {
252  this->ike_sa->enable_extension(this->ike_sa, EXT_NATT);
253 
254  this->ike_sa->set_condition(this->ike_sa, COND_NAT_HERE,
255  !this->dst_matched);
256  this->ike_sa->set_condition(this->ike_sa, COND_NAT_THERE,
257  !this->src_matched);
258  config = this->ike_sa->get_ike_cfg(this->ike_sa);
259  if (this->dst_matched && this->src_matched &&
260  force_encap(config))
261  {
262  this->ike_sa->set_condition(this->ike_sa, COND_NAT_FAKE, TRUE);
263  }
264  }
265 }

267 METHOD(task_t, process_i, status_t,
268  private_ike_natd_t *this, message_t *message)
269 {
270  process_payloads(this, message);
271 
272  if (message->get_exchange_type(message) == IKE_SA_INIT)
273  {
274  peer_cfg_t *peer_cfg = this->ike_sa->get_peer_cfg(this->ike_sa);
275  if (this->ike_sa->has_condition(this->ike_sa, COND_NAT_ANY) ||
276  /* if peer supports NAT-T, we switch to port 4500 even
277  * if no NAT is detected. can't be done later (when we would know
278  * whether the peer supports MOBIKE) because there would be no
279  * exchange to actually do the switch (other than a forced DPD). */
280  (peer_cfg->use_mobike(peer_cfg) &&
281  this->ike_sa->supports_extension(this->ike_sa, EXT_NATT)))
282  {
283  this->ike_sa->float_ports(this->ike_sa);
284  }
285  }
286 
287  return SUCCESS;

290 METHOD(task_t, build_i, status_t,
291  private_ike_natd_t *this, message_t *message)
292 {
293  notify_payload_t *notify;
294  enumerator_t *enumerator;
295  ike_cfg_t *ike_cfg;
296  host_t *host;
297 
298  if (this->hasher == NULL)
299  {
300  DBG1(DBG_IKE, "unable to build NATD payloads, SHA1 not supported");
301  return NEED_MORE;
302  }
303 
304  ike_cfg = this->ike_sa->get_ike_cfg(this->ike_sa);
305 
306  /* destination is always set */
307  host = message->get_destination(message);
308  notify = build_natd_payload(this, NAT_DETECTION_DESTINATION_IP, host);
309  if (notify)
310  {
311  message->add_payload(message, (payload_t*)notify);
312  }
313 
314  /* source may be any, we have 3 possibilities to get our source address:
315  * 1. It is defined in the config => use the one of the IKE_SA
316  * 2. We do a routing lookup in the kernel interface
317  * 3. Include all possbile addresses
318  */
319  host = message->get_source(message);
320  if (!host->is_anyaddr(host) || force_encap(ike_cfg))
321  { /* 1. or if we force UDP encap, as it doesn't matter if it's %any */
322  notify = build_natd_payload(this, NAT_DETECTION_SOURCE_IP, host);
323  if (notify)
324  {
325  message->add_payload(message, (payload_t*)notify);
326  }
327  }
328  else
329  {
330  host = hydra->kernel_interface->get_source_addr(hydra->kernel_interface,
331  this->ike_sa->get_other_host(this->ike_sa), NULL);
332  if (host)
333  { /* 2. */
334  host->set_port(host, ike_cfg->get_my_port(ike_cfg));
335  notify = build_natd_payload(this, NAT_DETECTION_SOURCE_IP, host);
336  if (notify)
337  {
338  message->add_payload(message, (payload_t*)notify);
339  }
340  host->destroy(host);
341  }
342  else
343  { /* 3. */
344  enumerator = hydra->kernel_interface->create_address_enumerator(
345  hydra->kernel_interface, ADDR_TYPE_REGULAR);
346  while (enumerator->enumerate(enumerator, (void**)&host))
347  {
348  /* apply port 500 to host, but work on a copy */
349  host = host->clone(host);
350  host->set_port(host, ike_cfg->get_my_port(ike_cfg));
351  notify = build_natd_payload(this, NAT_DETECTION_SOURCE_IP, host);
352  host->destroy(host);
353  if (notify)
354  {
355  message->add_payload(message, (payload_t*)notify);
356  }
357  }
358  enumerator->destroy(enumerator);
359  }
360  }
361  return NEED_MORE;
362 }

382  return SUCCESS;

384 
385  /* initiator seems to support NAT detection, add response */
386  me = message->get_source(message);
387  notify = build_natd_payload(this, NAT_DETECTION_SOURCE_IP, me);
388  if (notify)
