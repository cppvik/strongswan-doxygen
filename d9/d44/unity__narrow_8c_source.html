71 static void narrow_initiator(private_unity_narrow_t *this, ike_sa_t *ike_sa,
72  child_cfg_t *cfg, linked_list_t *remote)
73 {
74  traffic_selector_t *current, *orig = NULL;
75  enumerator_t *enumerator;
76 
77  enumerator = this->handler->create_include_enumerator(this->handler,
78  ike_sa->get_id(ike_sa));
79  while (enumerator->enumerate(enumerator, &current))
80  {
81  if (orig == NULL)
82  { /* got one, replace original TS */
83  if (remote->remove_first(remote, (void**)&orig) != SUCCESS)
84  {
85  break;
86  }
87  }
88  narrow_ts(cfg, current, remote);
89  }
90  enumerator->destroy(enumerator);
91  if (orig)
92  {
93  DBG1(DBG_CFG, "narrowed CHILD_SA to %N %#R",
94  configuration_attribute_type_names,
95  UNITY_SPLIT_INCLUDE, remote);
96  orig->destroy(orig);
97  }
98  else
99  { /* since we originally changed the traffic selector to 0.0.0.0/0 local
100  * narrowing is not applied if no Split-Include attrs are received */
101  if (remote->remove_first(remote, (void**)&orig) == SUCCESS)
102  {
103  narrow_ts(cfg, orig, remote);
104  orig->destroy(orig);
105  }
106  }
107 }

169 METHOD(listener_t, narrow, bool,
170  private_unity_narrow_t *this, ike_sa_t *ike_sa, child_sa_t *child_sa,
171  narrow_hook_t type, linked_list_t *local, linked_list_t *remote)
172 {
173  if (ike_sa->get_version(ike_sa) == IKEV1 &&
174  ike_sa->supports_extension(ike_sa, EXT_CISCO_UNITY))
175  {
176  /* depending on who initiates a rekeying the hooks will not match the
177  * roles in the IKE_SA */
178  if (ike_sa->has_condition(ike_sa, COND_ORIGINAL_INITIATOR))
179  {
180  switch (type)
181  {
182  case NARROW_INITIATOR_PRE_AUTH:
183  case NARROW_RESPONDER:
184  if (has_split_includes(this, ike_sa))
185  {
186  narrow_pre(remote, "other");
187  }
188  break;
189  case NARROW_INITIATOR_POST_AUTH:
190  case NARROW_RESPONDER_POST:
191  narrow_initiator(this, ike_sa,
192  child_sa->get_config(child_sa), remote);
193  break;
194  default:
195  break;
196  }
197  }
198  else
199  {
200  switch (type)
201  {
202  case NARROW_INITIATOR_PRE_AUTH:
203  case NARROW_RESPONDER:
204  if (this->wildcard_ts->get(this->wildcard_ts, ike_sa))
205  {
206  narrow_pre(local, "us");
207 
208  }
209  break;
210  case NARROW_INITIATOR_POST_AUTH:
211  case NARROW_RESPONDER_POST:
212  if (this->wildcard_ts->get(this->wildcard_ts, ike_sa))
213  {
214  narrow_responder_post(child_sa->get_config(child_sa),
215  local);
216  }
217  break;
218  default:
219  break;
220  }
221  }
222  }
223  return TRUE;
224 }

226 METHOD(listener_t, message, bool,
227  private_unity_narrow_t *this, ike_sa_t *ike_sa, message_t *message,
228  bool incoming, bool plain)
229 {
230  traffic_selector_t *tsr = NULL, *wildcard;
231  enumerator_t *enumerator;
232  id_payload_t *id_payload;
233  payload_t *payload;
234  bool first = TRUE;
235 
236  if (!incoming || !plain ||
237  message->get_exchange_type(message) != QUICK_MODE ||
238  !ike_sa || !ike_sa->supports_extension(ike_sa, EXT_CISCO_UNITY))
239  {
240  return TRUE;
241  }
242  enumerator = message->create_payload_enumerator(message);
243  while (enumerator->enumerate(enumerator, &payload))
244  {
245  if (payload->get_type(payload) == PLV1_ID)
246  {
247  if (!first)
248  {
249  id_payload = (id_payload_t*)payload;
250  tsr = id_payload->get_ts(id_payload);
251  break;
252  }
253  first = FALSE;
254  }
255  }
256  enumerator->destroy(enumerator);
257  if (!tsr)
258  {
259  return TRUE;
260  }
261  wildcard = traffic_selector_create_from_cidr("0.0.0.0/0", 0, 0, 65535);
262  if (tsr->equals(tsr, wildcard))
263  {
264  this->wildcard_ts->put(this->wildcard_ts, ike_sa, ike_sa);
265  }
266  else
267  {
268  this->wildcard_ts->remove(this->wildcard_ts, ike_sa);
269  }
270  wildcard->destroy(wildcard);
271  tsr->destroy(tsr);
272  return TRUE;
273 }
