125 static bool endecode_enumerate(endecode_enum_t *this, vici_type_t *type,
126  char **name, chunk_t *data)
127 {
128  if (this->next)
129  {
130  *type = this->next->type;
131  *name = this->next->name;
132  *data = this->next->data;
133  if (this->next->type == VICI_END)
134  {
135  this->next = NULL;
136  }
137  else
138  {
139  this->next++;
140  }
141  return TRUE;
142  }
143  return FALSE;
144 }

161 static void compare_vici(enumerator_t *parse, enumerator_t *tmpl)
162 {
163  vici_type_t type, ttype;
164  char *name, *tname;
165  chunk_t data, tdata;;
166 
167  while (TRUE)
168  {
169  ck_assert(parse->enumerate(parse, &type, &name, &data));
170  ck_assert(tmpl->enumerate(tmpl, &ttype, &tname, &tdata));
171  ck_assert_int_eq(type, ttype);
172  switch (type)
173  {
174  case VICI_END:
175  return;
176  case VICI_SECTION_START:
177  case VICI_LIST_START:
178  ck_assert(streq(name, tname));
179  break;
180  case VICI_LIST_ITEM:
181  ck_assert(chunk_equals(data, tdata));
182  break;
183  case VICI_KEY_VALUE:
184  ck_assert(streq(name, tname));
185  ck_assert(chunk_equals(data, tdata));
186  break;
187  case VICI_SECTION_END:
188  case VICI_LIST_END:
189  break;
190  default:
191  ck_assert(FALSE);
192  break;
193  }
194  }
195 }
