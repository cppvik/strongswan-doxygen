60 static void pacman_dbg(debug_t group, level_t level, char *fmt, ...)
61 {
62  int priority = LOG_INFO;
63  char buffer[8192];
64  char *current = buffer, *next;
65  va_list args;
66 
67  if (level <= debug_level)
68  {
69  if (!stderr_quiet)
70  {
71  va_start(args, fmt);
72  vfprintf(stderr, fmt, args);
73  fprintf(stderr, "\n");
74  va_end(args);
75  }
76 
77  /* write in memory buffer first */
78  va_start(args, fmt);
79  vsnprintf(buffer, sizeof(buffer), fmt, args);
80  va_end(args);
81 
82  /* do a syslog with every line */
83  while (current)
84  {
85  next = strchr(current, '\n');
86  if (next)
87  {
88  *(next++) = '\0';
89  }
90  syslog(priority, "%s\n", current);
91  current = next;
92  }
93  }

120 static bool update_database(database_t *db, char *package, char *version,
121  bool security, stats_t *stats)
122 {
123  char *cur_version, *version_update = NULL, *version_delete = NULL;
124  int cur_security, security_update = 0, security_delete = 0;
125  int pac_id = 0, vid = 0, vid_update = 0, vid_delete = 0;
126  u_int cur_time;
127  bool add_version = TRUE;
128  enumerator_t *e;
129 
130  /* increment package count */
131  stats->packages++;
132 
133  /* check if package is already in database */
134  e = db->query(db, "SELECT id FROM packages WHERE name = ?",
135  DB_TEXT, package, DB_INT);
136  if (!e)
137  {
138  return FALSE;
139  }
140  if (!e->enumerate(e, &pac_id))
141  {
142  pac_id = 0;
143  }
144  e->destroy(e);
145 
146  if (!pac_id && security)
147  {
148  if (db->execute(db, &pac_id, "INSERT INTO packages (name) VALUES (?)",
149  DB_TEXT, package) != 1)
150  {
151  fprintf(stderr, "could not store package '%s' to database\n",
152  package);
153  return FALSE;
154  }
155  stats->new_packages++;
156  }
157 
158  /* check for package versions already in database */
159  e = db->query(db,
160  "SELECT id, release, security, time FROM versions "
161  "WHERE package = ? AND product = ?", DB_INT, pac_id,
162  DB_INT, stats->product, DB_INT, DB_TEXT, DB_INT, DB_UINT);
163  if (!e)
164  {
165  return FALSE;
166  }
167 
168  while (e->enumerate(e, &vid, &cur_version, &cur_security, &cur_time))
169  {
170  if (streq(version, cur_version))
171  {
172  /* already in data base */
173  add_version = FALSE;
174  break;
175  }
176  else if (stats->release >= cur_time)
177  {
178  if (security)
179  {
180  if (cur_security)
181  {
182  vid_update = vid;
183  version_update = strdup(cur_version);
184  security_update = cur_security;
185  }
186  else
187  {
188  vid_delete = vid;
189  version_delete = strdup(cur_version);
190  security_delete = cur_security;
191  }
192  }
193  else
194  {
195  if (!cur_security)
196  {
197  vid_update = vid;
198  version_update = strdup(cur_version);
199  security_update = cur_security;
200  }
201  }
202  }
203  else
204  {
205  if (security == cur_security)
206  {
207  add_version = FALSE;
208  }
209  }
210  }
211  e->destroy(e);
212 
213  if ((!vid && !security) || (vid && !add_version))
214  {
215  free(version_update);
216  free(version_delete);
217  return TRUE;
218  }
219 
220  if ((!vid && security) || (vid && !vid_update))
221  {
222  printf("%s (%s) %s\n", package, version, security ? "[s]" : "");
223 
224  if (db->execute(db, &vid,
225  "INSERT INTO versions "
226  "(package, product, release, security, time) "
227  "VALUES (?, ?, ?, ?, ?)", DB_INT, pac_id, DB_INT, stats->product,
228  DB_TEXT, version, DB_INT, security, DB_INT, stats->release) != 1)
229  {
230  fprintf(stderr, "could not store version '%s' to database\n",
231  version);
232  free(version_update);
233  free(version_delete);
234  return FALSE;
235  }
236  stats->new_versions++;
237  }
238  else
239  {
240  printf("%s (%s) %s updated by\n",
241  package, version_update, security_update ? "[s]" : "");
242  printf("%s (%s) %s\n", package, version, security ? "[s]" : "");
243 
244  if (db->execute(db, NULL,
245  "UPDATE versions SET release = ?, time = ? WHERE id = ?",
246  DB_TEXT, version, DB_INT, stats->release, DB_INT, vid_update) <= 0)
247  {
248  fprintf(stderr, "could not update version '%s' to database\n",
249  version);
250  free(version_update);
251  free(version_delete);
252  return FALSE;
253  }
254  stats->updated_versions++;
255  }
256 
257  if (vid_delete)
258  {
259  printf("%s (%s) %s deleted\n",
260  package, version_delete, security_delete ? "[s]" : "");
261  if (db->execute(db, NULL,
262  "DELETE FROM versions WHERE id = ?",
263  DB_INT, vid_delete) <= 0)
264  {
265  fprintf(stderr, "could not delete version '%s' from database\n",
266  version_delete);
267  free(version_update);
268  free(version_delete);
269  return FALSE;
270  }
271  stats->deleted_versions++;
272  }
273  free(version_update);
274  free(version_delete);
275 
276  return TRUE;
277 }

282 static void process_packages(char *filename, char *product, bool security)
283 {
284  char *uri, line[BUF_LEN], *pos, *package = NULL, *version = NULL;
285  pacman_state_t pacman_state;
286  enumerator_t *e;
287  database_t *db;
288  int pid;
289  FILE *file;
290  stats_t stats;
291  bool success;
292 
293  /* initialize statistics */
294  memset(&stats, 0x00, sizeof(stats_t));
295 
296  /* Set release date to current time */
297  stats.release = time(NULL);
298 
299  /* opening package file */
300  printf("loading\"%s\"\n", filename);
301  file = fopen(filename, "r");
302  if (!file)
303  {
304  fprintf(stderr, "could not open \"%s\"\n", filename);
305  exit(EXIT_FAILURE);
306  }
307 
308  /* connect package database */
309  uri = lib->settings->get_str(lib->settings, "pacman.database", NULL);
310  if (!uri)
311  {
312  fprintf(stderr, "database URI pacman.database not set\n");
313  fclose(file);
314  exit(EXIT_FAILURE);
315  }
316  db = lib->db->create(lib->db, uri);
317  if (!db)
318  {
319  fprintf(stderr, "could not connect to database '%s'\n", uri);
320  fclose(file);
321  exit(EXIT_FAILURE);
322  }
323 
324  /* check if product is already in database */
325  e = db->query(db, "SELECT id FROM products WHERE name = ?",
326  DB_TEXT, product, DB_INT);
327  if (e)
328  {
329  if (e->enumerate(e, &pid))
330  {
331  stats.product = pid;
332  }
333  e->destroy(e);
334  }
335  if (!stats.product)
336  {
337  if (db->execute(db, &pid, "INSERT INTO products (name) VALUES (?)",
338  DB_TEXT, product) != 1)
339  {
340  fprintf(stderr, "could not store product '%s' to database\n",
341  product);
342  fclose(file);
343  db->destroy(db);
344  exit(EXIT_FAILURE);
345  }
346  stats.product = pid;
347  }
348 
349  pacman_state = PACMAN_STATE_BEGIN_PACKAGE;
350 
351  while (fgets(line, sizeof(line), file))
352  {
353  /* set read pointer to beginning of line */
354  pos = line;
355 
356  switch (pacman_state)
357  {
358  case PACMAN_STATE_BEGIN_PACKAGE:
359  pos = strstr(pos, "Package: ");
360  if (!pos)
361  {
362  continue;
363  }
364  pos += 9;
365  package = pos;
366  pos = strchr(pos, '\n');
367  if (pos)
368  {
369  package = strndup(package, pos - package);
370  pacman_state = PACMAN_STATE_VERSION;
371  }
372  break;
373  case PACMAN_STATE_VERSION:
374  pos = strstr(pos, "Version: ");
375  if (!pos)
376  {
377  continue;
378  }
379  pos += 9;
380  version = pos;
381  pos = strchr(pos, '\n');
382  if (pos)
383  {
384  version = strndup(version, pos - version);
385  pacman_state = PACMAN_STATE_END_PACKAGE;
386  }
387  break;
388  case PACMAN_STATE_END_PACKAGE:
389  if (*pos != '\n')
390  {
391  continue;
392  }
393  success = update_database(db, package, version, security, &stats);
394  free(package);
395  free(version);
396  if (!success)
397  {
398  fclose(file);
399  db->destroy(db);
400  exit(EXIT_FAILURE);
401  }
402  pacman_state = PACMAN_STATE_BEGIN_PACKAGE;
403  }
404  }
405  switch (pacman_state)
406  {
407  case PACMAN_STATE_END_PACKAGE:
408  free(version);
409  /* fall-through */
410  case PACMAN_STATE_VERSION:
411  free(package);
412  break;
413  default:
414  break;
415  }
416  fclose(file);
417  db->destroy(db);
418 
419  printf("processed %d packages, %d new packages, %d new versions, "
420  "%d updated versions, %d deleted versions\n",
421  stats.packages, stats.new_packages, stats.new_versions,
422  stats.updated_versions, stats.deleted_versions);
423 }

425 static void do_args(int argc, char *argv[])
426 {
427  char *filename = NULL, *product = NULL;
428  bool security = FALSE;
429 
430  /* reinit getopt state */
431  optind = 0;
432 
433  while (TRUE)
434  {
435  int c;
436 
437  struct option long_opts[] = {
438  { "help", no_argument, NULL, 'h' },
439  { "file", required_argument, NULL, 'f' },
440  { "product", required_argument, NULL, 'p' },
441  { "security", no_argument, NULL, 's' },
442  { 0,0,0,0 }
443  };
444 
445  c = getopt_long(argc, argv, "", long_opts, NULL);
446  switch (c)
447  {
448  case EOF:
449  break;
450  case 'h':
451  usage();
452  exit(EXIT_SUCCESS);
453  case 'f':
454  filename = optarg;
455  continue;
456  case 'p':
457  product = optarg;
458  continue;
459  case 's':
460  security = TRUE;
461  continue;
462  }
463  break;
464  }
465 
466  if (filename && product)
467  {
468  process_packages(filename, product, security);
469  }
470  else
471  {
472  usage();
473  exit(EXIT_FAILURE);
474  }
475 }
