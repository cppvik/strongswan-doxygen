283 METHOD(imv_swid_state_t, set_swid_inventory, void,
284  private_imv_swid_state_t *this, swid_inventory_t *inventory)
285 {
286  chunk_t tag_creator, unique_sw_id;
287  char software_id[256];
288  json_object *jstring;
289  swid_tag_id_t *tag_id;
290  enumerator_t *enumerator;
291 
292  enumerator = inventory->create_enumerator(inventory);
293  while (enumerator->enumerate(enumerator, &tag_id))
294  {
295  /* Construct software ID from tag creator and unique software ID */
296  tag_creator = tag_id->get_tag_creator(tag_id);
297  unique_sw_id = tag_id->get_unique_sw_id(tag_id, NULL);
298  snprintf(software_id, 256, "%.*s_%.*s",
299  tag_creator.len, tag_creator.ptr,
300  unique_sw_id.len, unique_sw_id.ptr);
301  DBG3(DBG_IMV, " %s", software_id);
302 
303  /* Add software ID to JSON array */
304  jstring = json_object_new_string(software_id);
305  json_object_array_add(this->jarray, jstring);
306  }
307  enumerator->destroy(enumerator);
308 }
