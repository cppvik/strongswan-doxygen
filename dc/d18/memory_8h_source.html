93 static inline void memwipe_inline(void *ptr, size_t n)
94 {
95  volatile char *c = (volatile char*)ptr;
96  size_t m, i;
97 
98  /* byte wise until long aligned */
99  for (i = 0; (uintptr_t)&c[i] % sizeof(long) && i < n; i++)
100  {
101  c[i] = 0;
102  }
103  /* word wise */
104  if (n >= sizeof(long))
105  {
106  for (m = n - sizeof(long); i <= m; i += sizeof(long))
107  {
108  *(volatile long*)&c[i] = 0;
109  }
110  }
111  /* byte wise of the rest */
112  for (; i < n; i++)
113  {
114  c[i] = 0;
115  }
