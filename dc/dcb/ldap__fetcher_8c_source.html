51 static bool parse(LDAP *ldap, LDAPMessage *result, chunk_t *response)
52 {
53  LDAPMessage *entry = ldap_first_entry(ldap, result);
54  bool success = FALSE;
55 
56  if (entry)
57  {
58  BerElement *ber = NULL;
59  char *attr;
60 
61  attr = ldap_first_attribute(ldap, entry, &ber);
62  if (attr)
63  {
64  struct berval **values = ldap_get_values_len(ldap, entry, attr);
65 
66  if (values)
67  {
68  if (values[0])
69  {
70  *response = chunk_alloc(values[0]->bv_len);
71  memcpy(response->ptr, values[0]->bv_val, response->len);
72  success = TRUE;
73  }
74  else
75  {
76  DBG1(DBG_LIB, "LDAP response contains no values");
77  }
78  ldap_value_free_len(values);
79  }
80  else
81  {
82  DBG1(DBG_LIB, "getting LDAP values failed: %s",
83  ldap_err2string(ldap_result2error(ldap, entry, 0)));
84  }
85  ldap_memfree(attr);
86  }
87  else
88  {
89  DBG1(DBG_LIB, "finding LDAP attributes failed: %s",
90  ldap_err2string(ldap_result2error(ldap, entry, 0)));
91  }
92  ber_free(ber, 0);
93  }
94  else
95  {
96  DBG1(DBG_LIB, "finding first LDAP entry failed: %s",
97  ldap_err2string(ldap_result2error(ldap, entry, 0)));
98  }
99  return success;
100 }

103 METHOD(fetcher_t, fetch, status_t,
104  private_ldap_fetcher_t *this, char *url, void *userdata)
105 {
106  LDAP *ldap;
107  LDAPURLDesc *lurl;
108  LDAPMessage *msg;
109  int res;
110  int ldap_version = LDAP_VERSION3;
111  struct timeval timeout;
112  status_t status = FAILED;
113  chunk_t *result = userdata;
114 
115  if (!strpfx(url, "ldap"))
116  {
117  return NOT_SUPPORTED;
118  }
119  if (ldap_url_parse(url, &lurl) != LDAP_SUCCESS)
120  {
121  return NOT_SUPPORTED;
122  }
123  ldap = ldap_init(lurl->lud_host, lurl->lud_port);
124  if (ldap == NULL)
125  {
126  DBG1(DBG_LIB, "LDAP initialization failed: %s", strerror(errno));
127  ldap_free_urldesc(lurl);
128  return FAILED;
129  }
130 
131  timeout.tv_sec = this->timeout;
132  timeout.tv_usec = 0;
133 
134  ldap_set_option(ldap, LDAP_OPT_PROTOCOL_VERSION, &ldap_version);
135  ldap_set_option(ldap, LDAP_OPT_NETWORK_TIMEOUT, &timeout);
136 
137  DBG2(DBG_LIB, "sending LDAP request to '%s'...", url);
138 
139  res = ldap_simple_bind_s(ldap, NULL, NULL);
140  if (res == LDAP_SUCCESS)
141  {
142  res = ldap_search_st(ldap, lurl->lud_dn, lurl->lud_scope,
143  lurl->lud_filter, lurl->lud_attrs,
144  0, &timeout, &msg);
145 
146  if (res == LDAP_SUCCESS)
147  {
148  if (parse(ldap, msg, result))
149  {
150  status = SUCCESS;
151  }
152  ldap_msgfree(msg);
153  }
154  else
155  {
156  DBG1(DBG_LIB, "LDAP search failed: %s", ldap_err2string(res));
157  }
158  }
159  else
160  {
161  DBG1(DBG_LIB, "LDAP bind to '%s' failed: %s", url,
162  ldap_err2string(res));
163  }
164  ldap_unbind_s(ldap);
165  ldap_free_urldesc(lurl);
166  return status;
167 }
