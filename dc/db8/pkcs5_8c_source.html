111 static bool verify_padding(crypter_t *crypter, chunk_t *blob)
112 {
113  u_int8_t padding, count;
114 
115  padding = count = blob->ptr[blob->len - 1];
116 
117  if (padding > crypter->get_block_size(crypter))
118  {
119  return FALSE;
120  }
121  for (; blob->len && count; --blob->len, --count)
122  {
123  if (blob->ptr[blob->len - 1] != padding)
124  {
125  return FALSE;
126  }
127  }
128  return TRUE;
129 }

141 static bool decrypt_generic(private_pkcs5_t *this, chunk_t password,
142  chunk_t data, chunk_t *decrypted, kdf_t kdf,
143  chunk_t keymat, chunk_t key, chunk_t iv)
144 {
145  if (!kdf(this, password, keymat))
146  {
147  return FALSE;
148  }
149  if (!this->crypter->set_key(this->crypter, key) ||
150  !this->crypter->decrypt(this->crypter, data, iv, decrypted))
151  {
152  memwipe(keymat.ptr, keymat.len);
153  return FALSE;
154  }
155  memwipe(keymat.ptr, keymat.len);
156  if (verify_padding(this->crypter, decrypted))
157  {
158  return TRUE;
159  }
160  chunk_free(decrypted);
161  return FALSE;
162 }

183 static bool pbkdf2_f(chunk_t block, prf_t *prf, chunk_t seed,
184  u_int64_t iterations)
185 {
186  chunk_t u;
187  u_int64_t i;
188 
189  u = chunk_alloca(prf->get_block_size(prf));
190  if (!prf->get_bytes(prf, seed, u.ptr))
191  {
192  return FALSE;
193  }
194  memcpy(block.ptr, u.ptr, block.len);
195 
196  for (i = 1; i < iterations; i++)
197  {
198  if (!prf->get_bytes(prf, u, u.ptr))
199  {
200  return FALSE;
201  }
202  memxor(block.ptr, u.ptr, block.len);
203  }
204  return TRUE;
205 }

210 static bool pbkdf2(private_pkcs5_t *this, chunk_t password, chunk_t key)
211 {
212  prf_t *prf;
213  chunk_t keymat, block, seed;
214  size_t blocks;
215  u_int32_t i = 0;
216 
217  prf = this->data.pbes2.prf;
218 
219  if (!prf->set_key(prf, password))
220  {
221  return FALSE;
222  }
223 
224  block.len = prf->get_block_size(prf);
225  blocks = (key.len - 1) / block.len + 1;
226  keymat = chunk_alloca(blocks * block.len);
227 
228  seed = chunk_cata("cc", this->salt, chunk_from_thing(i));
229 
230  for (; i < blocks; i++)
231  {
232  htoun32(seed.ptr + this->salt.len, i + 1);
233  block.ptr = keymat.ptr + (i * block.len);
234  if (!pbkdf2_f(block, prf, seed, this->iterations))
235  {
236  return FALSE;
237  }
238  }
239  memcpy(key.ptr, keymat.ptr, key.len);
240  return TRUE;
241 }

246 static bool pbkdf1(private_pkcs5_t *this, chunk_t password, chunk_t key)
247 {
248  hasher_t *hasher;

252  hasher = this->data.pbes1.hasher;
253 
254  hash = chunk_alloca(hasher->get_hash_size(hasher));
255  if (!hasher->get_hash(hasher, password, NULL) ||
256  !hasher->get_hash(hasher, this->salt, hash.ptr))
257  {
258  return FALSE;
259  }
260 
261  for (i = 1; i < this->iterations; i++)
262  {
263  if (!hasher->get_hash(hasher, hash, hash.ptr))

265  return FALSE;
266  }
267  }

272 static bool ensure_crypto_primitives(private_pkcs5_t *this, chunk_t data)
273 {
274  if (!this->crypter)
275  {
276  this->crypter = lib->crypto->create_crypter(lib->crypto, this->encr,
277  this->keylen);
278  if (!this->crypter)
279  {
280  DBG1(DBG_ASN, " %N encryption algorithm not available",
281  encryption_algorithm_names, this->encr);
282  return FALSE;
283  }
284  }
285  if (data.len % this->crypter->get_block_size(this->crypter))
286  {
287  DBG1(DBG_ASN, " data size is not a multiple of block size");
288  return FALSE;
289  }
290  switch (this->scheme)
291  {
292  case PKCS5_SCHEME_PBES1:
293  {
294  if (!this->data.pbes1.hasher)
295  {
296  hasher_t *hasher;
297 
298  hasher = lib->crypto->create_hasher(lib->crypto,
299  this->data.pbes1.hash);
300  if (!hasher)
301  {
302  DBG1(DBG_ASN, " %N hash algorithm not available",
303  hash_algorithm_names, this->data.pbes1.hash);
304  return FALSE;
305  }
306  if (hasher->get_hash_size(hasher) < this->keylen)
307  {
308  hasher->destroy(hasher);
309  return FALSE;
310  }
311  this->data.pbes1.hasher = hasher;
312  }
313  break;
314  }
315  case PKCS5_SCHEME_PBES2:
316  {
317  if (!this->data.pbes2.prf)
318  {
319  prf_t *prf;
320 
321  prf = lib->crypto->create_prf(lib->crypto,
322  this->data.pbes2.prf_alg);
323  if (!prf)
324  {
325  DBG1(DBG_ASN, " %N prf algorithm not available",
326  pseudo_random_function_names,
327  this->data.pbes2.prf_alg);
328  return FALSE;
329  }
330  this->data.pbes2.prf = prf;
331  }
332  break;
333  }
334  case PKCS5_SCHEME_PKCS12:
335  break;
336  }
337  return TRUE;
338 }

340 METHOD(pkcs5_t, decrypt, bool,
341  private_pkcs5_t *this, chunk_t password, chunk_t data, chunk_t *decrypted)
342 {
343  chunk_t keymat, key, iv;
344  kdf_t kdf;
345 
346  if (!ensure_crypto_primitives(this, data) || !decrypted)
347  {
348  return FALSE;
349  }
350  kdf = pbkdf1;
351  switch (this->scheme)
352  {
353  case PKCS5_SCHEME_PKCS12:
354  kdf = pkcs12_kdf;
355  /* fall-through */
356  case PKCS5_SCHEME_PBES1:
357  keymat = chunk_alloca(this->keylen +
358  this->crypter->get_iv_size(this->crypter));
359  key = chunk_create(keymat.ptr, this->keylen);
360  iv = chunk_create(keymat.ptr + this->keylen,
361  keymat.len - this->keylen);
362  break;
363  case PKCS5_SCHEME_PBES2:
364  kdf = pbkdf2;
365  keymat = chunk_alloca(this->keylen);
366  key = keymat;
367  iv = this->data.pbes2.iv;
368  break;
369  default:
370  return FALSE;
371  }
372  return decrypt_generic(this, password, data, decrypted, kdf,
373  keymat, key, iv);
374 }

503 static bool parse_pbes2_params(private_pkcs5_t *this, chunk_t blob, int level0)
504 {
505  asn1_parser_t *parser;
506  chunk_t object, params;
507  size_t keylen;
508  int objectID;
509  bool success = FALSE;
510 
511  parser = asn1_parser_create(pbes2ParamsObjects, blob);
512  parser->set_top_level(parser, level0);
513 
514  while (parser->iterate(parser, &objectID, &object))
515  {
516  switch (objectID)
517  {
518  case PBES2PARAMS_KEY_DERIVATION_FUNC:
519  {
520  int oid = asn1_parse_algorithmIdentifier(object,
521  parser->get_level(parser) + 1, &params);
522  if (oid != OID_PBKDF2)
523  { /* unsupported key derivation function */
524  goto end;
525  }
526  if (!parse_pbkdf2_params(this, params,
527  parser->get_level(parser) + 1))
528  {
529  goto end;
530  }
531  break;
532  }
533  case PBES2PARAMS_ENCRYPTION_SCHEME:
534  {
535  int oid = asn1_parse_algorithmIdentifier(object,
536  parser->get_level(parser) + 1, &params);
537  this->encr = encryption_algorithm_from_oid(oid, &keylen);
538  if (this->encr == ENCR_UNDEFINED)
539  { /* unsupported encryption scheme */
540  goto end;
541  }
542  /* prefer encoded key length */
543  this->keylen = this->keylen ?: keylen / 8;
544  if (!this->keylen)
545  { /* set default key length for known algorithms */
546  switch (this->encr)
547  {
548  case ENCR_DES:
549  this->keylen = 8;
550  break;
551  case ENCR_3DES:
552  this->keylen = 24;
553  break;
554  case ENCR_BLOWFISH:
555  this->keylen = 16;
556  break;
557  default:
558  goto end;
559  }
560  }
561  if (!asn1_parse_simple_object(&params, ASN1_OCTET_STRING,
562  parser->get_level(parser) + 1, "IV"))
563  {
564  goto end;
565  }
566  this->data.pbes2.iv = chunk_clone(params);
567  break;
568  }
569  }
570  }
571  success = parser->success(parser);
572 end:
573  parser->destroy(parser);
574  return success;
575 }

601 {

604  int oid;
605

622  break;
623  case OID_PBE_SHA1_DES_CBC:
624  this->encr = ENCR_DES;
625  this->data.pbes1.hash = HASH_SHA1;
626  break;
627  case OID_PBE_SHA1_3DES_CBC:
628  this->scheme = PKCS5_SCHEME_PKCS12;

630  this->encr = ENCR_3DES;
631  this->data.pbes1.hash = HASH_SHA1;
632  break;
633  case OID_PBE_SHA1_RC2_CBC_40:
634  case OID_PBE_SHA1_RC2_CBC_128:
635  this->scheme = PKCS5_SCHEME_PKCS12;
636  this->keylen = (oid == OID_PBE_SHA1_RC2_CBC_40) ? 5 : 16;
637  this->encr = ENCR_RC2_CBC;
638  this->data.pbes1.hash = HASH_SHA1;
639  break;
640  case OID_PBES2:
641  this->scheme = PKCS5_SCHEME_PBES2;
642  break;
643  default:
644  /* encryption scheme not supported */
645  goto failure;
646  }
647

650  case PKCS5_SCHEME_PBES1:
651  case PKCS5_SCHEME_PKCS12:
652  if (!parse_pbes1_params(this, params, level0))

655  }
656  break;
657  case PKCS5_SCHEME_PBES2:
658  if (!parse_pbes2_params(this, params, level0))
659  {
660  goto failure;
661  }
662  break;
663  }
664  return &this->public;
665
