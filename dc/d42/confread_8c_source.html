56 static bool is_deprecated(kw_token_t token, char *name, char *conn)
57 {
58  switch (token)
59  {
60  case KW_SETUP_DEPRECATED:
61  case KW_PKCS11_DEPRECATED:
62  DBG1(DBG_APP, "# deprecated keyword '%s' in config setup", name);
63  break;
64  case KW_CONN_DEPRECATED:
65  case KW_END_DEPRECATED:
66  case KW_PFS_DEPRECATED:
67  DBG1(DBG_APP, "# deprecated keyword '%s' in conn '%s'", name, conn);
68  break;
69  case KW_CA_DEPRECATED:
70  DBG1(DBG_APP, "# deprecated keyword '%s' in ca '%s'", name, conn);
71  break;
72  default:
73  return FALSE;
74  }
75  /* additional messages for some */
76  switch (token)
77  {
78  case KW_PKCS11_DEPRECATED:
79  DBG1(DBG_APP, " use the 'pkcs11' plugin instead");
80  break;
81  case KW_PFS_DEPRECATED:
82  DBG1(DBG_APP, " PFS is enabled by specifying a DH group in the "
83  "'esp' cipher suite");
84  break;
85  default:
86  break;
87  }
88  return TRUE;
89 }

103  if (!dict)

105  return;

239 static void kw_end(starter_conn_t *conn, starter_end_t *end, kw_token_t token,
240  char *key, char *value, starter_config_t *cfg)
241 {
242  bool assigned = FALSE;
243 
244  if (is_deprecated(token, key, conn->name))
245  {
246  cfg->non_fatal_err++;
247  return;
248  }
249 
250  if (!assign_arg(token, KW_END_FIRST, key, value, end, &assigned))
251  {
252  goto err;
253  }
254 
255  /* post processing of some keywords that were assigned automatically */
256  switch (token)
257  {
258  case KW_HOST:
259  if (value && strlen(value) > 0 && value[0] == '%')
260  {
261  if (streq(value, "%defaultroute"))
262  {
263  value = "%any";
264  }
265  if (!streq(value, "%any") && !streq(value, "%any4") &&
266  !streq(value, "%any6"))
267  { /* allow_any prefix */
268  end->allow_any = TRUE;
269  value++;
270  }
271  }
272  free(end->host);
273  end->host = strdupnull(value);
274  break;
275  case KW_SOURCEIP:
276  conn->mode = MODE_TUNNEL;
277  conn->proxy_mode = FALSE;
278  break;
279  case KW_SENDCERT:
280  if (end->sendcert == CERT_YES_SEND)
281  {
282  end->sendcert = CERT_ALWAYS_SEND;
283  }
284  else if (end->sendcert == CERT_NO_SEND)
285  {
286  end->sendcert = CERT_NEVER_SEND;
287  }
288  break;
289  default:
290  break;
291  }
292 
293  if (assigned)
294  {
295  return;
296  }
297 
298  /* individual processing of keywords that were not assigned automatically */
299  switch (token)
300  {
301  case KW_PROTOPORT:
302  {
303  struct protoent *proto;
304  struct servent *svc;
305  char *sep, *port = "", *endptr;
306  long int p;
307 
308  sep = strchr(value, '/');
309  if (sep)
310  { /* protocol/port */
311  *sep = '\0';
312  port = sep + 1;
313  }
314 
315  if (streq(value, "%any"))
316  {
317  end->protocol = 0;
318  }
319  else
320  {
321  proto = getprotobyname(value);
322  if (proto)
323  {
324  end->protocol = proto->p_proto;
325  }
326  else
327  {
328  p = strtol(value, &endptr, 0);
329  if ((*value && *endptr) || p < 0 || p > 0xff)
330  {
331  DBG1(DBG_APP, "# bad protocol: %s=%s", key, value);
332  goto err;
333  }
334  end->protocol = (u_int8_t)p;
335  }
336  }
337  if (streq(port, "%any"))
338  {
339  end->from_port = 0;
340  end->to_port = 0xffff;
341  }
342  else if (streq(port, "%opaque"))
343  {
344  end->from_port = 0xffff;
345  end->to_port = 0;
346  }
347  else if (*port)
348  {
349  svc = getservbyname(port, NULL);
350  if (svc)
351  {
352  end->from_port = end->to_port = ntohs(svc->s_port);
353  }
354  else
355  {
356  p = strtol(port, &endptr, 0);
357  if (p < 0 || p > 0xffff)
358  {
359  DBG1(DBG_APP, "# bad port: %s=%s", key, port);
360  goto err;
361  }
362  end->from_port = p;
363  if (*endptr == '-')
364  {
365  port = endptr + 1;
366  p = strtol(port, &endptr, 0);
367  if (p < 0 || p > 0xffff)
368  {
369  DBG1(DBG_APP, "# bad port: %s=%s", key, port);
370  goto err;
371  }
372  }
373  end->to_port = p;
374  if (*endptr)
375  {
376  DBG1(DBG_APP, "# bad port: %s=%s", key, port);
377  goto err;
378  }
379  }
380  }
381  if (sep)
382  { /* restore the original text in case also= is used */
383  *sep = '/';
384  }
385  break;
386  }
387  default:
388  break;
389  }
390  return;
391 
392 err:
393  DBG1(DBG_APP, " bad argument value in conn '%s'", conn->name);
394  cfg->err++;
395 }

405 /*
406  * parse settings not handled by the simple argument parser
407  */
408 static void handle_keyword(kw_token_t token, starter_conn_t *conn, char *key,
409  char *value, starter_config_t *cfg)
410 {
411  if ((token == KW_ESP && conn->ah) || (token == KW_AH && conn->esp))
412  {
413  DBG1(DBG_APP, "# can't have both 'ah' and 'esp' options");
414  cfg->err++;
415  return;
416  }
417  switch (token)
418  {
419  case KW_TYPE:
420  {
421  conn->mode = MODE_TRANSPORT;
422  conn->proxy_mode = FALSE;
423  if (streq(value, "tunnel"))
424  {
425  conn->mode = MODE_TUNNEL;
426  }
427  else if (streq(value, "beet"))
428  {
429  conn->mode = MODE_BEET;
430  }
431  else if (streq(value, "transport_proxy"))
432  {
433  conn->mode = MODE_TRANSPORT;
434  conn->proxy_mode = TRUE;
435  }
436  else if (streq(value, "passthrough") || streq(value, "pass"))
437  {
438  conn->mode = MODE_PASS;
439  }
440  else if (streq(value, "drop") || streq(value, "reject"))
441  {
442  conn->mode = MODE_DROP;
443  }
444  else if (!streq(value, "transport"))
445  {
446  DBG1(DBG_APP, "# bad policy value: %s=%s", key, value);
447  cfg->err++;
448  }
449  break;
450  }
451  case KW_COMPRESS:
452  KW_SA_OPTION_FLAG("yes", "no", SA_OPTION_COMPRESS)
453  break;
454  case KW_MARK:
455  if (!mark_from_string(value, &conn->mark_in))
456  {
457  cfg->err++;
458  break;
459  }
460  conn->mark_out = conn->mark_in;
461  break;
462  case KW_MARK_IN:
463  if (!mark_from_string(value, &conn->mark_in))
464  {
465  cfg->err++;
466  }
467  break;
468  case KW_MARK_OUT:
469  if (!mark_from_string(value, &conn->mark_out))
470  {
471  cfg->err++;
472  }
473  break;
474  case KW_TFC:
475  if (streq(value, "%mtu"))
476  {
477  conn->tfc = -1;
478  }
479  else
480  {
481  char *endptr;
482 
483  conn->tfc = strtoul(value, &endptr, 10);
484  if (*endptr != '\0')
485  {
486  DBG1(DBG_APP, "# bad integer value: %s=%s", key, value);
487  cfg->err++;
488  }
489  }
490  break;
491  case KW_KEYINGTRIES:
492  if (streq(value, "%forever"))
493  {
494  conn->sa_keying_tries = 0;
495  }
496  else
497  {
498  char *endptr;
499 
500  conn->sa_keying_tries = strtoul(value, &endptr, 10);
501  if (*endptr != '\0')
502  {
503  DBG1(DBG_APP, "# bad integer value: %s=%s", key, value);
504  cfg->err++;
505  }
506  }
507  break;
508  case KW_REKEY:
509  KW_SA_OPTION_FLAG("no", "yes", SA_OPTION_DONT_REKEY)
510  break;
511  case KW_REAUTH:
512  KW_SA_OPTION_FLAG("no", "yes", SA_OPTION_DONT_REAUTH)
513  break;
514  case KW_MOBIKE:
515  KW_SA_OPTION_FLAG("yes", "no", SA_OPTION_MOBIKE)
516  break;
517  case KW_FORCEENCAPS:
518  KW_SA_OPTION_FLAG("yes", "no", SA_OPTION_FORCE_ENCAP)
519  break;
520  case KW_MODECONFIG:
521  KW_SA_OPTION_FLAG("push", "pull", SA_OPTION_MODECFG_PUSH)
522  break;
523  case KW_XAUTH:
524  KW_SA_OPTION_FLAG("server", "client", SA_OPTION_XAUTH_SERVER)
525  break;
526  default:
527  break;
528  }
529 }

554  * parse a conn section

556 static void load_conn(starter_conn_t *conn, starter_config_t *cfg,
557  conf_parser_t *parser)
558 {
559  enumerator_t *enumerator;
560  dictionary_t *dict;
561  kw_entry_t *entry;
562  kw_token_t token;
563  char *key, *value;
564 
565  DBG2(DBG_APP, "Loading conn '%s'", conn->name);
566  dict = parser->get_section(parser, CONF_PARSER_CONN, conn->name);
567  if (!dict)
568  {
569  return;
570  }
571  enumerator = dict->create_enumerator(dict);
572  while (enumerator->enumerate(enumerator, &key, &value))
573  {
574  bool assigned = FALSE;
575 
576  entry = in_word_set(key, strlen(key));
577  if (!entry)
578  {
579  DBG1(DBG_APP, "# unknown keyword '%s'", key);
580  cfg->non_fatal_err++;
581  continue;
582  }
583  token = entry->token;
584  if (token >= KW_LEFT_FIRST && token <= KW_LEFT_LAST)
585  {
586  kw_end(conn, &conn->left, token - KW_LEFT_FIRST + KW_END_FIRST,
587  key, value, cfg);
588  continue;
589  }
590  else if (token >= KW_RIGHT_FIRST && token <= KW_RIGHT_LAST)
591  {
592  kw_end(conn, &conn->right, token - KW_RIGHT_FIRST + KW_END_FIRST,
593  key, value, cfg);
594  continue;
595  }
596  if (token == KW_AUTO)
597  {
598  token = KW_CONN_SETUP;
599  }
600  if (token < KW_CONN_FIRST || token > KW_CONN_LAST)
601  {
602  DBG1(DBG_APP, "# unsupported keyword '%s' in conn '%s'",
603  key, conn->name);
604  cfg->err++;
605  continue;
606  }
607  if (is_deprecated(token, key, conn->name))
608  {
609  cfg->non_fatal_err++;
610  continue;
611  }
612  if (!assign_arg(token, KW_CONN_FIRST, key, value, conn,
613  &assigned))
614  {
615  DBG1(DBG_APP, " bad argument value in conn '%s'", conn->name);
616  cfg->err++;
617  continue;
618  }
619  if (!assigned)
620  {
621  handle_keyword(token, conn, key, value, cfg);
622  }
623  }
624  enumerator->destroy(enumerator);
625  dict->destroy(dict);
626 
627  handle_firewall("left", &conn->left, cfg);
628  handle_firewall("right", &conn->right, cfg);
629 
630  if (!conn->esp && !conn->ah)
631  {
632  conn->esp = strdupnull(esp_defaults);
633  }
634 }

685 /*
686  * load and parse an IPsec configuration file
687  */
688 starter_config_t* confread_load(const char *file)
689 {
690  conf_parser_t *parser;
691  starter_config_t *cfg = NULL;
692  enumerator_t *enumerator;
693  u_int total_err;
694  char *name;
695 
696  parser = conf_parser_create(file);
697  if (!parser->parse(parser))
698  {
699  parser->destroy(parser);
700  return NULL;
701  }
702 
703  INIT(cfg,
704  .setup = {
705  .uniqueids = TRUE,
706  }
707  );
708 
709  /* load config setup section */
710  load_setup(cfg, parser);
711 
712  /* load ca sections */
713  enumerator = parser->get_sections(parser, CONF_PARSER_CA);
714  while (enumerator->enumerate(enumerator, &name))
715  {
716  u_int previous_err = cfg->err;
717  starter_ca_t *ca;
718 
719  INIT(ca,
720  .name = strdup(name),
721  );
722  load_ca(ca, cfg, parser);
723 
724  if (cfg->err > previous_err)
725  {
726  total_err = cfg->err - previous_err;
727  DBG1(DBG_APP, "# ignored ca '%s' due to %d parsing error%s", name,
728  total_err, (total_err > 1) ? "s" : "");
729  confread_free_ca(ca);
730  cfg->non_fatal_err += cfg->err - previous_err;
731  cfg->err = previous_err;
732  }
733  else
734  {
735  if (cfg->ca_last)
736  {
737  cfg->ca_last->next = ca;
738  }
739  cfg->ca_last = ca;
740  if (!cfg->ca_first)
741  {
742  cfg->ca_first = ca;
