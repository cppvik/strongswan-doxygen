201 static conn_t *conn_get(private_mysql_database_t *this, transaction_t **trans)
202 {
203  conn_t *current, *found = NULL;
204  enumerator_t *enumerator;
205  transaction_t *transaction;
206 
207  thread_initialize();
208 
209  transaction = this->transaction->get(this->transaction);
210  if (transaction)
211  {
212  if (trans)
213  {
214  *trans = transaction;
215  }
216  return transaction->conn;
217  }
218 
219  while (TRUE)
220  {
221  this->mutex->lock(this->mutex);
222  enumerator = this->pool->create_enumerator(this->pool);
223  while (enumerator->enumerate(enumerator, &current))
224  {
225  if (!current->in_use)
226  {
227  found = current;
228  found->in_use = TRUE;
229  break;
230  }
231  }
232  enumerator->destroy(enumerator);
233  this->mutex->unlock(this->mutex);
234  if (found)
235  { /* check connection if found, release if ping fails */
236  if (mysql_ping(found->mysql) == 0)
237  {
238  break;
239  }
240  this->mutex->lock(this->mutex);
241  this->pool->remove(this->pool, found, NULL);
242  this->mutex->unlock(this->mutex);
243  conn_destroy(found);
244  found = NULL;
245  continue;
246  }
247  break;
248  }
249  if (found == NULL)
250  {
251  INIT(found,
252  .in_use = TRUE,
253  .mysql = mysql_init(NULL),
254  );
255  if (!mysql_real_connect(found->mysql, this->host, this->username,
256  this->password, this->database, this->port,
257  NULL, 0))
258  {
259  DBG1(DBG_LIB, "connecting to mysql://%s:***@%s:%d/%s failed: %s",
260  this->username, this->host, this->port, this->database,
261  mysql_error(found->mysql));
262  conn_destroy(found);
263  found = NULL;
264  }
265  else

267  this->mutex->lock(this->mutex);
268  this->pool->insert_last(this->pool, found);
269  DBG2(DBG_LIB, "increased MySQL connection pool size to %d",
270  this->pool->get_count(this->pool));

272  }
273  }

280 static MYSQL_STMT* run(MYSQL *mysql, char *sql, va_list *args)
281 {
282  MYSQL_STMT *stmt;
283  int params;
284 
285  stmt = mysql_stmt_init(mysql);
286  if (stmt == NULL)
287  {
288  DBG1(DBG_LIB, "creating MySQL statement failed: %s",
289  mysql_error(mysql));
290  return NULL;
291  }
292  if (mysql_stmt_prepare(stmt, sql, strlen(sql)))
293  {
294  DBG1(DBG_LIB, "preparing MySQL statement failed: %s",
295  mysql_stmt_error(stmt));
296  mysql_stmt_close(stmt);
297  return NULL;
298  }
299  params = mysql_stmt_param_count(stmt);
300  if (params > 0)
301  {
302  int i;
303  MYSQL_BIND *bind;
304 
305  bind = alloca(sizeof(MYSQL_BIND) * params);
306  memset(bind, 0, sizeof(MYSQL_BIND) * params);
307 
308  for (i = 0; i < params; i++)
309  {
310  switch (va_arg(*args, db_type_t))
311  {
312  case DB_INT:
313  {
314  bind[i].buffer_type = MYSQL_TYPE_LONG;
315  bind[i].buffer = (char*)alloca(sizeof(int));
316  *(int*)bind[i].buffer = va_arg(*args, int);
317  bind[i].buffer_length = sizeof(int);
318  break;
319  }
320  case DB_UINT:
321  {
322  bind[i].buffer_type = MYSQL_TYPE_LONG;
323  bind[i].buffer = (char*)alloca(sizeof(u_int));
324  *(u_int*)bind[i].buffer = va_arg(*args, u_int);
325  bind[i].buffer_length = sizeof(u_int);
326  bind[i].is_unsigned = TRUE;
327  break;
328  }
329  case DB_TEXT:
330  {
331  bind[i].buffer_type = MYSQL_TYPE_STRING;;
332  bind[i].buffer = va_arg(*args, char*);
333  if (bind[i].buffer)
334  {
335  bind[i].buffer_length = strlen(bind[i].buffer);
336  }
337  break;
338  }
339  case DB_BLOB:
340  {
341  chunk_t chunk = va_arg(*args, chunk_t);
342  bind[i].buffer_type = MYSQL_TYPE_BLOB;
343  bind[i].buffer = chunk.ptr;
344  bind[i].buffer_length = chunk.len;
345  break;
346  }
347  case DB_DOUBLE:
348  {
349  bind[i].buffer_type = MYSQL_TYPE_DOUBLE;
350  bind[i].buffer = (char*)alloca(sizeof(double));
351  *(double*)bind[i].buffer = va_arg(*args, double);
352  bind[i].buffer_length = sizeof(double);
353  break;
354  }
355  case DB_NULL:
356  {
357  bind[i].buffer_type = MYSQL_TYPE_NULL;
358  break;
359  }
360  default:
361  DBG1(DBG_LIB, "invalid data type supplied");
362  mysql_stmt_close(stmt);
363  return NULL;
364  }
365  }
366  if (mysql_stmt_bind_param(stmt, bind))
367  {
368  DBG1(DBG_LIB, "binding MySQL param failed: %s",
369  mysql_stmt_error(stmt));
370  mysql_stmt_close(stmt);
371  return NULL;
372  }
373  }
374  if (mysql_stmt_execute(stmt))
375  {
376  DBG1(DBG_LIB, "executing MySQL statement failed: %s",
377  mysql_stmt_error(stmt));
378  mysql_stmt_close(stmt);
379  return NULL;
380  }
381  return stmt;
382 }

440 static bool mysql_enumerator_enumerate(mysql_enumerator_t *this, ...)
441 {
442  int i, columns;
443  va_list args;
444 
445  columns = mysql_stmt_field_count(this->stmt);
446 
447  /* free/reset data set of previous call */
448  for (i = 0; i < columns; i++)
449  {
450  switch (this->bind[i].buffer_type)
451  {
452  case MYSQL_TYPE_STRING:
453  case MYSQL_TYPE_BLOB:
454  {
455  free(this->bind[i].buffer);
456  this->bind[i].buffer = NULL;
457  this->bind[i].buffer_length = 0;
458  this->bind[i].length = &this->length[i];
459  this->length[i] = 0;
460  break;
461  }
462  default:
463  break;
464  }
465  }
466 
467  switch (mysql_stmt_fetch(this->stmt))
468  {
469  case 0:
470  case MYSQL_DATA_TRUNCATED:
471  break;
472  case MYSQL_NO_DATA:
473  return FALSE;
474  default:
475  DBG1(DBG_LIB, "fetching MySQL row failed: %s",
476  mysql_stmt_error(this->stmt));
477  return FALSE;
478  }
479 
480  va_start(args, this);
481  for (i = 0; i < columns; i++)
482  {
483  switch (this->bind[i].buffer_type)
484  {
485  case MYSQL_TYPE_LONG:
486  {
487  if (this->bind[i].is_unsigned)
488  {
489  u_int *value = va_arg(args, u_int*);
490  *value = this->val.p_uint[i];
491  }
492  else
493  {
494  int *value = va_arg(args, int*);
495  *value = this->val.p_int[i];
496  }
497  break;
498  }
499  case MYSQL_TYPE_STRING:
500  {
501  char **value = va_arg(args, char**);
502  this->bind[i].buffer = malloc(this->length[i]+1);
503  this->bind[i].buffer_length = this->length[i];
504  *value = this->bind[i].buffer;
505  mysql_stmt_fetch_column(this->stmt, &this->bind[i], i, 0);
506  ((char*)this->bind[i].buffer)[this->length[i]] = '\0';
507  break;
508  }
509  case MYSQL_TYPE_BLOB:
510  {
511  chunk_t *value = va_arg(args, chunk_t*);
512  this->bind[i].buffer = malloc(this->length[i]);
513  this->bind[i].buffer_length = this->length[i];
514  value->ptr = this->bind[i].buffer;
515  value->len = this->length[i];
516  mysql_stmt_fetch_column(this->stmt, &this->bind[i], i, 0);
517  break;
518  }
519  case MYSQL_TYPE_DOUBLE:
520  {
521  double *value = va_arg(args, double*);
522  *value = this->val.p_double[i];
523  break;
524  }
525  default:
526  break;
527  }
528  }
529  va_end(args);
530  return TRUE;
531 }

702 static bool finalize_transaction(private_mysql_database_t *this,
703  bool rollback)
704 {
705  transaction_t *trans;
706  char *command = "COMMIT";
707  bool success;
708 
709  trans = this->transaction->get(this->transaction);
710  if (!trans)
711  {
712  DBG1(DBG_LIB, "no database transaction found");
713  return FALSE;
714  }
715  /* set flag, can't be unset */
716  trans->rollback |= rollback;
717 
718  if (ref_put(&trans->refs))
719  {
720  if (trans->rollback)
721  {
722  command = "ROLLBACK";
723  }
724  success = mysql_query(trans->conn->mysql, command) == 0;
725 
726  this->transaction->set(this->transaction, NULL);
727  transaction_destroy(this, trans);
728  return success;
729  }
730  return TRUE;
