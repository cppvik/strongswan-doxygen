33  if (!hasher)

35  return FALSE;
36  }
37 
38  ctx = EVP_MD_CTX_create();
39  if (!ctx)
40  {
41  goto error;
42  }
43 
44  if (!EVP_DigestInit_ex(ctx, hasher, NULL))
45  {
46  goto error;
47  }
48

50  {

73 bool openssl_bn_cat(int len, BIGNUM *a, BIGNUM *b, chunk_t *chunk)
74 {
