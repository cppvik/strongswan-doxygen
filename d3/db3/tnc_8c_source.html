103 static bool load_imcvs_from_config(char *filename, bool is_imc)
104 {
105  bool success = FALSE;
106  int line_nr = 0;
107  chunk_t *src, line;
108  char *label;
109 
110  if (!filename || !*filename)
111  {
112  return TRUE;
113  }
114 
115  label = is_imc ? "IMC" : "IMV";
116 
117  DBG1(DBG_TNC, "loading %ss from '%s'", label, filename);
118  src = chunk_map(filename, FALSE);
119  if (!src)
120  {
121  DBG1(DBG_TNC, "opening configuration file '%s' failed: %s", filename,
122  strerror(errno));
123  return FALSE;
124  }
125 
126  while (fetchline(src, &line))
127  {
128  char *name, *path;
129  chunk_t token;
130 
131  line_nr++;
132 
133  /* skip comments or empty lines */
134  if (*line.ptr == '#' || !eat_whitespace(&line))
135  {
136  continue;
137  }
138 
139  /* determine keyword */
140  if (!extract_token(&token, ' ', &line))
141  {
142  DBG1(DBG_TNC, "line %d: keyword must be followed by a space",
143  line_nr);
144  break;
145  }
146 
147  /* only interested in IMCs or IMVs depending on label */
148  if (!match(label, &token))
149  {
150  continue;
151  }
152 
153  /* advance to the IMC/IMV name and extract it */
154  if (!extract_token(&token, '"', &line) ||
155  !extract_token(&token, '"', &line))
156  {
157  DBG1(DBG_TNC, "line %d: %s name must be set in double quotes",
158  line_nr, label);
159  break;
160  }
161 
162  /* copy the IMC/IMV name */
163  name = strndup(token.ptr, token.len);
164 
165  /* advance to the IMC/IMV path and extract it */
166  if (!eat_whitespace(&line))
167  {
168  DBG1(DBG_TNC, "line %d: %s path is missing", line_nr, label);
169  free(name);
170  break;
171  }
172  if (!extract_token(&token, ' ', &line))
173  {
174  token = line;
175  }
176 
177  /* copy the IMC/IMV path */
178  path = strndup(token.ptr, token.len);
179 
180  /* load and register an IMC/IMV instance */
181  if (is_imc)
182  {
183  success = tnc->imcs->load(tnc->imcs, name, path);
184  }
185  else
186  {
187  success = tnc->imvs->load(tnc->imvs, name, path);
188  }
189  free(name);
190  free(path);
191  if (!success)
192  {
193  break;
194  }
195  }
196  chunk_unmap(src);
197  return success;
198 }
