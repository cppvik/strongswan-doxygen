106 static void calc_range(private_traffic_selector_t *this, u_int8_t netbits)
107 {
108  size_t len;
109  int bytes, bits;
110  u_int8_t mask;
111 
112  this->netbits = netbits;
113 
114  len = (this->type == TS_IPV4_ADDR_RANGE) ? 4 : 16;
115  bytes = (netbits + 7)/8;
116  bits = (bytes * 8) - netbits;
117  mask = bits ? (1 << bits) - 1 : 0;
118 
119  memcpy(this->to, this->from, bytes);
120  memset(this->from + bytes, 0x00, len - bytes);
121  memset(this->to + bytes, 0xff, len - bytes);
122  this->from[bytes-1] &= ~mask;
123  this->to[bytes-1] |= mask;
124 }

129 static u_int8_t calc_netbits(private_traffic_selector_t *this)
130 {
131  int byte, bit;
132  u_int8_t netbits;
133  size_t size = (this->type == TS_IPV4_ADDR_RANGE) ? 4 : 16;
134  bool prefix = TRUE;
135 
136  /* a perfect match results in a single address with a /32 or /128 netmask */
137  netbits = (size * 8);
138  this->netbits = netbits;
139 
140  /* go through all bits of the addresses, beginning in the front.
141  * as long as they are equal, the subnet gets larger
142  */
143  for (byte = 0; byte < size; byte++)
144  {
145  for (bit = 7; bit >= 0; bit--)
146  {
147  u_int8_t bitmask = 1 << bit;
148 
149  if (prefix)
150  {
151  if ((bitmask & this->from[byte]) != (bitmask & this->to[byte]))
152  {
153  /* store the common prefix which might be a true subnet */
154  netbits = (7 - bit) + (byte * 8);
155  this->netbits = netbits;
156  prefix = FALSE;
157  }
158  }
159  else
160  {
161  if ((bitmask & this->from[byte]) || !(bitmask & this->to[byte]))
162  {
163  this->netbits = NON_SUBNET_ADDRESS_RANGE;
164  return netbits; /* return a pseudo subnet */
165 
166  }
167  }
168  }
169  }
170  return netbits; /* return a true subnet */
171 }

282  has_ports = !is_any(this);

284  if (!has_proto && !has_ports)

322  written += print_icmp(data, this->from_port);
323  }
324  else

328  {
329  written += print_in_hook(data, "%s", serv->s_name);

331  else
332  {
333  written += print_in_hook(data, "%d", this->from_port);
334  }
335  }
336  }
337  else if (is_opaque(this))
338  {
339  written += print_in_hook(data, "OPAQUE");
340  }
341  else if (this->protocol == IPPROTO_ICMP ||
342  this->protocol == IPPROTO_ICMPV6)
343  {
344  written += print_icmp(data, this->from_port);
345  written += print_in_hook(data, "-");
346  written += print_icmp(data, this->to_port);
347  }
348  else
349  {
350  written += print_in_hook(data, "%d-%d",

352  }
353  }
354 
355  written += print_in_hook(data, "]");

360 METHOD(traffic_selector_t, get_subset, traffic_selector_t*,
361  private_traffic_selector_t *this, traffic_selector_t *other_public)
362 {
363  private_traffic_selector_t *other, *subset;
364  u_int16_t from_port, to_port;
365  u_char *from, *to;
366  u_int8_t protocol;
367  size_t size;
368 
369  other = (private_traffic_selector_t*)other_public;
370 
371  if (this->dynamic || other->dynamic)
372  { /* no set_address() applied, TS has no subset */
373  return NULL;
374  }
375 
376  if (this->type != other->type)
377  {
378  return NULL;
379  }
380  switch (this->type)
381  {
382  case TS_IPV4_ADDR_RANGE:
383  size = sizeof(this->from4);
384  break;
385  case TS_IPV6_ADDR_RANGE:
386  size = sizeof(this->from6);
387  break;
388  default:
389  return NULL;
390  }
391 
392  if (this->protocol != other->protocol &&
393  this->protocol != 0 && other->protocol != 0)
394  {
395  return NULL;
396  }
397  /* select protocol, which is not zero */
398  protocol = max(this->protocol, other->protocol);
399 
400  if ((is_opaque(this) && is_opaque(other)) ||
401  (is_opaque(this) && is_any(other)) ||
402  (is_opaque(other) && is_any(this)))
403  {
404  from_port = 0xffff;
405  to_port = 0;
406  }
407  else
408  {
409  /* calculate the maximum port range allowed for both */
410  from_port = max(this->from_port, other->from_port);
411  to_port = min(this->to_port, other->to_port);
412  if (from_port > to_port)
413  {
414  return NULL;
415  }
416  }
417  /* get higher from-address */
418  if (memcmp(this->from, other->from, size) > 0)
419  {
420  from = this->from;
421  }
422  else
423  {
424  from = other->from;
425  }
426  /* get lower to-address */
427  if (memcmp(this->to, other->to, size) > 0)
428  {
429  to = other->to;
430  }
431  else
432  {
433  to = this->to;
434  }
435  /* if "from" > "to", we don't have a match */
436  if (memcmp(from, to, size) > 0)
437  {
438  return NULL;
439  }
440 
441  /* we have a match in protocol, port, and address: return it... */
442  subset = traffic_selector_create(protocol, this->type, from_port, to_port);
443  memcpy(subset->from, from, size);
444  memcpy(subset->to, to, size);
445  calc_netbits(subset);
446 
447  return &subset->public;
448 }

612 METHOD(traffic_selector_t, to_subnet, bool,
613  private_traffic_selector_t *this, host_t **net, u_int8_t *mask)
614 {
615  /* there is no way to do this cleanly, as the address range may
616  * be anything else but a subnet. We use from_addr as subnet
617  * and try to calculate a usable subnet mask.
618  */
619  int family, non_zero_bytes;
620  u_int16_t port = 0;
621  chunk_t net_chunk;
622 
623  *mask = (this->netbits == NON_SUBNET_ADDRESS_RANGE) ? calc_netbits(this)
624  : this->netbits;
625 
626  switch (this->type)
627  {
628  case TS_IPV4_ADDR_RANGE:
629  family = AF_INET;
630  net_chunk.len = sizeof(this->from4);
631  break;
632  case TS_IPV6_ADDR_RANGE:
633  family = AF_INET6;
634  net_chunk.len = sizeof(this->from6);
635  break;
636  default:
637  /* unreachable */
638  return FALSE;
639  }
640 
641  net_chunk.ptr = malloc(net_chunk.len);
642  memset(net_chunk.ptr, 0x00, net_chunk.len);
643  if (*mask)
644  {
645  non_zero_bytes = (*mask + 7) / 8;
646  memcpy(net_chunk.ptr, this->from, non_zero_bytes);
647  net_chunk.ptr[non_zero_bytes-1] &= 0xFF << (8 * non_zero_bytes - *mask);
648  }
649 
650  if (this->to_port == this->from_port)
651  {
652  port = this->to_port;
653  }
654 
655  *net = host_create_from_chunk(family, net_chunk, port);
656  chunk_free(&net_chunk);
657 
658  return this->netbits != NON_SUBNET_ADDRESS_RANGE;
659 }

715 int traffic_selector_cmp(traffic_selector_t *a_pub, traffic_selector_t *b_pub,
716  void *opts)
717 {
718  private_traffic_selector_t *a, *b;
719  int res;
720 
721  a = (private_traffic_selector_t*)a_pub;
722  b = (private_traffic_selector_t*)b_pub;
723 
724  /* IPv4 before IPv6 */
725  res = compare_int(a->type, b->type);
726  if (res)
727  {
728  return res;
729  }
730  switch (a->type)
731  {
732  case TS_IPV4_ADDR_RANGE:
733  /* lower starting subnets first */
734  res = memcmp(a->from4, b->from4, sizeof(a->from4));
735  if (res)
736  {
737  return res;
738  }
739  /* larger subnets first */
740  res = memcmp(b->to4, a->to4, sizeof(a->to4));
741  if (res)
742  {
743  return res;
744  }
745  break;
746  case TS_IPV6_ADDR_RANGE:
747  res = memcmp(a->from6, b->from6, sizeof(a->from6));
748  if (res)
749  {
750  return res;
751  }
752  res = memcmp(b->to6, a->to6, sizeof(a->to6));
753  if (res)
754  {
755  return res;
756  }
757  break;
758  default:
759  return 1;
760  }
761  /* lower protocols first */
762  res = compare_int(a->protocol, b->protocol);
763  if (res)
764  {
765  return res;
766  }
767  /* lower starting ports first */
768  res = compare_int(a->from_port, b->from_port);
769  if (res)
770  {
771  return res;
772  }
773  /* larger port ranges first */
774  return compare_int(b->to_port, a->to_port);
775 }
