58  if (!this->salt)

60  return FALSE;
61  }
62  if (size < sizeof(u_int64_t))
63  {
64  return FALSE;
65  }
66  if (this->prevl != SEQ_IV_INIT_STATE && seq <= this->prevl)
67  {
68  seq |= SEQ_IV_HIGH_MASK;

71  return FALSE;
72  }
73  }

75  {
76  return FALSE;
77  }
78  if (seq & SEQ_IV_HIGH_MASK)
79  {
80  this->prevh = seq;
81  }
82  else

84  this->prevl = seq;
85  }
86  if (len > sizeof(u_int64_t))

89  memset(buffer, 0, size - len);
