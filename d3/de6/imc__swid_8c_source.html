75 TNC_Result TNC_IMC_NotifyConnectionChange(TNC_IMCID imc_id,
76  TNC_ConnectionID connection_id,
77  TNC_ConnectionState new_state)
78 {
79  imc_state_t *state;
80 
81  if (!imc_swid)
82  {
83  DBG1(DBG_IMC, "IMC \"%s\" has not been initialized", imc_name);
84  return TNC_RESULT_NOT_INITIALIZED;
85  }
86  switch (new_state)
87  {
88  case TNC_CONNECTION_STATE_CREATE:
89  state = imc_swid_state_create(connection_id);
90  return imc_swid->create_state(imc_swid, state);
91  case TNC_CONNECTION_STATE_HANDSHAKE:
92  if (imc_swid->change_state(imc_swid, connection_id, new_state,
93  &state) != TNC_RESULT_SUCCESS)
94  {
95  return TNC_RESULT_FATAL;
96  }
97  state->set_result(state, imc_id,
98  TNC_IMV_EVALUATION_RESULT_DONT_KNOW);
99  return TNC_RESULT_SUCCESS;
100  case TNC_CONNECTION_STATE_DELETE:
101  return imc_swid->delete_state(imc_swid, connection_id);
102  default:
103  return imc_swid->change_state(imc_swid, connection_id,
104  new_state, NULL);
105  }
106 }

111 TNC_Result TNC_IMC_BeginHandshake(TNC_IMCID imc_id,
112  TNC_ConnectionID connection_id)
113 {
114  imc_state_t *state;
115  imc_msg_t *out_msg;
116  pa_tnc_attr_t *attr;
117  seg_contract_t *contract;
118  seg_contract_manager_t *contracts;
119  size_t max_attr_size = SWID_MAX_ATTR_SIZE;
120  size_t max_seg_size;
121  char buf[BUF_LEN];
122  TNC_Result result = TNC_RESULT_SUCCESS;
123 
124  if (!imc_swid)
125  {
126  DBG1(DBG_IMC, "IMC \"%s\" has not been initialized", imc_name);
127  return TNC_RESULT_NOT_INITIALIZED;
128  }
129  if (!imc_swid->get_state(imc_swid, connection_id, &state))
130  {
131  return TNC_RESULT_FATAL;
132  }
133 
134  /* Determine maximum PA-TNC attribute segment size */
135  max_seg_size = state->get_max_msg_len(state) - PA_TNC_HEADER_SIZE
136  - PA_TNC_ATTR_HEADER_SIZE
137  - TCG_SEG_ATTR_SEG_ENV_HEADER;
138 
139  /* Announce support of PA-TNC segmentation to IMV */
140  contract = seg_contract_create(msg_types[0], max_attr_size, max_seg_size,
141  TRUE, imc_id, TRUE);
142  contract->get_info_string(contract, buf, BUF_LEN, TRUE);
143  DBG2(DBG_IMC, "%s", buf);
144  contracts = state->get_contracts(state);
145  contracts->add_contract(contracts, contract);
146  attr = tcg_seg_attr_max_size_create(max_attr_size, max_seg_size, TRUE);
147 
148  /* send PA-TNC message with the excl flag not set */
149  out_msg = imc_msg_create(imc_swid, state, connection_id, imc_id,
150  TNC_IMVID_ANY, msg_types[0]);
151  out_msg->add_attribute(out_msg, attr);
152  result = out_msg->send(out_msg, FALSE);
153  out_msg->destroy(out_msg);
154 
155  return result;
156 }

161 static bool add_swid_inventory(imc_state_t *state, imc_msg_t *msg,
162  uint32_t request_id, bool full_tags,
163  swid_inventory_t *targets)
164 {
165  pa_tnc_attr_t *attr, *attr_error;
166  imc_swid_state_t *swid_state;
167  swid_inventory_t *swid_inventory;
168  char *swid_directory, *swid_generator;
169  uint32_t eid_epoch;
170  bool swid_pretty, swid_full;
171  enumerator_t *enumerator;
172 
173  swid_directory = lib->settings->get_str(lib->settings,
174  "%s.plugins.imc-swid.swid_directory",
175  SWID_DIRECTORY, lib->ns);
176  swid_generator = lib->settings->get_str(lib->settings,
177  "%s.plugins.imc-swid.swid_generator",
178  SWID_GENERATOR, lib->ns);
179  swid_pretty = lib->settings->get_bool(lib->settings,
180  "%s.plugins.imc-swid.swid_pretty",
181  FALSE, lib->ns);
182  swid_full = lib->settings->get_bool(lib->settings,
183  "%s.plugins.imc-swid.swid_full",
184  FALSE, lib->ns);
185 
186  swid_inventory = swid_inventory_create(full_tags);
187  if (!swid_inventory->collect(swid_inventory, swid_directory, swid_generator,
188  targets, swid_pretty, swid_full))
189  {
190  swid_inventory->destroy(swid_inventory);
191  attr_error = swid_error_create(TCG_SWID_ERROR, request_id,
192  0, "error in SWID tag collection");
193  msg->add_attribute(msg, attr_error);
194  return FALSE;
195  }
196  DBG1(DBG_IMC, "collected %d SWID tag%s%s",
197  swid_inventory->get_count(swid_inventory), full_tags ? "" : " ID",
198  swid_inventory->get_count(swid_inventory) == 1 ? "" : "s");
199 
200  swid_state = (imc_swid_state_t*)state;
201  eid_epoch = swid_state->get_eid_epoch(swid_state);
202 
203  if (full_tags)
204  {
205  tcg_swid_attr_tag_inv_t *swid_attr;
206  swid_tag_t *tag;
207 
208  /* Send a TCG SWID Tag Inventory attribute */
209  attr = tcg_swid_attr_tag_inv_create(request_id, eid_epoch, 1);
210  swid_attr = (tcg_swid_attr_tag_inv_t*)attr;
211 
212  enumerator = swid_inventory->create_enumerator(swid_inventory);
213  while (enumerator->enumerate(enumerator, &tag))
214  {
215  swid_attr->add(swid_attr, tag->get_ref(tag));
216  }
217  enumerator->destroy(enumerator);
218  }
219  else
220  {
221  tcg_swid_attr_tag_id_inv_t *swid_id_attr;
222  swid_tag_id_t *tag_id;
223 
224  /* Send a TCG SWID Tag ID Inventory attribute */
225  attr = tcg_swid_attr_tag_id_inv_create(request_id, eid_epoch, 1);
226  swid_id_attr = (tcg_swid_attr_tag_id_inv_t*)attr;
227 
228  enumerator = swid_inventory->create_enumerator(swid_inventory);
229  while (enumerator->enumerate(enumerator, &tag_id))
230  {
231  swid_id_attr->add(swid_id_attr, tag_id->get_ref(tag_id));
232  }
233  enumerator->destroy(enumerator);
234  }
235 
236  msg->add_attribute(msg, attr);
237  swid_inventory->destroy(swid_inventory);
238 
239  return TRUE;
240 }

242 static TNC_Result receive_message(imc_state_t *state, imc_msg_t *in_msg)
243 {
244  imc_msg_t *out_msg;
245  pa_tnc_attr_t *attr;
246  enumerator_t *enumerator;
247  pen_type_t type;
248  TNC_Result result;
249  bool fatal_error = FALSE;
250 
251  /* generate an outgoing PA-TNC message - we might need it */
252  out_msg = imc_msg_create_as_reply(in_msg);
253 
254  /* parse received PA-TNC message and handle local and remote errors */
255  result = in_msg->receive(in_msg, out_msg, &fatal_error);
256  if (result != TNC_RESULT_SUCCESS)
257  {
258  out_msg->destroy(out_msg);
259  return result;
260  }
261 
262  /* analyze PA-TNC attributes */
263  enumerator = in_msg->create_attribute_enumerator(in_msg);
264  while (enumerator->enumerate(enumerator, &attr))
265  {
266  tcg_swid_attr_req_t *attr_req;
267  uint8_t flags;
268  uint32_t request_id;
269  bool full_tags;
270  swid_inventory_t *targets;
271 
272  type = attr->get_type(attr);
273 
274  if (type.vendor_id != PEN_TCG || type.type != TCG_SWID_REQUEST)
275  {
276  continue;
277  }
278 
279  attr_req = (tcg_swid_attr_req_t*)attr;
280  flags = attr_req->get_flags(attr_req);
281  request_id = attr_req->get_request_id(attr_req);
282  targets = attr_req->get_targets(attr_req);
283 
284  if (flags & (TCG_SWID_ATTR_REQ_FLAG_S | TCG_SWID_ATTR_REQ_FLAG_C))
285  {
286  attr = swid_error_create(TCG_SWID_SUBSCRIPTION_DENIED, request_id,
287  0, "no subscription available yet");
288  out_msg->add_attribute(out_msg, attr);
289  break;
290  }
291  full_tags = (flags & TCG_SWID_ATTR_REQ_FLAG_R) == 0;
292 
293  if (!add_swid_inventory(state, out_msg, request_id, full_tags, targets))
294  {
295  break;
296  }
297  }
298  enumerator->destroy(enumerator);
299 
300  if (fatal_error)
301  {
302  result = TNC_RESULT_FATAL;
303  }
304  else
305  {
306  /* send PA-TNC message with the EXCL flag set */
307  result = out_msg->send(out_msg, TRUE);
308  }
309  out_msg->destroy(out_msg);
310 
311  return result;
312 }
