172

176  id, auth);

178  {
179  if (public->verify(public, scheme, hash, sig))
180  {
181  DBG1(DBG_IKE, "authentication of '%Y' with %N successful",
182  id, key_type_names, this->type);
183  status = SUCCESS;
184  auth->merge(auth, current_auth, FALSE);
185  auth->add(auth, AUTH_RULE_AUTH_CLASS, AUTH_CLASS_PUBKEY);
186  break;
187  }
188  else
189  {
190  DBG1(DBG_IKE, "signature validation failed, looking for another key");
191  status = FAILED;
192  }
193  }
194  enumerator->destroy(enumerator);
195  free(hash.ptr);
196  if (status != SUCCESS)
197  {
198  DBG1(DBG_IKE, "no trusted %N public key found for '%Y'",
199  key_type_names, this->type, id);
200  }
