153

161  pcr_before = chunk_from_hex(
162  chunk_create(pcr_before_hex, strlen(pcr_before_hex)), NULL);

164  chunk_create(pcr_after_hex, strlen(pcr_after_hex)), NULL);

185 METHOD(pts_component_t, verify, status_t,
186  pts_ita_comp_tboot_t *this, u_int8_t qualifier,pts_t *pts,
187  pts_comp_evidence_t *evidence)
188 {
189  bool has_pcr_info;
190  u_int32_t extended_pcr, vid, name;
191  enum_name_t *names;
192  pts_meas_algorithms_t algo;
193  pts_pcr_transform_t transform;
194  pts_pcr_t *pcrs;
195  time_t measurement_time;
196  chunk_t measurement, pcr_before, pcr_after;
197  status_t status;
198 
199  this->aik_id = pts->get_aik_id(pts);
200  pcrs = pts->get_pcrs(pts);
201  measurement = evidence->get_measurement(evidence, &extended_pcr,
202  &algo, &transform, &measurement_time);
203 
204  status = this->pts_db->get_comp_measurement_count(this->pts_db,
205  this->name, this->aik_id, algo,
206  &this->cid, &this->count);
207  if (status != SUCCESS)
208  {
209  return status;
210  }
211  vid = this->name->get_vendor_id(this->name);
212  name = this->name->get_name(this->name);
213  names = imcv_pts_components->get_comp_func_names(imcv_pts_components, vid);
214 
215  if (this->count)
216  {
217  DBG1(DBG_PTS, "checking %d %N '%N' functional component evidence "
218  "measurements", this->count, pen_names, vid, names, name);
219  }
220  else
221  {
222  DBG1(DBG_PTS, "registering %N '%N' functional component evidence "
223  "measurements", pen_names, vid, names, name);
224  this->is_registering = TRUE;
225  }
226 
227  if (this->is_registering)
228  {
229  status = this->pts_db->insert_comp_measurement(this->pts_db,
230  measurement, this->cid, this->aik_id,
231  ++this->seq_no, extended_pcr, algo);
232  if (status != SUCCESS)
233  {
234  return status;
235  }
236  this->count = this->seq_no + 1;
237  }
238  else
239  {
240  status = this->pts_db->check_comp_measurement(this->pts_db,
241  measurement, this->cid, this->aik_id,
242  ++this->seq_no, extended_pcr, algo);
243  if (status != SUCCESS)
244  {
245  return status;
246  }
247  }
248 
249  has_pcr_info = evidence->get_pcr_info(evidence, &pcr_before, &pcr_after);
250  if (has_pcr_info)
251  {
252  if (!chunk_equals_const(pcr_before, pcrs->get(pcrs, extended_pcr)))
253  {
254  DBG1(DBG_PTS, "PCR %2u: pcr_before is not equal to register value",
255  extended_pcr);
256  }
257  if (pcrs->set(pcrs, extended_pcr, pcr_after))
258  {
259  return SUCCESS;
260  }
261  }
262 
263  return SUCCESS;
264 }

266 METHOD(pts_component_t, finalize, bool,

268 {

275 
276  snprintf(result_buf, BUF_LEN, "registered %d evidence measurements",
277  this->seq_no);
278  }
279  else if (this->seq_no < this->count)
280  {
281  snprintf(result_buf, BUF_LEN, "%d of %d evidence measurements "
282  "missing", this->count - this->seq_no, this->count);
283  return FALSE;
284  }
285  else
286  {
287  snprintf(result_buf, BUF_LEN, "%d evidence measurements are ok",
288  this->count);
289  }
290  DBG1(DBG_PTS, "%s", result_buf);
291  result->write_data(result, chunk_from_str(result_buf));
292 
293  return TRUE;
294 }
