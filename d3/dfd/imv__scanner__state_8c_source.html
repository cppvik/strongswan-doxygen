255 METHOD(imv_state_t, get_reason_string, bool,
256  private_imv_scanner_state_t *this, enumerator_t *language_enumerator,
257  chunk_t *reason_string, char **reason_language)
258 {
259  if (this->violating_ports->get_count(this->violating_ports) == 0)
260  {
261  return FALSE;
262  }
263  *reason_language = imv_lang_string_select_lang(language_enumerator,
264  languages, countof(languages));
265 
266  /* Instantiate a TNC Reason String object */
267  DESTROY_IF(this->reason_string);
268  this->reason_string = imv_reason_string_create(*reason_language, "\n");
269  if (this->rec != TNC_IMV_ACTION_RECOMMENDATION_NO_RECOMMENDATION)
270  {
271  this->reason_string->add_reason(this->reason_string, reasons);
272  }
273  *reason_string = this->reason_string->get_encoding(this->reason_string);
274 
275  return TRUE;
276 }
