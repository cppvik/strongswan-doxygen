80 static void generate_encoding(private_ntru_private_key_t *this)
81 {
82  size_t pubkey_len, privkey_len, privkey_trits_len, privkey_indices_len;
83  int privkey_pack_type;
84  uint16_t *indices;
85  uint8_t *trits;
86  u_char *enc;
87 
88  /* compute public key length encoded as packed coefficients */
89  pubkey_len = (this->params->N * this->params->q_bits + 7) / 8;
90 
91  /* compute private key length encoded as packed trits coefficients */
92  privkey_trits_len = (this->params->N + 4) / 5;
93 
94  /* compute private key length encoded as packed indices */
95  privkey_indices_len = (this->privkey->get_size(this->privkey) *
96  this->params->N_bits + 7) / 8;
97 
98  if (this->params->is_product_form ||
99  privkey_indices_len <= privkey_trits_len)
100  {
101  privkey_pack_type = NTRU_KEY_PACKED_INDICES;
102  privkey_len = privkey_indices_len;
103  }
104  else
105  {
106  privkey_pack_type = NTRU_KEY_PACKED_TRITS;
107  privkey_len = privkey_trits_len;
108  }
109 
110  /* allocate memory for private key encoding */
111  this->encoding = chunk_alloc(2 + NTRU_OID_LEN + pubkey_len + privkey_len);
112  enc = this->encoding.ptr;
113 
114  /* format header and packed public key */
115  *enc++ = NTRU_PRIVKEY_DEFAULT_TAG;
116  *enc++ = NTRU_OID_LEN;
117  memcpy(enc, this->params->oid, NTRU_OID_LEN);
118  enc += NTRU_OID_LEN;
119  ntru_elements_2_octets(this->params->N, this->pubkey,
120  this->params->q_bits, enc);
121  enc += pubkey_len;
122 
123  /* add packed private key */
124  indices = this->privkey->get_indices(this->privkey);
125 
126  if (privkey_pack_type == NTRU_KEY_PACKED_TRITS)
127  {
128  /* encode private key as packed trits */
129  trits = malloc(this->params->N);
130  ntru_indices_2_packed_trits(indices, this->params->dF_r,
131  this->params->dF_r, this->params->N, trits, enc);
132  memwipe(trits, this->params->N);
133  free(trits);
134  }
135  else
136  {
137  /* encode private key as packed indices */
138  ntru_elements_2_octets(this->privkey->get_size(this->privkey),
139  indices, this->params->N_bits, enc);
140  }

178 METHOD(ntru_private_key_t, decrypt, bool,
179  private_ntru_private_key_t *this, chunk_t ciphertext, chunk_t *plaintext)
180 {
181  hash_algorithm_t hash_algid;
182  size_t t_len, seed1_len, seed2_len;
183  uint16_t *t1, *t2, *t = NULL;
184  uint16_t mod_q_mask, q_mod_p, cmprime_len, cm_len = 0, num_zeros;
185  uint8_t *Mtrin, *M, *cm, *mask_trits, *ptr;
186  int16_t m1 = 0;
187  chunk_t seed = chunk_empty;
188  ntru_trits_t *mask;
189  ntru_poly_t *r_poly;
190  bool msg_rep_good, success = TRUE;
191  int i;
192 
193  *plaintext = chunk_empty;
194 
195  if (ciphertext.len != (this->params->N * this->params->q_bits + 7) / 8)
196  {
197  DBG1(DBG_LIB, "wrong NTRU ciphertext length");
198  return FALSE;
199  }
200 
201  /* allocate temporary array t */
202  t_len = 2 * this->params->N * sizeof(uint16_t);
203  t = malloc(t_len);
204  t1 = t;
205  t2 = t + this->params->N;
206  Mtrin = (uint8_t *)t1;
207  M = Mtrin + this->params->N;
208 
209  /* set hash algorithm based on security strength */
210  hash_algid = (this->params->sec_strength_len <= 20) ? HASH_SHA1 :
211  HASH_SHA256;
212 
213  /* set constants */
214  mod_q_mask = this->params->q - 1;
215  q_mod_p = this->params->q % 3;
216 
217  /* unpack the ciphertext */
218  ntru_octets_2_elements(ciphertext.len, ciphertext.ptr,
219  this->params->q_bits, t2);
220 
221  /* form cm':
222  * F * e
223  * A = e * (1 + pF) mod q = e + pFe mod q
224  * a = A in the range [-q/2, q/2)
225  * cm' = a mod p
226  */
227  this->privkey->ring_mult(this->privkey, t2, t1);
228 
229  cmprime_len = this->params->N;
230  if (this->params->is_product_form)
231  {
232  --cmprime_len;
233  for (i = 0; i < cmprime_len; i++)
234  {
235  t1[i] = (t2[i] + 3 * t1[i]) & mod_q_mask;
236  if (t1[i] >= (this->params->q / 2))
237  {
238  t1[i] -= q_mod_p;
239  }
240  Mtrin[i] = (uint8_t)(t1[i] % 3);
241  if (Mtrin[i] == 1)
242  {
243  ++m1;
244  }
245  else if (Mtrin[i] == 2)
246  {
247  --m1;
248  }
249  }
250  }
251  else
252  {
253  for (i = 0; i < cmprime_len; i++)
254  {
255  t1[i] = (t2[i] + 3 * t1[i]) & mod_q_mask;
256  if (t1[i] >= (this->params->q / 2))
257  {
258  t1[i] -= q_mod_p;
259  }
260  Mtrin[i] = (uint8_t)(t1[i] % 3);
261  }
262  }
263 
268  if (this->params->is_product_form)
269  {
270  msg_rep_good = (abs(m1) <= this->params->min_msg_rep_wt);
271  }
272  else
273  {
274  msg_rep_good = ntru_check_min_weight(cmprime_len, Mtrin,
275  this->params->min_msg_rep_wt);
276  }
277  if (!msg_rep_good)
278  {
279  DBG1(DBG_LIB, "decryption failed due to unsufficient minimum weight");
280  success = FALSE;
281  }
282 
283  /* form cR = e - cm' mod q */
284  for (i = 0; i < cmprime_len; i++)
285  {
286  if (Mtrin[i] == 1)
287  {
288  t2[i] = (t2[i] - 1) & mod_q_mask;
289  }
290  else if (Mtrin[i] == 2)
291  {
292  t2[i] = (t2[i] + 1) & mod_q_mask;
293  }
294  }
295  if (this->params->is_product_form)
296  {
297  t2[i] = (t2[i] + m1) & mod_q_mask;
298  }
299 
300  /* allocate memory for the larger of the two seeds */
301  seed1_len = (this->params->N + 3)/4;
302  seed2_len = 3 + 2*this->params->sec_strength_len + this->params->m_len_max;
303  seed = chunk_alloc(max(seed1_len, seed2_len));
304  seed.len = seed1_len;
305 
306  /* form cR mod 4 */
307  ntru_coeffs_mod4_2_octets(this->params->N, t2, seed.ptr);
308 
309  /* form mask */
310  mask = ntru_trits_create(this->params->N, hash_algid, seed);
311  if (!mask)
312  {
313  DBG1(DBG_LIB, "mask creation failed");
314  success = FALSE;
315  goto err;
316  }
317 
318  mask_trits = mask->get_trits(mask);
319 
320  /* form cMtrin by subtracting mask from cm', mod p */
321  for (i = 0; i < cmprime_len; i++)
322  {
323  Mtrin[i] -= mask_trits[i];
324  if (Mtrin[i] >= 3)
325  {
326  Mtrin[i] += 3;
327  }
328  }
329  mask->destroy(mask);
330 
331  if (this->params->is_product_form)
332  {
333  /* set the last trit to zero since that's what it was, and
334  * because it can't be calculated from (cm' - mask) since
335  * we don't have the correct value for the last cm' trit
336  */
337  Mtrin[i] = 0;
338  }
339 
340  /* convert cMtrin to cM (Mtrin to Mbin) */
341  if (!ntru_trits_2_bits(Mtrin, this->params->N, M))
342  {
343  success = FALSE;
344  goto err;
345  }
346 
347  /* skip the random padding */
348  ptr = M + this->params->sec_strength_len;
349 
350  /* validate the padded message cM and copy cm to m_buf */
351  if (this->params->m_len_len == 2)
352  {
353  cm_len = (uint16_t)(*ptr++) << 16;
354  }
355  cm_len |= (uint16_t)(*ptr++);
356 
357  if (cm_len > this->params->m_len_max)
358  {
359  cm_len = this->params->m_len_max;
360  DBG1(DBG_LIB, "NTRU message length is larger than maximum length");
361  success = FALSE;
362  }
363  cm = ptr;
364  ptr += cm_len;
365 
366  /* check if the remaining padding consists of zeros */
367  num_zeros = this->params->m_len_max - cm_len + 1;
368  for (i = 0; i < num_zeros; i++)
369  {
370  if (ptr[i] != 0)
371  {
372  DBG1(DBG_LIB, "non-zero trailing padding detected");
373  success = FALSE;
374  break;
375  }
376  }
377 
378  /* form sData (OID || m || b || hTrunc) */
379  ptr = seed.ptr;
380  memcpy(ptr, this->params->oid, 3);
381  ptr += 3;
382  memcpy(ptr, cm, cm_len);
383  ptr += cm_len;
384  memcpy(ptr, M, this->params->sec_strength_len);
385  ptr += this->params->sec_strength_len;
386  memcpy(ptr, this->encoding.ptr + 2 + NTRU_OID_LEN,
387  this->params->sec_strength_len);
388  ptr += this->params->sec_strength_len;
389  seed.len = ptr - seed.ptr;
390 
391  /* generate cr */
392  DBG2(DBG_LIB, "generate polynomial r");
393  r_poly = ntru_poly_create_from_seed(hash_algid, seed,
394  this->params->c_bits, this->params->N,
395  this->params->q, this->params->dF_r,
396  this->params->dF_r, this->params->is_product_form);
397  if (!r_poly)
398  {
399  success = FALSE;
400  goto err;
401  }
402 
403  /* output plaintext in allocated chunk */
404  *plaintext = chunk_clone(chunk_create(cm, cm_len));
405 
406  /* form cR' = h * cr */
407  r_poly->ring_mult(r_poly, this->pubkey, t1);
408  r_poly->destroy(r_poly);
409 
410  /* compare cR' to cR */
411  for (i = 0; i < this->params->N; i++)
412  {
413  if (t[i] != t2[i])
414  {
415  DBG1(DBG_LIB, "cR' does not equal cR'");
416  chunk_clear(plaintext);
417  success = FALSE;
418  break;
419  }
420  }
421  memwipe(t, t_len);
422 
423 err:
424  /* cleanup */
425  chunk_clear(&seed);
426  free(t);
427 
428  return success;
429 }

457 static void ring_mult_c(uint16_t *a, uint16_t *b, uint16_t N, uint16_t q,
458  uint16_t *c)
459 {
460  uint16_t *bptr = b;
461  uint16_t mod_q_mask = q - 1;
462  int i, k;
463 
464  /* c[k] = sum(a[i] * b[k-i]) mod q */
465  memset(c, 0, N * sizeof(uint16_t));
466  for (k = 0; k < N; k++)
467  {
468  i = 0;
469  while (i <= k)
470  {
471  c[k] += a[i++] * *bptr--;
472  }
473  bptr += N;
474  while (i < N)
475  {
476  c[k] += a[i++] * *bptr--;
477  }
478  c[k] &= mod_q_mask;
479  ++bptr;
480  }
481 }

495 static bool ring_inv(uint16_t *a, uint16_t N, uint16_t q, uint16_t *t,
496  uint16_t *a_inv)
497 {
498  uint8_t *b = (uint8_t *)t;
499  uint8_t *c = b + N;
500  uint8_t *f = c + N;
501  uint8_t *g = (uint8_t *)a_inv;
502  uint16_t *t2 = t + N;
503  uint16_t deg_b, deg_c, deg_f, deg_g;
504  bool done = FALSE;
505  int i, j, k = 0;
506 
507  /* form a^-1 in (Z/2Z)[X]/X^N - 1) */
508  memset(b, 0, 2 * N); /* clear to init b, c */
509 
510  /* b(X) = 1 */
511  b[0] = 1;
512  deg_b = 0;
513 
514  /* c(X) = 0 (cleared above) */
515  deg_c = 0;
516 
517  /* f(X) = a(X) mod 2 */
518  for (i = 0; i < N; i++)
519  {
520  f[i] = (uint8_t)(a[i] & 1);
521  }
522  deg_f = N - 1;
523 
524  /* g(X) = X^N - 1 */
525  g[0] = 1;
526  memset(g + 1, 0, N - 1);
527  g[N] = 1;
528  deg_g = N;
529 
530  /* until f(X) = 1 */
531  while (!done)
532  {
533  /* while f[0] = 0, f(X) /= X, c(X) *= X, k++ */
534  for (i = 0; (i <= deg_f) && (f[i] == 0); ++i);
535 
536  if (i > deg_f)
537  {
538  return FALSE;
539  }
540  if (i)
541  {
542  f = f + i;
543  deg_f = deg_f - i;
544  deg_c = deg_c + i;
545  for (j = deg_c; j >= i; j--)
546  {
547  c[j] = c[j-i];
548  }
549  for (j = 0; j < i; j++)
550  {
551  c[j] = 0;
552  }
553  k = k + i;
554  }
555 
556  /* adjust degree of f(X) if the highest coefficients are zero
557  * Note: f[0] = 1 from above so the loop will terminate.
558  */
559  while (f[deg_f] == 0)
560  {
561  --deg_f;
562  }
563 
564  /* if f(X) = 1, done
565  * Note: f[0] = 1 from above, so only check the x term and up
566  */
567  for (i = 1; (i <= deg_f) && (f[i] == 0); ++i);
568 
569  if (i > deg_f)
570  {
571  done = TRUE;
572  break;
573  }
574 
575  /* if deg_f < deg_g, f <-> g, b <-> c */
576  if (deg_f < deg_g)
577  {
578  uint8_t *x;
579 
580  x = f;
581  f = g;
582  g = x;
583  deg_f ^= deg_g;
584  deg_g ^= deg_f;
585  deg_f ^= deg_g;
586  x = b;
587  b = c;
588  c = x;
589  deg_b ^= deg_c;
590  deg_c ^= deg_b;
591  deg_b ^= deg_c;
592  }
593 
594  /* f(X) += g(X), b(X) += c(X) */
595  for (i = 0; i <= deg_g; i++)
596  {
597  f[i] ^= g[i];
598  }
599  if (deg_c > deg_b)
600  {
601  deg_b = deg_c;
602  }
603  for (i = 0; i <= deg_c; i++)
604  {
605  b[i] ^= c[i];
606  }
607  }
608 
609  /* a^-1 in (Z/2Z)[X]/(X^N - 1) = b(X) shifted left k coefficients */
610  j = 0;
611  if (k >= N)
612  {
613  k = k - N;
614  }
615  for (i = k; i < N; i++)
616  {
617  a_inv[j++] = (uint16_t)(b[i]);
618  }
619  for (i = 0; i < k; i++)
620  {
621  a_inv[j++] = (uint16_t)(b[i]);
622  }
623 
624  /* lift a^-1 in (Z/2Z)[X]/(X^N - 1) to a^-1 in (Z/qZ)[X]/(X^N -1) */
625  for (j = 0; j < 4; ++j) /* assumes 256 < q <= 65536 */
626  {
627  /* a^-1 = a^-1 * (2 - a * a^-1) mod q */
628  memcpy(t2, a_inv, N * sizeof(uint16_t));
629  ring_mult_c(a, t2, N, q, t);
630  for (i = 0; i < N; ++i)
631  {
632  t[i] = q - t[i];
633  }
634  t[0] = t[0] + 2;
635  ring_mult_c(t2, t, N, q, a_inv);
636  }
637 
638  return TRUE;
639 }

668 
669  /* set hash algorithm and seed length based on security strength */
670  if (params->sec_strength_len <= 20)
671  {
672  hash_algid = HASH_SHA1;
673  }
674  else
675  {
676  hash_algid = HASH_SHA256;
677  }

685  }

688  this->privkey = ntru_poly_create_from_seed(hash_algid, seed, params->c_bits,
689  params->N, params->q,
690  params->dF_r, params->dF_r,
691  params->is_product_form);
692  if (!this->privkey)
693  {
694  goto err;
695  }
696 
697  /* allocate temporary array t */
698  t_len = 3 * params->N * sizeof(uint16_t);
699  t = malloc(t_len);
700  t1 = t + 2 * params->N;
701 
702  /* extend sparse private key polynomial f to N array elements */
703  this->privkey->get_array(this->privkey, t1);
704 
705  /* set mask for large modulus */
706  mod_q_mask = params->q - 1;
707 
708  /* form f = 1 + pF */
709  for (i = 0; i < params->N; i++)
710  {
711  t1[i] = (t1[i] * 3) & mod_q_mask;
712  }
713  t1[0] = (t1[0] + 1) & mod_q_mask;
714 
715  /* use the public key array as a temporary buffer */
716  t2 = this->pubkey;
717 
718  /* find f^-1 in (Z/qZ)[X]/(X^N - 1) */
719  if (!ring_inv(t1, params->N, params->q, t, t2))
720  {
721  goto err;
722  }
723 
724  /* get random seed for generating trinary g as a list of indices */
725  if (!drbg->generate(drbg, params->sec_strength_len * BITS_PER_BYTE,
726  seed.len, seed.ptr))
727  {
728  goto err;
729  }
730 
731  DBG2(DBG_LIB, "generate polynomial g");
732  g_poly = ntru_poly_create_from_seed(hash_algid, seed, params->c_bits,
733  params->N, params->q, params->dg + 1,
734  params->dg, FALSE);
735  if (!g_poly)
736  {

770 ntru_private_key_t *ntru_private_key_create_from_data(ntru_drbg_t *drbg,
771  chunk_t data)
772 {
773  private_ntru_private_key_t *this;

775  size_t privkey_packed_trits_len, privkey_packed_indices_len;
776  uint8_t *privkey_packed, tag;
777  uint16_t *indices, dF;
778  ntru_param_set_t *params;
779 
780  header_len = 2 + NTRU_OID_LEN;
781 
782  /* check the NTRU public key header format */
783  if (data.len < header_len ||
784  !(data.ptr[0] == NTRU_PRIVKEY_DEFAULT_TAG ||
785  data.ptr[0] == NTRU_PRIVKEY_TRITS_TAG ||
786  data.ptr[0] == NTRU_PRIVKEY_INDICES_TAG) ||
787  data.ptr[1] != NTRU_OID_LEN)
788  {
789  DBG1(DBG_LIB, "loaded NTRU private key with invalid header");
790  return NULL;
791  }
792  tag = data.ptr[0];
793  params = ntru_param_set_get_by_oid(data.ptr + 2);
794 
795  if (!params)
796  {
797  DBG1(DBG_LIB, "loaded NTRU private key with unknown OID");
798  return NULL;
799  }
800 
801  pubkey_packed_len = (params->N * params->q_bits + 7) / 8;
802  privkey_packed_trits_len = (params->N + 4) / 5;
803 
804  /* check packing type for product-form private keys */
805  if (params->is_product_form && tag == NTRU_PRIVKEY_TRITS_TAG)
806  {
807  DBG1(DBG_LIB, "a product-form NTRU private key cannot be trits-encoded");
808  return NULL;
809  }
810 
811  /* set packed-key length for packed indices */
812  if (params->is_product_form)
813  {
814  dF = (uint16_t)((params->dF_r & 0xff) + /* df1 */
815  ((params->dF_r >> 8) & 0xff) + /* df2 */
816  ((params->dF_r >> 16) & 0xff)); /* df3 */
817  }
818  else
819  {
820  dF = (uint16_t)params->dF_r;
821  }
822  privkey_packed_indices_len = (2 * dF * params->N_bits + 7) / 8;
823 
824  /* set private-key packing type if defaulted */
825  if (tag == NTRU_PRIVKEY_DEFAULT_TAG)
826  {
827  if (params->is_product_form ||
828  privkey_packed_indices_len <= privkey_packed_trits_len)
829  {
830  tag = NTRU_PRIVKEY_INDICES_TAG;
831  }
832  else
833  {
834  tag = NTRU_PRIVKEY_TRITS_TAG;
835  }
836  }
837  privkey_packed_len = (tag == NTRU_PRIVKEY_TRITS_TAG) ?
838  privkey_packed_trits_len : privkey_packed_indices_len;
839 
840  if (data.len < header_len + pubkey_packed_len + privkey_packed_len)
841  {
842  DBG1(DBG_LIB, "loaded NTRU private key with wrong packed key size");
843  return NULL;
844  }
845 
846  INIT(this,
847  .public = {
848  .get_id = _get_id,
849  .get_public_key = _get_public_key,
850  .get_encoding = _get_encoding,
851  .decrypt = _decrypt,
852  .destroy = _destroy,
853  },
854  .params = params,
855  .pubkey = malloc(params->N * sizeof(uint16_t)),
856  .encoding = chunk_clone(data),
857  .drbg = drbg->get_ref(drbg),
858  );
859 
860  /* unpack the encoded public key */
861  ntru_octets_2_elements(pubkey_packed_len, data.ptr + header_len,
862  params->q_bits, this->pubkey);
863 
864  /* allocate temporary memory for indices */
865  indices = malloc(2 * dF * sizeof(uint16_t));
866 
867  /* unpack the private key */
868  privkey_packed = data.ptr + header_len + pubkey_packed_len;
869  if (tag == NTRU_PRIVKEY_TRITS_TAG)
870  {
871  ntru_packed_trits_2_indices(privkey_packed, params->N,
872  indices, indices + dF);
873  }
874  else
875  {
876  ntru_octets_2_elements(privkey_packed_indices_len, privkey_packed,
877  params->N_bits, indices);
878  }
879  this->privkey = ntru_poly_create_from_data(indices, params->N, params->q,
880  params->dF_r, params->dF_r,
881  params->is_product_form);
882 
883  /* cleanup */
884  memwipe(indices, 2 * dF * sizeof(uint16_t));
885  free(indices);
886 
887  return &this->public;
888 }
