91 static int request(private_mconsole_t *this, void(*cb)(void*,char*,size_t),
92  void *data, char *command, ...)
93 {
94  mconsole_request request;
95  mconsole_reply reply;
96  int len, flags = 0;
97  va_list args;
98 
99  memset(&request, 0, sizeof(request));
100  request.magic = MCONSOLE_MAGIC;
101  request.version = MCONSOLE_VERSION;
102  va_start(args, command);
103  request.len = vsnprintf(request.data, sizeof(request.data), command, args);
104  va_end(args);
105 
106  if (this->idle)
107  {
108  flags = MSG_DONTWAIT;
109  }
110  do
111  {
112  if (this->idle)
113  {
114  this->idle();
115  }
116  len = sendto(this->console, &request, sizeof(request), flags,
117  (struct sockaddr*)&this->uml, sizeof(this->uml));
118  }
119  while (len < 0 && (errno == EINTR || errno == EAGAIN));
120 
121  if (len < 0)
122  {
123  DBG1(DBG_LIB, "sending mconsole command to UML failed: %m");
124  return -1;
125  }
126  do
127  {
128  len = recv(this->console, &reply, sizeof(reply), flags);
129  if (len < 0 && (errno == EINTR || errno == EAGAIN))
130  {
131  if (this->idle)
132  {
133  this->idle();
134  }
135  continue;
136  }
137  if (len < 0)
138  {
139  DBG1(DBG_LIB, "receiving from mconsole failed: %m");
140  return -1;
141  }
142  if (len > 0)
143  {
144  if (cb)
145  {
146  cb(data, reply.data, reply.len);
147  }
148  else if (reply.err)
149  {
150  if (reply.len && *reply.data)
151  {
152  DBG1(DBG_LIB, "received mconsole error %d: %.*s",
153  reply.err, (int)reply.len, reply.data);
154  }
155  break;
156  }
157  }
158  }
159  while (reply.more);
160 
161  return reply.err;
162 }

231 static bool wait_for_notify(private_mconsole_t *this, char *nsock)
232 {
233  struct sockaddr_un addr;
234  mconsole_notify notify;
235  int len, flags = 0;
236 
237  this->notify = socket(AF_UNIX, SOCK_DGRAM, 0);
238  if (this->notify < 0)
239  {
240  DBG1(DBG_LIB, "opening mconsole notify socket failed: %m");
241  return FALSE;
242  }
243  memset(&addr, 0, sizeof(addr));
244  addr.sun_family = AF_UNIX;
245  strncpy(addr.sun_path, nsock, sizeof(addr.sun_path));
246  if (bind(this->notify, (struct sockaddr*)&addr, sizeof(addr)) < 0)
247  {
248  DBG1(DBG_LIB, "binding mconsole notify socket to '%s' failed: %m",
249  nsock);
250  close(this->notify);
251  return FALSE;
252  }
253  if (this->idle)
254  {
255  flags = MSG_DONTWAIT;
256  }
257  do
258  {
259  if (this->idle)
260  {
261  this->idle();
262  }
263  len = recvfrom(this->notify, &notify, sizeof(notify), flags, NULL, 0);
264  }
265  while (len < 0 && (errno == EINTR || errno == EAGAIN));
266 
267  if (len < 0 || len >= sizeof(notify))
268  {
269  DBG1(DBG_LIB, "reading from mconsole notify socket failed: %m");
270  close(this->notify);
271  unlink(nsock);
272  return FALSE;
273  }
274  if (notify.magic != MCONSOLE_MAGIC ||
275  notify.version != MCONSOLE_VERSION ||
276  notify.type != MCONSOLE_SOCKET)
277  {
278  DBG1(DBG_LIB, "received unexpected message from mconsole notify"
279  " socket: %b", &notify, sizeof(notify));
280  close(this->notify);
281  unlink(nsock);
282  return FALSE;
283  }
284  memset(&this->uml, 0, sizeof(this->uml));
285  this->uml.sun_family = AF_UNIX;
286  strncpy(this->uml.sun_path, (char*)&notify.data, sizeof(this->uml.sun_path));
287  return TRUE;
288 }
