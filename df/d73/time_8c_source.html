31 time_t time_monotonic(timeval_t *tv)
32 {
33 #ifdef WIN32
34  ULONGLONG ms;
35  time_t s;
36 
37  ms = GetTickCount64();
38  s = ms / 1000;
39  if (tv)
40  {
41  tv->tv_sec = s;
42  tv->tv_usec = (ms - (s * 1000)) * 1000;
43  }
44  return s;
45 #else /* !WIN32 */
46 #if defined(HAVE_CLOCK_GETTIME) && \
47  (defined(HAVE_CONDATTR_CLOCK_MONOTONIC) || \
48  defined(HAVE_PTHREAD_COND_TIMEDWAIT_MONOTONIC))
49  /* as we use time_monotonic() for condvar operations, we use the
50  * monotonic time source only if it is also supported by pthread. */
51  timespec_t ts;
52 
53  if (clock_gettime(CLOCK_MONOTONIC, &ts) == 0)
54  {
55  if (tv)
56  {
57  tv->tv_sec = ts.tv_sec;
58  tv->tv_usec = ts.tv_nsec / 1000;
59  }
60  return ts.tv_sec;
61  }
62 #endif /* HAVE_CLOCK_GETTIME && (...) */
63  /* Fallback to non-monotonic timestamps:
64  * On MAC OS X, creating monotonic timestamps is rather difficult. We
65  * could use mach_absolute_time() and catch sleep/wakeup notifications.
66  * We stick to the simpler (non-monotonic) gettimeofday() for now.
67  * But keep in mind: we need the same time source here as in condvar! */
68  if (!tv)
69  {
70  return time(NULL);
71  }
72  if (gettimeofday(tv, NULL) != 0)
73  { /* should actually never fail if passed pointers are valid */
74  return -1;
75  }
76  return tv->tv_sec;
77 #endif /* !WIN32 */
78 }
