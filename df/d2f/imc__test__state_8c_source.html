150 METHOD(imc_state_t, set_result, void,
151  private_imc_test_state_t *this, TNC_IMCID id,
152  TNC_IMV_Evaluation_Result result)
153 {
154  enumerator_t *enumerator;
155  entry_t *entry;
156  bool found = FALSE;
157 
158  enumerator = this->results->create_enumerator(this->results);
159  while (enumerator->enumerate(enumerator, &entry))
160  {
161  if (entry->id == id)
162  {
163  entry->result = result;
164  found = TRUE;
165  break;
166  }
167  }
168  enumerator->destroy(enumerator);
169 
170  if (!found)
171  {
172  entry = malloc_thing(entry_t);
173  entry->id = id;
174  entry->result = result;
175  this->results->insert_last(this->results, entry);
176  }
177 }
