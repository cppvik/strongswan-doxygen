23 void memxor(u_int8_t dst[], u_int8_t src[], size_t n)
24 {
25  int m, i;
26 
27  /* byte wise XOR until dst aligned */
28  for (i = 0; (uintptr_t)&dst[i] % sizeof(long) && i < n; i++)
29  {
30  dst[i] ^= src[i];
31  }
32  /* try to use words if src shares an aligment with dst */
33  switch (((uintptr_t)&src[i] % sizeof(long)))
34  {
35  case 0:
36  for (m = n - sizeof(long); i <= m; i += sizeof(long))
37  {
38  *(long*)&dst[i] ^= *(long*)&src[i];
39  }
40  break;
41  case sizeof(int):
42  for (m = n - sizeof(int); i <= m; i += sizeof(int))
43  {
44  *(int*)&dst[i] ^= *(int*)&src[i];
45  }
46  break;
47  case sizeof(short):
48  for (m = n - sizeof(short); i <= m; i += sizeof(short))
49  {
50  *(short*)&dst[i] ^= *(short*)&src[i];
51  }
52  break;
53  default:
54  break;
55  }
56  /* byte wise XOR of the rest */
57  for (; i < n; i++)
58  {
59  dst[i] ^= src[i];
60  }
61 }
