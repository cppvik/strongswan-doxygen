58 static status_t read_swid_tags(private_swid_inventory_t *this, FILE *file)
59 {
60  swid_tag_t *tag;
61  bio_writer_t *writer;
62  chunk_t tag_encoding, tag_file_path = chunk_empty;
63  bool more_tags = TRUE, last_newline;
64  char line[8192];
65  size_t len;
66 
67  while (more_tags)
68  {
69  last_newline = TRUE;
70  writer = bio_writer_create(512);
71  while (TRUE)
72  {
73  if (!fgets(line, sizeof(line), file))
74  {
75  more_tags = FALSE;
76  break;
77  }
78  len = strlen(line);
79 
80  if (last_newline && line[0] == '\n')
81  {
82  break;
83  }
84  else
85  {
86  last_newline = (line[len-1] == '\n');
87  writer->write_data(writer, chunk_create(line, len));
88  }
89  }
90 
91  tag_encoding = writer->get_buf(writer);
92 
93  if (tag_encoding.len > 1)
94  {
95  /* remove trailing newline if present */
96  if (tag_encoding.ptr[tag_encoding.len - 1] == '\n')
97  {
98  tag_encoding.len--;
99  }
100  DBG3(DBG_IMC, " %.*s", tag_encoding.len, tag_encoding.ptr);
101 
102  tag = swid_tag_create(tag_encoding, tag_file_path);
103  this->list->insert_last(this->list, tag);
104  }
105  writer->destroy(writer);
106  }
107 
108  return SUCCESS;
109 }

114 static status_t read_swid_tag_ids(private_swid_inventory_t *this, FILE *file)
115 {
116  swid_tag_id_t *tag_id;
117  chunk_t tag_creator, unique_sw_id, tag_file_path = chunk_empty;
118  char line[BUF_LEN];
119 
120  while (TRUE)
121  {
122  char *separator;
123  size_t len;
124 
125  if (!fgets(line, sizeof(line), file))
126  {
127  return SUCCESS;
128  }
129  len = strlen(line);
130 
131  /* remove trailing newline if present */
132  if (len > 0 && line[len - 1] == '\n')
133  {
134  len--;
135  }
136  DBG3(DBG_IMC, " %.*s", len, line);
137 
138  separator = strchr(line, '_');
139  if (!separator)
140  {
141  DBG1(DBG_IMC, "separation of regid from unique software ID failed");
142  return FAILED;
143  }
144  tag_creator = chunk_create(line, separator - line);
145  separator++;
146 
147  unique_sw_id = chunk_create(separator, len - (separator - line));
148  tag_id = swid_tag_id_create(tag_creator, unique_sw_id, tag_file_path);
149  this->list->insert_last(this->list, tag_id);
150  }

153 static status_t generate_tags(private_swid_inventory_t *this, char *generator,
154  swid_inventory_t *targets, bool pretty, bool full)
155 {
156  FILE *file;
157  char command[BUF_LEN];
158  char doc_separator[] = "'\n\n'";
159 
160  status_t status = SUCCESS;
161 
162  if (targets->get_count(targets) == 0)
163  {
164  /* Assemble the SWID generator command */
165  if (this->full_tags)
166  {
167  snprintf(command, BUF_LEN, "%s swid --doc-separator %s%s%s",
168  generator, doc_separator, pretty ? " --pretty" : "",
169  full ? " --full" : "");
170  }
171  else
172  {
173  snprintf(command, BUF_LEN, "%s software-id", generator);
174  }
175 
176  /* Open a pipe stream for reading the SWID generator output */
177  file = popen(command, "r");
178  if (!file)
179  {
180  DBG1(DBG_IMC, "failed to run swid_generator command");
181  return NOT_SUPPORTED;
182  }
183 
184  if (this->full_tags)
185  {
186  DBG2(DBG_IMC, "SWID tag generation by package manager");
187  status = read_swid_tags(this, file);
188  }
189  else
190  {
191  DBG2(DBG_IMC, "SWID tag ID generation by package manager");
192  status = read_swid_tag_ids(this, file);
193  }
194  pclose(file);
195  }
196  else if (this->full_tags)
197  {
198  swid_tag_id_t *tag_id;
199  enumerator_t *enumerator;
200 
201  enumerator = targets->create_enumerator(targets);
202  while (enumerator->enumerate(enumerator, &tag_id))
203  {
204  char software_id[BUF_LEN];
205  chunk_t tag_creator, unique_sw_id;
206 
207  tag_creator = tag_id->get_tag_creator(tag_id);
208  unique_sw_id = tag_id->get_unique_sw_id(tag_id, NULL);
209  snprintf(software_id, BUF_LEN, "%.*s_%.*s",
210  tag_creator.len, tag_creator.ptr,
211  unique_sw_id.len, unique_sw_id.ptr);
212 
213  /* Assemble the SWID generator command */
214  snprintf(command, BUF_LEN, "%s swid --software-id %s%s%s",
215  generator, software_id, pretty ? " --pretty" : "",
216  full ? " --full" : "");
217 
218  /* Open a pipe stream for reading the SWID generator output */
219  file = popen(command, "r");
220  if (!file)
221  {
222  DBG1(DBG_IMC, "failed to run swid_generator command");
223  return NOT_SUPPORTED;
224  }
225  status = read_swid_tags(this, file);
226  pclose(file);
227 
228  if (status != SUCCESS)
229  {
230  break;
231  }
232  }
233  enumerator->destroy(enumerator);
234  }
235 
236  return status;
237 }

239 static bool collect_tags(private_swid_inventory_t *this, char *pathname,
240  swid_inventory_t *targets)
241 {
242  char *rel_name, *abs_name;
243  struct stat st;
244  bool success = FALSE;
245  enumerator_t *enumerator;
246 
247  enumerator = enumerator_create_directory(pathname);
248  if (!enumerator)
249  {
250  DBG1(DBG_IMC, "directory '%s' can not be opened, %s",
251  pathname, strerror(errno));
252  return FALSE;
253  }
254  DBG2(DBG_IMC, "entering %s", pathname);
255 
256  while (enumerator->enumerate(enumerator, &rel_name, &abs_name, &st))
257  {
258  char * start, *stop;
259  chunk_t tag_creator;
260  chunk_t unique_sw_id = chunk_empty, tag_file_path = chunk_empty;
261 
262  if (!strstr(rel_name, "regid."))
263  {
264  continue;
265  }
266  if (S_ISDIR(st.st_mode))
267  {
268  /* In case of a targeted request */
269  if (targets->get_count(targets))
270  {
271  enumerator_t *target_enumerator;
272  swid_tag_id_t *tag_id;
273  bool match = FALSE;
274 
275  target_enumerator = targets->create_enumerator(targets);
276  while (target_enumerator->enumerate(target_enumerator, &tag_id))
277  {
278  if (chunk_equals(tag_id->get_tag_creator(tag_id),
279  chunk_from_str(rel_name)))
280  {
281  match = TRUE;
282  break;
283  }
284  }
285  target_enumerator->destroy(target_enumerator);
286 
287  if (!match)
288  {
289  continue;
290  }
291  }
292 
293  if (!collect_tags(this, abs_name, targets))
294  {
295  goto end;
296  }
297  continue;
298  }
299 
300  /* parse the regid filename into its components */
301  start = rel_name;
302  stop = strchr(start, '_');
303  if (!stop)
304  {
305  DBG1(DBG_IMC, " %s", rel_name);
306  DBG1(DBG_IMC, " '_' separator not found");
307  goto end;
308  }
309  tag_creator = chunk_create(start, stop-start);
310  start = stop + 1;
311 
312  stop = strstr(start, ".swidtag");
313  if (!stop)
314  {
315  DBG1(DBG_IMC, " %s", rel_name);
316  DBG1(DBG_IMC, " swidtag postfix not found");
317  goto end;
318  }
319  unique_sw_id = chunk_create(start, stop-start);
320  tag_file_path = chunk_from_str(abs_name);
321 
322  /* In case of a targeted request */
323  if (targets->get_count(targets))
324  {
325  chunk_t target_unique_sw_id, target_tag_creator;
326  enumerator_t *target_enumerator;
327  swid_tag_id_t *tag_id;
328  bool match = FALSE;
329 
330  target_enumerator = targets->create_enumerator(targets);
331  while (target_enumerator->enumerate(target_enumerator, &tag_id))
332  {
333  target_unique_sw_id = tag_id->get_unique_sw_id(tag_id, NULL);
334  target_tag_creator = tag_id->get_tag_creator(tag_id);
335 
336  if (chunk_equals(target_unique_sw_id, unique_sw_id) &&
337  chunk_equals(target_tag_creator, tag_creator))
338  {
339  match = TRUE;
340  break;
341  }
342  }
343  target_enumerator->destroy(target_enumerator);
344 
345  if (!match)
346  {
347  continue;
348  }
349  }
350  DBG2(DBG_IMC, " %s", rel_name);
351 
352  if (this->full_tags)
353  {
354  swid_tag_t *tag;
355  chunk_t *xml_tag;
356 
357  xml_tag = chunk_map(abs_name, FALSE);
358  if (!xml_tag)
359  {
360  DBG1(DBG_IMC, " opening '%s' failed: %s", abs_name,
361  strerror(errno));
362  goto end;
363  }
364 
365  tag = swid_tag_create(*xml_tag, tag_file_path);
366  this->list->insert_last(this->list, tag);
367  chunk_unmap(xml_tag);
368  }
369  else
370  {
371  swid_tag_id_t *tag_id;
372 
373  tag_id = swid_tag_id_create(tag_creator, unique_sw_id, tag_file_path);
374  this->list->insert_last(this->list, tag_id);
375  }
376  }
377  success = TRUE;
378 
379 end:
380  enumerator->destroy(enumerator);
381  DBG2(DBG_IMC, "leaving %s", pathname);
382 
383  return success;
384 }
