102 static bool derive_ike_aead(private_keymat_v2_t *this, u_int16_t alg,
103  u_int16_t key_size, prf_plus_t *prf_plus)
104 {
105  aead_t *aead_i, *aead_r;
106  chunk_t key = chunk_empty;
107  u_int salt_size;
108 
109  switch (alg)
110  {
111  case ENCR_AES_GCM_ICV8:
112  case ENCR_AES_GCM_ICV12:
113  case ENCR_AES_GCM_ICV16:
114  /* RFC 4106 */
115  case ENCR_CHACHA20_POLY1305:
116  salt_size = 4;
117  break;
118  case ENCR_AES_CCM_ICV8:
119  case ENCR_AES_CCM_ICV12:
120  case ENCR_AES_CCM_ICV16:
121  /* RFC 4309 */
122  case ENCR_CAMELLIA_CCM_ICV8:
123  case ENCR_CAMELLIA_CCM_ICV12:
124  case ENCR_CAMELLIA_CCM_ICV16:
125  /* RFC 5529 */
126  salt_size = 3;
127  break;
128  default:
129  DBG1(DBG_IKE, "nonce size for %N unknown!",
130  encryption_algorithm_names, alg);
131  return FALSE;
132  }
133 
134  /* SK_ei/SK_er used for encryption */
135  aead_i = lib->crypto->create_aead(lib->crypto, alg, key_size / 8, salt_size);
136  aead_r = lib->crypto->create_aead(lib->crypto, alg, key_size / 8, salt_size);
137  if (aead_i == NULL || aead_r == NULL)
138  {
139  DBG1(DBG_IKE, "%N %N (key size %d) not supported!",
140  transform_type_names, ENCRYPTION_ALGORITHM,
141  encryption_algorithm_names, alg, key_size);
142  goto failure;
143  }
144  key_size = aead_i->get_key_size(aead_i);
145  if (key_size != aead_r->get_key_size(aead_r))
146  {
147  goto failure;
148  }
149  if (!prf_plus->allocate_bytes(prf_plus, key_size, &key))
150  {
151  goto failure;
152  }
153  DBG4(DBG_IKE, "Sk_ei secret %B", &key);
154  if (!aead_i->set_key(aead_i, key))
155  {
156  goto failure;
157  }
158  chunk_clear(&key);
159 
160  if (!prf_plus->allocate_bytes(prf_plus, key_size, &key))
161  {
162  goto failure;
163  }
164  DBG4(DBG_IKE, "Sk_er secret %B", &key);
165  if (!aead_r->set_key(aead_r, key))
166  {
167  goto failure;
168  }
169 
170  if (this->initiator)
171  {
172  this->aead_in = aead_r;
173  this->aead_out = aead_i;
174  }
175  else
176  {
177  this->aead_in = aead_i;
178  this->aead_out = aead_r;
179  }
180  aead_i = aead_r = NULL;
181 
182 failure:
183  DESTROY_IF(aead_i);
184  DESTROY_IF(aead_r);
185  chunk_clear(&key);
186  return this->aead_in && this->aead_out;
187 }

285  signer_i = signer_r = NULL;
286  crypter_i = crypter_r = NULL;
287 
288 failure:
289  chunk_clear(&key);
290  DESTROY_IF(signer_i);
291  DESTROY_IF(signer_r);

297 METHOD(keymat_v2_t, derive_ike_keys, bool,
298  private_keymat_v2_t *this, proposal_t *proposal, diffie_hellman_t *dh,
299  chunk_t nonce_i, chunk_t nonce_r, ike_sa_id_t *id,
300  pseudo_random_function_t rekey_function, chunk_t rekey_skd)
301 {
302  chunk_t skeyseed, key, secret, full_nonce, fixed_nonce, prf_plus_seed;
303  chunk_t spi_i, spi_r;
304  prf_plus_t *prf_plus = NULL;
305  u_int16_t alg, key_size, int_alg;
306  prf_t *rekey_prf = NULL;
307 
308  spi_i = chunk_alloca(sizeof(u_int64_t));
309  spi_r = chunk_alloca(sizeof(u_int64_t));
310 
311  if (!dh->get_shared_secret(dh, &secret))
312  {
313  return FALSE;
314  }
315 
316  /* Create SAs general purpose PRF first, we may use it here */
317  if (!proposal->get_algorithm(proposal, PSEUDO_RANDOM_FUNCTION, &alg, NULL))
318  {
319  DBG1(DBG_IKE, "no %N selected",
320  transform_type_names, PSEUDO_RANDOM_FUNCTION);
321  chunk_clear(&secret);
322  return FALSE;
323  }
324  this->prf_alg = alg;
325  this->prf = lib->crypto->create_prf(lib->crypto, alg);
326  if (this->prf == NULL)
327  {
328  DBG1(DBG_IKE, "%N %N not supported!",
329  transform_type_names, PSEUDO_RANDOM_FUNCTION,
330  pseudo_random_function_names, alg);
331  chunk_clear(&secret);
332  return FALSE;
333  }
334  DBG4(DBG_IKE, "shared Diffie Hellman secret %B", &secret);
335  /* full nonce is used as seed for PRF+ ... */
336  full_nonce = chunk_cat("cc", nonce_i, nonce_r);
337  /* but the PRF may need a fixed key which only uses the first bytes of
338  * the nonces. */
339  switch (alg)
340  {
341  case PRF_AES128_XCBC:
342  /* while rfc4434 defines variable keys for AES-XCBC, rfc3664 does
343  * not and therefore fixed key semantics apply to XCBC for key
344  * derivation. */
345  case PRF_CAMELLIA128_XCBC:
346  /* draft-kanno-ipsecme-camellia-xcbc refers to rfc 4434, we
347  * assume fixed key length. */
348  key_size = this->prf->get_key_size(this->prf)/2;
349  nonce_i.len = min(nonce_i.len, key_size);
350  nonce_r.len = min(nonce_r.len, key_size);
351  break;
352  default:
353  /* all other algorithms use variable key length, full nonce */
354  break;
355  }
356  fixed_nonce = chunk_cat("cc", nonce_i, nonce_r);
357  *((u_int64_t*)spi_i.ptr) = id->get_initiator_spi(id);
358  *((u_int64_t*)spi_r.ptr) = id->get_responder_spi(id);
359  prf_plus_seed = chunk_cat("ccc", full_nonce, spi_i, spi_r);
360 
361  /* KEYMAT = prf+ (SKEYSEED, Ni | Nr | SPIi | SPIr)
362  *
363  * if we are rekeying, SKEYSEED is built on another way
364  */
365  if (rekey_function == PRF_UNDEFINED) /* not rekeying */
366  {
367  /* SKEYSEED = prf(Ni | Nr, g^ir) */
368  if (this->prf->set_key(this->prf, fixed_nonce) &&
369  this->prf->allocate_bytes(this->prf, secret, &skeyseed) &&
370  this->prf->set_key(this->prf, skeyseed))
371  {
372  prf_plus = prf_plus_create(this->prf, TRUE, prf_plus_seed);
373  }
374  }
375  else
376  {
377  /* SKEYSEED = prf(SK_d (old), [g^ir (new)] | Ni | Nr)
378  * use OLD SAs PRF functions for both prf_plus and prf */
379  rekey_prf = lib->crypto->create_prf(lib->crypto, rekey_function);
380  if (!rekey_prf)
381  {
382  DBG1(DBG_IKE, "PRF of old SA %N not supported!",
383  pseudo_random_function_names, rekey_function);
384  chunk_clear(&secret);
385  chunk_free(&full_nonce);
386  chunk_free(&fixed_nonce);
387  chunk_clear(&prf_plus_seed);
388  return FALSE;
389  }
390  secret = chunk_cat("mc", secret, full_nonce);
391  if (rekey_prf->set_key(rekey_prf, rekey_skd) &&
392  rekey_prf->allocate_bytes(rekey_prf, secret, &skeyseed) &&
393  rekey_prf->set_key(rekey_prf, skeyseed))
394  {
395  prf_plus = prf_plus_create(rekey_prf, TRUE, prf_plus_seed);
396  }
397  }
398  DBG4(DBG_IKE, "SKEYSEED %B", &skeyseed);
399 
400  chunk_clear(&skeyseed);
401  chunk_clear(&secret);
402  chunk_free(&full_nonce);
403  chunk_free(&fixed_nonce);
404  chunk_clear(&prf_plus_seed);
405 
406  if (!prf_plus)
407  {
408  goto failure;
409  }
410 
411  /* KEYMAT = SK_d | SK_ai | SK_ar | SK_ei | SK_er | SK_pi | SK_pr */
412 
413  /* SK_d is used for generating CHILD_SA key mat => store for later use */
414  key_size = this->prf->get_key_size(this->prf);
415  if (!prf_plus->allocate_bytes(prf_plus, key_size, &this->skd))
416  {
417  goto failure;
418  }
419  DBG4(DBG_IKE, "Sk_d secret %B", &this->skd);
420 
421  if (!proposal->get_algorithm(proposal, ENCRYPTION_ALGORITHM, &alg, &key_size))
422  {
423  DBG1(DBG_IKE, "no %N selected",
424  transform_type_names, ENCRYPTION_ALGORITHM);
425  goto failure;
426  }
427 
428  if (encryption_algorithm_is_aead(alg))
429  {
430  if (!derive_ike_aead(this, alg, key_size, prf_plus))
431  {
432  goto failure;
433  }
434  }
435  else
436  {
437  if (!proposal->get_algorithm(proposal, INTEGRITY_ALGORITHM,
438  &int_alg, NULL))
439  {
440  DBG1(DBG_IKE, "no %N selected",
441  transform_type_names, INTEGRITY_ALGORITHM);
442  goto failure;
443  }
444  if (!derive_ike_traditional(this, alg, key_size, int_alg, prf_plus))
445  {
446  goto failure;
447  }
448  }
449 
450  /* SK_pi/SK_pr used for authentication => stored for later */
451  key_size = this->prf->get_key_size(this->prf);
452  if (!prf_plus->allocate_bytes(prf_plus, key_size, &key))
453  {
454  goto failure;
455  }
456  DBG4(DBG_IKE, "Sk_pi secret %B", &key);
457  if (this->initiator)
458  {
459  this->skp_build = key;
460  }
461  else
462  {
463  this->skp_verify = key;
464  }
465  if (!prf_plus->allocate_bytes(prf_plus, key_size, &key))
466  {
467  goto failure;
468  }
469  DBG4(DBG_IKE, "Sk_pr secret %B", &key);
470  if (this->initiator)
471  {
472  this->skp_verify = key;
473  }
474  else
475  {
476  this->skp_build = key;
477  }
478 
479  /* all done, prf_plus not needed anymore */
480 failure:
481  DESTROY_IF(prf_plus);
482  DESTROY_IF(rekey_prf);
483 
484  return this->skp_build.len && this->skp_verify.len;
485 }

487 METHOD(keymat_v2_t, derive_child_keys, bool,
488  private_keymat_v2_t *this, proposal_t *proposal, diffie_hellman_t *dh,
489  chunk_t nonce_i, chunk_t nonce_r, chunk_t *encr_i, chunk_t *integ_i,
490  chunk_t *encr_r, chunk_t *integ_r)
491 {
492  u_int16_t enc_alg, int_alg, enc_size = 0, int_size = 0;
493  chunk_t seed, secret = chunk_empty;
494  prf_plus_t *prf_plus;
495 
496  if (proposal->get_algorithm(proposal, ENCRYPTION_ALGORITHM,
497  &enc_alg, &enc_size))
498  {
499  DBG2(DBG_CHD, " using %N for encryption",

501

507  {

511  }
512  /* to bytes */
513  enc_size /= 8;
514 
515  /* CCM/GCM/CTR/GMAC needs additional bytes */
516  switch (enc_alg)
517  {
518  case ENCR_AES_CCM_ICV8:
519  case ENCR_AES_CCM_ICV12:
520  case ENCR_AES_CCM_ICV16:
521  case ENCR_CAMELLIA_CCM_ICV8:
522  case ENCR_CAMELLIA_CCM_ICV12:
523  case ENCR_CAMELLIA_CCM_ICV16:
524  enc_size += 3;

526  case ENCR_AES_GCM_ICV8:
527  case ENCR_AES_GCM_ICV12:
528  case ENCR_AES_GCM_ICV16:
529  case ENCR_AES_CTR:
530  case ENCR_CAMELLIA_CTR:
531  case ENCR_NULL_AUTH_AES_GMAC:
532  case ENCR_CHACHA20_POLY1305:
533  enc_size += 4;
534  break;
535  default:
536  break;
537  }
538  }
539 
540  if (proposal->get_algorithm(proposal, INTEGRITY_ALGORITHM,
541  &int_alg, &int_size))
542  {
543  DBG2(DBG_CHD, " using %N for integrity",
544  integrity_algorithm_names, int_alg);
545 
546  if (!int_size)
547  {
548  int_size = keymat_get_keylen_integ(int_alg);
549  }
550  if (!int_size)
551  {
552  DBG1(DBG_CHD, "no keylength defined for %N",
553  integrity_algorithm_names, int_alg);
554  return FALSE;
555  }
556  /* to bytes */
557  int_size /= 8;
558  }
559 
560  if (!this->prf->set_key(this->prf, this->skd))
561  {
562  return FALSE;
563  }
564 
565  if (dh)
566  {
567  if (!dh->get_shared_secret(dh, &secret))
568  {
569  return FALSE;
570  }
571  DBG4(DBG_CHD, "DH secret %B", &secret);
572  }
573  seed = chunk_cata("scc", secret, nonce_i, nonce_r);
574  DBG4(DBG_CHD, "seed %B", &seed);
575 
576  prf_plus = prf_plus_create(this->prf, TRUE, seed);
577  memwipe(seed.ptr, seed.len);
578 
579  if (!prf_plus)
580  {
581  return FALSE;
582  }
583 
584  *encr_i = *integ_i = *encr_r = *integ_r = chunk_empty;
585  if (!prf_plus->allocate_bytes(prf_plus, enc_size, encr_i) ||
586  !prf_plus->allocate_bytes(prf_plus, int_size, integ_i) ||
587  !prf_plus->allocate_bytes(prf_plus, enc_size, encr_r) ||
588  !prf_plus->allocate_bytes(prf_plus, int_size, integ_r))
589  {
590  chunk_free(encr_i);
591  chunk_free(integ_i);
592  chunk_free(encr_r);
593  chunk_free(integ_r);
594  prf_plus->destroy(prf_plus);
595  return FALSE;
596  }
597 
598  prf_plus->destroy(prf_plus);
599 
600  if (enc_size)
601  {
602  DBG4(DBG_CHD, "encryption initiator key %B", encr_i);
603  DBG4(DBG_CHD, "encryption responder key %B", encr_r);
604  }
605  if (int_size)
606  {
607  DBG4(DBG_CHD, "integrity initiator key %B", integ_i);
608  DBG4(DBG_CHD, "integrity responder key %B", integ_r);
609  }
610  return TRUE;
611 }
