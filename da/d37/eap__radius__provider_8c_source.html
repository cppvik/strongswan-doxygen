180 static host_t* remove_addr(private_eap_radius_provider_t *this,
181  hashtable_t *hashtable, uintptr_t id, host_t *addr)
182 {
183  enumerator_t *enumerator;
184  entry_t *entry;
185  host_t *found = NULL, *current;
186 
187  entry = hashtable->remove(hashtable, (void*)id);
188  if (entry)
189  {
190  enumerator = entry->addrs->create_enumerator(entry->addrs);
191  while (enumerator->enumerate(enumerator, &current))
192  {
193  if (addr->ip_equals(addr, current))
194  { /* prefer an exact match */
195  entry->addrs->remove_at(entry->addrs, enumerator);
196  enumerator->destroy(enumerator);
197  put_or_destroy_entry(hashtable, entry);
198  return current;
199  }
200  if (!found && addr->get_family(addr) == current->get_family(current))
201  { /* fallback to the first IP with a matching address family */
202  found = current;
203  }
204  }
205  enumerator->destroy(enumerator);
206  if (found)
207  {
208  entry->addrs->remove(entry->addrs, found, NULL);
209  }
210  put_or_destroy_entry(hashtable, entry);
211  }
212  return found;
213 }
