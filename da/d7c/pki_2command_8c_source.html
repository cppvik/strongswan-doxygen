253 int command_dispatch(int c, char *v[])
254 {
255  int op, i;
256 
257  options = options_create();
258  atexit(cleanup);
259  active = help_idx = registered;
260  argc = c;
261  argv = v;
262  command_register((command_t){help, 'h', "help", "show usage information"});
263 
264  build_opts();
265  op = getopt_long(c, v, command_optstring, command_opts, NULL);
266  for (i = 0; i < MAX_COMMANDS && cmds[i].cmd; i++)
267  {
268  if (cmds[i].op == op)
269  {
270  active = i;
271  build_opts();
272  return cmds[i].call();
273  }
274  }
275  return command_usage(c > 1 ? "invalid operation" : NULL);
276 }
