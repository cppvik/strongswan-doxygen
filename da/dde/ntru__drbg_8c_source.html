144 METHOD(ntru_drbg_t, generate, bool,
145  private_ntru_drbg_t *this, u_int32_t strength, u_int32_t len, u_int8_t *out)
146 {
147  size_t delta;
148  chunk_t output;
149 
150  DBG2(DBG_LIB, "DRBG generates %u pseudorandom bytes", len);
151  if (!out || len == 0)
152  {
153  return FALSE;
154  }
155  output = chunk_create(out, len);
156 
157  if (this->reseed_counter > this->max_requests)
158  {
159  if (!reseed(this))
160  {
161  return FALSE;
162  }
163  }
164  while (len)
165  {
166  if (!this->hmac->get_signature(this->hmac, this->value,
167  this->value.ptr))
168  {
169  return FALSE;
170  }
171  delta = min(len, this->value.len);
172  memcpy(out, this->value.ptr, delta);
173  len -= delta;
174  out += delta;
175  }
176  DBG4(DBG_LIB, "HMAC_DRBG Out: %B", &output);
177 
178  if (!update(this, chunk_empty))
179  {
180  return FALSE;
181  }
182  this->reseed_counter++;
183 
184  return TRUE;

253  .reseed = _reseed,
254  .generate = _generate,
255  .get_ref = _get_ref,
256  .destroy = _destroy,
257  },
258  .strength = strength,
259  .entropy = entropy,
260  .hmac = hmac,
261  .key = chunk_alloc(hmac->get_key_size(hmac)),

273  DBG2(DBG_LIB, "DRBG requests %u bytes of entropy", entropy_len);
274
