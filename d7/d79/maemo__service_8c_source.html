165 static void load_ca_dir(private_maemo_service_t *this, char *dir)
166 {
167  enumerator_t *enumerator;
168  char *rel, *abs;
169  struct stat st;
170 
171  enumerator = enumerator_create_directory(dir);
172  if (enumerator)
173  {
174  while (enumerator->enumerate(enumerator, &rel, &abs, &st))
175  {
176  if (rel[0] != '.')
177  {
178  if (S_ISREG(st.st_mode))
179  {
180  certificate_t *cert;
181  cert = lib->creds->create(lib->creds, CRED_CERTIFICATE,
182  CERT_X509, BUILD_FROM_FILE, abs,
183  BUILD_END);
184  if (!cert)
185  {
186  DBG1(DBG_CFG, "loading CA certificate '%s' failed",
187  abs);
188  continue;
189  }
190  DBG2(DBG_CFG, "loaded CA certificate '%Y'",
191  cert->get_subject(cert));
192  this->creds->add_cert(this->creds, TRUE, cert);
193  }
194  }
195  }
196  enumerator->destroy(enumerator);
197  }
198 }

200 static void disconnect(private_maemo_service_t *this)
201 {
202  ike_sa_t *ike_sa;
203  u_int id;
204 
205  if (!this->current)
206  {
207  return;
208  }
209 
210  /* avoid status updates, as this is called from the Glib main loop */
211  charon->bus->remove_listener(charon->bus, &this->public.listener);
212 
213  ike_sa = charon->ike_sa_manager->checkout_by_name(charon->ike_sa_manager,
214  this->current, FALSE);
215  if (ike_sa)
216  {
217  id = ike_sa->get_unique_id(ike_sa);
218  charon->ike_sa_manager->checkin(charon->ike_sa_manager, ike_sa);
219  charon->controller->terminate_ike(charon->controller, id,
220  NULL, NULL, 0);
221  }
222  this->current = (g_free(this->current), NULL);
223  this->status = VPN_STATUS_DISCONNECTED;
224 }

226 static gboolean initiate_connection(private_maemo_service_t *this,
227  GArray *arguments)
228 {
229  gint i;
230  gchar *hostname = NULL, *cacert = NULL, *username = NULL, *password = NULL;
231  identification_t *gateway = NULL, *user = NULL;
232  ike_sa_t *ike_sa;
233  ike_cfg_t *ike_cfg;
234  peer_cfg_t *peer_cfg;
235  child_cfg_t *child_cfg;
236  traffic_selector_t *ts;
237  auth_cfg_t *auth;
238  certificate_t *cert;
239  lifetime_cfg_t lifetime = {
240  .time = {
241  .life = 10800, /* 3h */
242  .rekey = 10200, /* 2h50min */
243  .jitter = 300 /* 5min */
244  }
245  };
246 
247  if (this->status == VPN_STATUS_CONNECTED ||
248  this->status == VPN_STATUS_CONNECTING)
249  {
250  DBG1(DBG_CFG, "currently connected to '%s', disconnecting first",
251  this->current);
252  disconnect (this);
253  }
254 
255  if (arguments->len != 5)
256  {
257  DBG1(DBG_CFG, "wrong number of arguments: %d", arguments->len);
258  return FALSE;
259  }
260 
261  for (i = 0; i < arguments->len; i++)
262  {
263  osso_rpc_t *arg = &g_array_index(arguments, osso_rpc_t, i);
264  if (arg->type != DBUS_TYPE_STRING)
265  {
266  DBG1(DBG_CFG, "invalid argument [%d]: %d", i, arg->type);
267  return FALSE;
268  }
269  switch (i)
270  {
271  case 0: /* name */
272  this->current = (g_free(this->current), NULL);
273  this->current = g_strdup(arg->value.s);
274  break;
275  case 1: /* hostname */
276  hostname = arg->value.s;
277  break;
278  case 2: /* CA certificate path */
279  cacert = arg->value.s;
280  break;
281  case 3: /* username */
282  username = arg->value.s;
283  break;
284  case 4: /* password */
285  password = arg->value.s;
286  break;
287  }
288  }
289 
290  DBG1(DBG_CFG, "received initiate for connection '%s'", this->current);
291 
292  this->creds->clear(this->creds);
293 
294  if (cacert && !streq(cacert, ""))
295  {
296  cert = lib->creds->create(lib->creds, CRED_CERTIFICATE, CERT_X509,
297  BUILD_FROM_FILE, cacert, BUILD_END);
298  if (cert)
299  {
300  this->creds->add_cert(this->creds, TRUE, cert);
301  }
302  else
303  {
304  DBG1(DBG_CFG, "failed to load CA certificate");
305  }
306  /* if this is a server cert we could use the cert subject as id */
307  }
308  else
309  {
310  load_ca_dir(this, MAEMO_COMMON_CA_DIR);
311  load_ca_dir(this, MAEMO_USER_CA_DIR);
312  }
313 
314  gateway = identification_create_from_string(hostname);
315  DBG1(DBG_CFG, "using CA certificate, gateway identitiy '%Y'", gateway);
316 
317  {
318  shared_key_t *shared_key;
319  chunk_t secret = chunk_create(password, strlen(password));
320  user = identification_create_from_string(username);
321  shared_key = shared_key_create(SHARED_EAP, chunk_clone(secret));
322  this->creds->add_shared(this->creds, shared_key, user->clone(user),
323  NULL);
324  }
325 
326  ike_cfg = ike_cfg_create(IKEV2, TRUE, FALSE, "0.0.0.0",
327  charon->socket->get_port(charon->socket, FALSE),
328  hostname, IKEV2_UDP_PORT, FRAGMENTATION_NO, 0);
329  ike_cfg->add_proposal(ike_cfg, proposal_create_default(PROTO_IKE));
330  ike_cfg->add_proposal(ike_cfg, proposal_create_default_aead(PROTO_IKE));
331 
332  peer_cfg = peer_cfg_create(this->current, ike_cfg,
333  CERT_SEND_IF_ASKED,
334  UNIQUE_REPLACE, 1, /* keyingtries */
335  36000, 0, /* rekey 10h, reauth none */
336  600, 600, /* jitter, over 10min */
337  TRUE, FALSE, TRUE, /* mobike, aggressive, pull */
338  0, 0, /* DPD delay, timeout */
339  FALSE, NULL, NULL); /* mediation */
340  peer_cfg->add_virtual_ip(peer_cfg, host_create_from_string("0.0.0.0", 0));
341 
342  auth = auth_cfg_create();
343  auth->add(auth, AUTH_RULE_AUTH_CLASS, AUTH_CLASS_EAP);
344  auth->add(auth, AUTH_RULE_IDENTITY, user);
345  peer_cfg->add_auth_cfg(peer_cfg, auth, TRUE);
346  auth = auth_cfg_create();
347  auth->add(auth, AUTH_RULE_AUTH_CLASS, AUTH_CLASS_PUBKEY);
348  auth->add(auth, AUTH_RULE_IDENTITY, gateway);
349  peer_cfg->add_auth_cfg(peer_cfg, auth, FALSE);
350 
351  child_cfg = child_cfg_create(this->current, &lifetime, NULL /* updown */,
352  TRUE, MODE_TUNNEL, ACTION_NONE, ACTION_NONE,
353  ACTION_NONE, FALSE, 0, 0, NULL, NULL, 0);
354  child_cfg->add_proposal(child_cfg, proposal_create_default(PROTO_ESP));
355  child_cfg->add_proposal(child_cfg, proposal_create_default_aead(PROTO_ESP));
356  ts = traffic_selector_create_dynamic(0, 0, 65535);
357  child_cfg->add_traffic_selector(child_cfg, TRUE, ts);
358  ts = traffic_selector_create_from_string(0, TS_IPV4_ADDR_RANGE, "0.0.0.0",
359  0, "255.255.255.255", 65535);
360  child_cfg->add_traffic_selector(child_cfg, FALSE, ts);
361  peer_cfg->add_child_cfg(peer_cfg, child_cfg);
362 
363  /* get us an IKE_SA */
364  ike_sa = charon->ike_sa_manager->checkout_by_config(charon->ike_sa_manager,
365  peer_cfg);
366  if (!ike_sa)
367  {
368  peer_cfg->destroy(peer_cfg);
369  this->status = VPN_STATUS_CONNECTION_FAILED;
370  return FALSE;
371  }
372  if (!ike_sa->get_peer_cfg(ike_sa))
373  {
374  ike_sa->set_peer_cfg(ike_sa, peer_cfg);
375  }
376  peer_cfg->destroy(peer_cfg);
377 
378  /* store the IKE_SA, so we can track its progress */
379  this->ike_sa = ike_sa;
380  this->status = VPN_STATUS_CONNECTING;
381  this->public.listener.ike_updown = _ike_updown;
382  this->public.listener.ike_state_change = _ike_state_change;
383  charon->bus->add_listener(charon->bus, &this->public.listener);
384 
385  /* get an additional reference because initiate consumes one */
386  child_cfg->get_ref(child_cfg);
387  if (ike_sa->initiate(ike_sa, child_cfg, 0, NULL, NULL) != SUCCESS)
388  {
389  DBG1(DBG_CFG, "failed to initiate tunnel");
390  charon->bus->remove_listener(charon->bus, &this->public.listener);
391  charon->ike_sa_manager->checkin_and_destroy(charon->ike_sa_manager,
392  ike_sa);
393  this->status = VPN_STATUS_CONNECTION_FAILED;
394  return FALSE;
395  }
396  charon->ike_sa_manager->checkin(charon->ike_sa_manager, ike_sa);
397  return TRUE;
398 }

403 static gint dbus_req_handler(const gchar *interface, const gchar *method,
404  GArray *arguments, private_maemo_service_t *this,
405  osso_rpc_t *retval)
406 {
407  if (streq(method, "Start"))
408  { /* void start (void), dummy function to start charon as root */
409  return OSSO_OK;
410  }
411  else if (streq(method, "Connect"))
412  { /* bool connect (name, host, cert, user, pass) */
413  retval->value.b = initiate_connection(this, arguments);
414  retval->type = DBUS_TYPE_BOOLEAN;
415  }
416  else if (streq(method, "Disconnect"))
417  { /* void disconnect (void) */
418  disconnect(this);
419  }
420  else
421  {
422  return OSSO_ERROR;
423  }
424  return OSSO_OK;
425 }
